84  public sealed class MethodScorable : ScorableBase<IResolver, Binding, Binding>
85  {
86  private readonly MethodInfo method;
87  private readonly ParameterInfo[] parameters;
88  public MethodScorable(MethodInfo method)
89  {
90  SetField.NotNull(out this.method, nameof(method), method);
91  this.parameters = this.method.GetParameters();
92  }
93  public override string ToString()
94  {
95  return $"{this.GetType().Name}({method})";
96  }
97  private bool TryResolveInstance(IResolver resolver, out object instance)
98  {
99  if (this.method.IsStatic)
100  {
101  instance = null;
102  return true;
103  }
104 
105  return resolver.TryResolve(this.method.DeclaringType, null, out instance);
106  }
107  private static bool TryResolveArgument(IResolver resolver, ParameterInfo parameter, out object argument)
108  {
109  var entity = parameter.GetCustomAttribute<EntityAttribute>();
110  if (entity != null)
111  {
112  if (resolver.TryResolve(parameter.ParameterType, entity.Entity, out argument))
113  {
114  return true;
115  }
116  }
117 
118  if (resolver.TryResolve(parameter.ParameterType, parameter.Name, out argument))
119  {
120  return true;
121  }
122 
123  return resolver.TryResolve(parameter.ParameterType, null, out argument);
124  }
125  private bool TryResolveArguments(IResolver resolver, out object[] arguments)
126  {
127  if (this.parameters.Length == 0)
128  {
129  arguments = Array.Empty<object>();
130  return true;
131  }
132 
133  arguments = null;
134  for (int index = 0; index < this.parameters.Length; ++index)
135  {
136  var parameter = this.parameters[index];
137 
138  object argument;
139 
140  if (!TryResolveArgument(resolver, parameter, out argument))
141  {
142  arguments = null;
143  return false;
144  }
145 
146  if (arguments == null)
147  {
148  arguments = new object[this.parameters.Length];
149  }
150 
151  arguments[index] = argument;
152  }
153 
154  return arguments != null;
155  }
156 
157  protected override Task<Binding> PrepareAsync(IResolver item, CancellationToken token)
158  {
159  try
160  {
161  object instance;
162  if (!TryResolveInstance(item, out instance))
163  {
164  return Tasks<Binding>.Null;
165  }
166 
167  object[] arguments;
168  if (!TryResolveArguments(item, out arguments))
169  {
170  return Tasks<Binding>.Null;
171  }
172 
173  var binding = new Binding(this.method, this.parameters, instance, arguments);
174  return Task.FromResult(binding);
175  }
176  catch (OperationCanceledException error)
177  {
178  return Task.FromCanceled<Binding>(error.CancellationToken);
179  }
180  catch (Exception error)
181  {
182  return Task.FromException<Binding>(error);
183  }
184  }
185  protected override bool HasScore(IResolver resolver, Binding state)
186  {
187  return state != null;
188  }
189  protected override Binding GetScore(IResolver resolver, Binding state)
190  {
191  return state;
192  }
193  protected override Task PostAsync(IResolver item, Binding state, CancellationToken token)
194  {
195  try
196  {
197  return state.InvokeAsync(token);
198  }
199  catch (OperationCanceledException error)
200  {
201  return Task.FromCanceled(error.CancellationToken);
202  }
203  catch (Exception error)
204  {
205  return Task.FromException(error);
206  }
207  }
208  protected override Task DoneAsync(IResolver item, Binding state, CancellationToken token)
209  {
210  return Task.CompletedTask;
211  }
212  }
