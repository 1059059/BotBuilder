64  public sealed class OrderScorableFactory<Item, Score> : IScorableFactory<Item, Score>
65  {
66  private readonly IEnumerable<IScorableFactory<Item, Score>> factories;
67  public OrderScorableFactory(IEnumerable<IScorableFactory<Item, Score>> factories)
68  {
69  SetField.NotNull(out this.factories, nameof(factories), factories);
70  }
71  public OrderScorableFactory(params IScorableFactory<Item, Score>[] factories)
72  : this((IEnumerable<IScorableFactory<Item, Score>>)factories)
73  {
74  }
75  IScorable<Item, Score> IScorableFactory<Item, Score>.ScorableFor(IEnumerable<MethodInfo> methods)
76  {
77  var levels = from method in methods
78  // note, this is non-deterministic across executions, which seems lame
79  let defaultOrder = method.Name.GetHashCode()
80  let orders = InheritedAttributes.For<ScorableOrderAttribute>(method).Select(order => order.Order).DefaultIfEmpty(defaultOrder)
81  from order in orders
82  group method by order into g
83  orderby g.Key
84  select g;
85 
86  var scorables = from level in levels
87  from factory in this.factories
88  let scorable = factory.ScorableFor(level)
89  //where scorable != null
90  select scorable;
91 
92  var winner = scorables.ToArray().First();
93  return winner;
94  }
95  }
