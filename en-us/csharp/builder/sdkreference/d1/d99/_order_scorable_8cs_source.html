103  IScorable<Item, Score> IScorableFactory<Item, Score>.ScorableFor(IEnumerable<MethodInfo> methods)
104  {
105  var levels = from method in methods
106  // note, this is non-deterministic across executions, which seems lame
107  let defaultOrder = method.Name.GetHashCode()
108  let attributes = InheritedAttributes.For<ScorableGroupAttribute>(method)
109  let orders = attributes.Select(order => order.Order).DefaultIfEmpty(defaultOrder)
110  from order in orders
111  group method by order into g
112  orderby g.Key
113  select g;
114 
115  var scorables = from level in levels
116  from factory in this.factories
117  let scorable = factory.ScorableFor(level)
118  where Scorable.Keep(scorable)
119  select scorable;
120 
121  var winner = scorables.ToArray().Fold(this.comparer, this.onStage);
122  return winner;
123  }
124  }
