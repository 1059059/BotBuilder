108  public WhereScorable(IScorable<IResolver, Score> scorable, Delegate lambda)
109  {
110  SetField.NotNull(out this.scorable, nameof(scorable), scorable);
111  SetField.NotNull(out this.lambda, nameof(lambda), lambda);
112  }
113 
114  protected override async Task<Token> PrepareAsync(IResolver innerItem, CancellationToken token)
115  {
116  var innerState = await this.scorable.PrepareAsync(innerItem, token);
117  bool hasScore = false;
118  if (this.scorable.HasScore(innerItem, innerState))
119  {
120  var innerScore = this.scorable.GetScore(innerItem, innerState);
121  var outerItem = new ArrayResolver(innerItem, innerScore);
122 
123  IBinding<bool> binding;
124  if (Binder.Instance.TryBind<bool>(lambda, outerItem, out binding))
125  {
126  hasScore = await binding.InvokeAsync(token);
127  }
128  }
129 
130  var state = new Token()
131  {
132  Item = innerItem,
133  Scorable = this.scorable,
134  State = innerState,
135  HasScore = hasScore
136  };
137 
138  return state;
139  }
140 
141  protected override bool HasScore(IResolver item, Token state)
142  {
143  return state.HasScore;
144  }
145 
146  protected override Score GetScore(IResolver item, Token state)
147  {
148  return state.Scorable.GetScore(state.Item, state.State);
