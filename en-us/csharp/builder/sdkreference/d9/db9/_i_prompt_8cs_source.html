272  public FormPrompt Prompt(T state, IField<T> field, params object[] args)
273  {
274  string currentChoice = null;
275  string noValue = null;
276  if (field != null)
277  {
278  currentChoice = field.Template(TemplateUsage.CurrentChoice).Pattern();
279  if (field.Optional)
280  {
281  noValue = field.Template(TemplateUsage.NoPreference).Pattern();
282  }
283  else
284  {
285  noValue = field.Template(TemplateUsage.Unspecified).Pattern();
286  }
287  }
288  IList<DescribeAttribute> buttons = new List<DescribeAttribute>();
289  var response = ExpandTemplate(_annotation.Pattern(), currentChoice, noValue, state, field, args, ref buttons);
290  return new FormPrompt
291  {
292  Prompt = (response == null ? "" : _spacesPunc.Replace(_spaces.Replace(Language.ANormalization(response), "$1 "), "$1")),
293  Description = field?.FieldDescription,
294  Buttons = buttons,
295  Style = _annotation.ChoiceStyle
296  };

312  public static bool ValidatePattern(IForm<T> form, string pattern, IField<T> field, int argLimit = 0)
313  {
314  bool ok = true;
315  var fields = form.Fields;
316  foreach (Match match in _args.Matches(pattern))
317  {
318  var expr = match.Groups[1].Value.Trim();
319  int numeric;
320  if (expr == "||")
321  {
322  ok = true;
323  }
324  else if (expr.StartsWith("&"))
325  {
326  var name = expr.Substring(1);
327  if (name == "" && field != null) name = field.Name;
328  ok = (name == "" || fields.Field(name) != null);
329  }
330  else if (expr.StartsWith("?"))
331  {
332  ok = ValidatePattern(form, expr.Substring(1), field, argLimit);
333  }
334  else if (expr.StartsWith("["))
335  {
336  if (expr.EndsWith("]"))
337  {
338  ok = ValidatePattern(form, expr.Substring(1, expr.Length - 2), field, argLimit);
339  }
340  else
341  {
342  ok = false;
343  }
344  }
345  else if (expr.StartsWith("*"))
346  {
347  ok = (expr == "*" || expr == "*filled");
348  }
349  else if (TryParseFormat(expr, out numeric))
350  {
351  ok = numeric <= argLimit - 1;
352  }
353  else
354  {
355  var formatArgs = expr.Split(':');
356  var name = formatArgs[0];
357  if (name == "" && field != null) name = field.Name;
358  ok = (name == "" || fields.Field(name) != null);
359  }
360  if (!ok)
361  {
362  break;
363  }
364  }
365  return ok;
366  }

368  private string ExpandTemplate(string template, string currentChoice, string noValue, T state, IField<T> field, object[] args, ref IList<DescribeAttribute> buttons)
369  {
370  bool foundUnspecified = false;
371  int last = 0;
372  int numeric;
373  var response = new StringBuilder();
374 
375  foreach (Match match in _args.Matches(template))
376  {
377  var expr = match.Groups[1].Value.Trim();
378  var substitute = "";
379  if (expr.StartsWith("&"))
380  {
381  var name = expr.Substring(1);
382  if (name == "" && field != null) name = field.Name;
383  var pathField = _fields.Field(name);
384  substitute = Language.Normalize(pathField == null ? field.Name : pathField.FieldDescription.Description, _annotation.FieldCase);
385  }
386  else if (expr == "||")
387  {
388  var builder = new StringBuilder();
389  var values = _recognizer.ValueDescriptions();
390  var useButtons = !field.AllowsMultiple
391  && (_annotation.ChoiceStyle == ChoiceStyleOptions.Auto
392  || _annotation.ChoiceStyle == ChoiceStyleOptions.Buttons
393  || _annotation.ChoiceStyle == ChoiceStyleOptions.Carousel);
394  if (values.Any() && _annotation.AllowDefault != BoolDefault.False && field.Optional)
395  {
396  values = values.Concat(new DescribeAttribute[] { new DescribeAttribute(Language.Normalize(noValue, _annotation.ChoiceCase)) });
397  }
398  string current = null;
399  if (_annotation.AllowDefault != BoolDefault.False)
400  {
401  if (!field.Optional)
402  {
403  if (!field.IsUnknown(state))
404  {
405  current = ExpandTemplate(currentChoice, null, noValue, state, field, args, ref buttons);
406  }
407  }
408  else
409  {
410  current = ExpandTemplate(currentChoice, null, noValue, state, field, args, ref buttons);
411  }
412  }
413  if (values.Any())
414  {
415  if (useButtons)
416  {
417  foreach (var value in values)
418  {
419  buttons.Add(value);
420  }
421  }
422  else
423  {
424  // Buttons do not support multiple selection so we fall back to text
425  if (((_annotation.ChoiceStyle == ChoiceStyleOptions.Auto || _annotation.ChoiceStyle == ChoiceStyleOptions.AutoText)
426  && values.Count() < 4)
427  || (_annotation.ChoiceStyle == ChoiceStyleOptions.Inline))
428  {
429  // Inline choices
430  if (_annotation.ChoiceParens == BoolDefault.True) builder.Append('(');
431  var choices = new List<string>();
432  var i = 1;
433  foreach (var value in values)
434  {
435  choices.Add(string.Format(_annotation.ChoiceFormat, i, Language.Normalize(value.Description, _annotation.ChoiceCase)));
436  ++i;
437  }
438  builder.Append(Language.BuildList(choices, _annotation.ChoiceSeparator, _annotation.ChoiceLastSeparator));
439  if (_annotation.ChoiceParens == BoolDefault.True) builder.Append(')');
440  if (current != null)
441  {
442  builder.Append(" ");
443  builder.Append(current);
444  }
445  }
446  else
447  {
448  // Separate line choices
449  if (current != null)
450  {
451  builder.Append(current);
452  builder.Append(" ");
453  }
454  var i = 1;
455  foreach (var value in values)
456  {
457  builder.Append("\n ");
458  if (!_annotation.AllowNumbers)
459  {
460  builder.Append("* ");
461  }
462  builder.AppendFormat(_annotation.ChoiceFormat, i, Language.Normalize(value.Description, _annotation.ChoiceCase));
463  ++i;
464  }
465  }
466  }
467  }
468  else if (current != null)
469  {
470  builder.Append(" ");
471  builder.Append(current);
472  }
473  substitute = builder.ToString();
474  }
475  else if (expr.StartsWith("*"))
476  {
477  // Status display of active results
478  var filled = expr.ToLower().Trim().EndsWith("filled");
479  var builder = new StringBuilder();
480  if (match.Index > 0)
481  {
482  builder.Append("\n");
483  }
484  foreach (var entry in (from step in _fields where (!filled || !step.IsUnknown(state)) && step.Role == FieldRole.Value && step.Active(state) select step))
485  {
486  var format = new Prompter<T>(Template(entry, TemplateUsage.StatusFormat), _form, null);
487  builder.Append("* ").AppendLine(format.Prompt(state, entry).Prompt);
488  }
489  substitute = builder.ToString();
490  }
491  else if (expr.StartsWith("[") && expr.EndsWith("]"))
492  {
493  // Generate a list from multiple fields
494  var paths = expr.Substring(1, expr.Length - 2).Split(' ');
495  var values = new List<Tuple<IField<T>, object, string>>();
496  foreach (var spec in paths)
497  {
498  if (!spec.StartsWith("{") || !spec.EndsWith("}"))
499  {
500  throw new ArgumentException("Only {<field>} references are allowed in lists.");
501  }
502  var formatArgs = spec.Substring(1, spec.Length - 2).Trim().Split(':');
503  var name = formatArgs[0];
504  if (name == "" && field != null) name = field.Name;
505  var format = (formatArgs.Length > 1 ? "0:" + formatArgs[1] : "0");
506  var eltDesc = _fields.Field(name);
507  if (!eltDesc.IsUnknown(state))
508  {
509  var value = eltDesc.GetValue(state);
510  if (value.GetType() != typeof(string) && value.GetType().IsIEnumerable())
511  {
512  var eltValues = (value as System.Collections.IEnumerable);
513  foreach (var elt in eltValues)
514  {
515  values.Add(Tuple.Create(eltDesc, elt, format));
516  }
517  }
518  else
519  {
520  values.Add(Tuple.Create(eltDesc, eltDesc.GetValue(state), format));
521  }
522  }
523  }
524  if (values.Count() > 0)
525  {
526  var elements = (from elt in values
527  select Language.Normalize(ValueDescription(elt.Item1, elt.Item2, elt.Item3), _annotation.ValueCase)).ToArray();
528  substitute = Language.BuildList(elements, _annotation.Separator, _annotation.LastSeparator);
529  }
530  }
531  else if (expr.StartsWith("?"))
532  {
533  // Conditional template
534  var subValue = ExpandTemplate(expr.Substring(1), currentChoice, null, state, field, args, ref buttons);
535  if (subValue == null)
536  {
537  substitute = "";
538  }
539  else
540  {
541  substitute = subValue;
542  }
543  }
544  else if (TryParseFormat(expr, out numeric))
545  {
546  // Process ad hoc arg
547  if (numeric < args.Length && args[numeric] != null)
548  {
549  substitute = string.Format("{" + expr + "}", args);
550  }
551  else
552  {
553  foundUnspecified = true;
554  break;
555  }
556  }
557  else
558  {
559  var formatArgs = expr.Split(':');
560  var name = formatArgs[0];
561  if (name == "" && field != null) name = field.Name;
562  var pathDesc = _fields.Field(name);
563  if (pathDesc.IsUnknown(state))
564  {
565  if (noValue == null)
566  {
567  foundUnspecified = true;
568  break;
569  }
570  substitute = noValue;
571  }
572  else
573  {
574  var value = pathDesc.GetValue(state);
575  if (value.GetType() != typeof(string) && value.GetType().IsIEnumerable())
576  {
577  var values = (value as System.Collections.IEnumerable);
578  substitute = Language.BuildList(from elt in values.Cast<object>()
579  select Language.Normalize(ValueDescription(pathDesc, elt, "0"), _annotation.ValueCase),
580  _annotation.Separator, _annotation.LastSeparator);
581  }
582  else
583  {
584  var format = (formatArgs.Length > 1 ? "0:" + formatArgs[1] : "0");
585  substitute = ValueDescription(pathDesc, value, format);
586  }
587  }
588  }
589  response.Append(template.Substring(last, match.Index - last)).Append(substitute);
590  last = match.Index + match.Length;
591  }
592  return (foundUnspecified ? null : response.Append(template.Substring(last, template.Length - last)).ToString());
593  }
