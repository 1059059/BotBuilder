203  public virtual void Apply<T>(ref IMessageActivity message, string prompt, IList<T> options)
204  {
205  SetField.CheckNull(nameof(prompt), prompt);
206  SetField.CheckNull(nameof(options), options);
207  switch (PromptStyle)
208  {
209  case PromptStyle.Auto:
210  if (options != null && options.Any())
211  {
212  message.AddHeroCard(prompt, options);
213  }
214  else
215  {
216  message.Text = prompt;
217  }
218  break;
219  case PromptStyle.AutoText:
220  Apply(ref message, prompt, options, options?.Count() > 4 ? PromptStyle.PerLine : PromptStyle.Inline);
221  break;
222  case PromptStyle.Inline:
223  //TODO: Refactor buildlist function to a more generic namespace when changing prompt to use recognizers.
224  message.Text = $"{prompt} {FormFlow.Advanced.Language.BuildList(options.Select(option => option.ToString()), Resources.DefaultChoiceSeparator, Resources.DefaultChoiceLastSeparator)}";
225  break;
226  case PromptStyle.PerLine:
227  message.Text = $"{prompt}\n{FormFlow.Advanced.Language.BuildList(options.Select(option => $"* {option.ToString()}"), "\n", "\n")}";
228  break;
229  case PromptStyle.None:
230  default:
231  message.Text = prompt;
232  break;
233  }
234  }

427  protected override bool TryParse(IMessageActivity message, out bool result)
428  {
429  var found = false;
430  result = false;
431  if (message.Text != null)
432  {
433  var term = message.Text.Trim().ToLower();
434  if ((from r in Patterns[Yes] select r.ToLower()).Contains(term))
435  {
436  result = true;
437  found = true;
438  }
439  else if ((from r in Patterns[No] select r.ToLower()).Contains(term))
440  {
441  result = false;
442  found = true;
443  }
444  }
445  return found;
446  }

523  public virtual Tuple<bool, int> ScoreMatch(T option, string input)
524  {
525  var trimmed = input.Trim();
526  var text = option.ToString();
527  bool occurs = text.IndexOf(trimmed, StringComparison.CurrentCultureIgnoreCase) >= 0;
528  bool equals = text == trimmed;
529  return occurs
530  ? Tuple.Create(equals, trimmed.Length)
531  : null;
532  }
533 
534  protected override bool TryParse(IMessageActivity message, out T result)
535  {
536  if (!string.IsNullOrWhiteSpace(message.Text))
537  {
538  var scores = from option in this.promptOptions.Options
539  let score = ScoreMatch(option, message.Text)
540  select new { score, option };
541 
542  var winner = scores.MaxBy(s => s.score);
543  if (winner.score != null)
544  {
545  result = winner.option;
546  return true;
547  }
548  }
549 
550  result = default(T);
551  return false;
552  }
553  }

651  async Task IDialog<T>.StartAsync(IDialogContext context)
652  {
653  await context.PostAsync(this.MakePrompt(context, promptOptions.Prompt, promptOptions.Options));
654  context.Wait(MessageReceived);
655  }
656 
657  private async Task MessageReceived(IDialogContext context, IAwaitable<IMessageActivity> message)
658  {
659  T result;
660  if (this.TryParse(await message, out result))
661  {
662  context.Done(result);
663  }
664  else
665  {
666  --promptOptions.Attempts;
667  if (promptOptions.Attempts > 0)
668  {
669  await context.PostAsync(this.MakePrompt(context, promptOptions.Retry ?? promptOptions.DefaultRetry, promptOptions.Options));
670  context.Wait(MessageReceived);
671  }
672  else
673  {
674  //too many attempts, throw.
675  await context.PostAsync(this.MakePrompt(context, promptOptions.TooManyAttempts));
676  throw new TooManyAttemptsException(promptOptions.TooManyAttempts);
677  }
678  }
679  }
680 
681  protected abstract bool TryParse(IMessageActivity message, out T result);
682 
683  protected virtual IMessageActivity MakePrompt(IDialogContext context, string prompt, IList<U> options = null)
684  {
685  var msg = context.MakeMessage();
686  if (options != null && options.Count > 0)
687  {
688  promptOptions.PromptStyler.Apply(ref msg, prompt, options);
689  }
690  else
691  {
692  promptOptions.PromptStyler.Apply(ref msg, prompt);
693  }
694  return msg;
695  }
696  }
