227  public virtual void Apply<T>(ref IMessageActivity message, string prompt, IReadOnlyList<T> options, IReadOnlyList<string> descriptions = null)
228  {
229  SetField.CheckNull(nameof(prompt), prompt);
230  SetField.CheckNull(nameof(options), options);
231  if (descriptions == null)
232  {
233  descriptions = (from option in options select option.ToString()).ToList();
234  }
235  switch (PromptStyle)
236  {
237  case PromptStyle.Auto:
238  case PromptStyle.Keyboard:
239  if (options != null && options.Any())
240  {
241  if (PromptStyle == PromptStyle.Keyboard)
242  {
243  message.AddKeyboardCard(prompt, options, descriptions);
244  }
245  else
246  {
247  message.AddHeroCard(prompt, options, descriptions);
248  }
249  }
250  else
251  {
252  message.Text = prompt;
253  }
254  break;
255  case PromptStyle.AutoText:
256  Apply(ref message, prompt, options, options?.Count() > 4 ? PromptStyle.PerLine : PromptStyle.Inline, descriptions);
257  break;
258  case PromptStyle.Inline:
259  //TODO: Refactor buildlist function to a more generic namespace when changing prompt to use recognizers.
260  message.Text = $"{prompt} {FormFlow.Advanced.Language.BuildList(descriptions, Resources.DefaultChoiceSeparator, Resources.DefaultChoiceLastSeparator)}";
261  break;
262  case PromptStyle.PerLine:
263  message.Text = $"{prompt}\n{FormFlow.Advanced.Language.BuildList(descriptions.Select(description => $"* {description}"), "\n", "\n")}";
264  break;
265  case PromptStyle.None:
266  default:
267  message.Text = prompt;
268  break;
269  }
270  }

464  protected override bool TryParse(IMessageActivity message, out bool result)
465  {
466  var found = false;
467  result = false;
468  if (message.Text != null)
469  {
470  var term = message.Text.Trim().ToLower();
471  if ((from r in Patterns[Yes] select r.ToLower()).Contains(term))
472  {
473  result = true;
474  found = true;
475  }
476  else if ((from r in Patterns[No] select r.ToLower()).Contains(term))
477  {
478  result = false;
479  found = true;
480  }
481  }
482  return found;
483  }

561  public virtual Tuple<bool, int> ScoreMatch(T option, string input)
562  {
563  var trimmed = input.Trim();
564  var text = option.ToString();
565  bool occurs = text.IndexOf(trimmed, StringComparison.CurrentCultureIgnoreCase) >= 0;
566  bool equals = text == trimmed;
567  return occurs
568  ? Tuple.Create(equals, trimmed.Length)
569  : null;
570  }
571 
572  protected override bool TryParse(IMessageActivity message, out T result)
573  {
574  if (!string.IsNullOrWhiteSpace(message.Text))
575  {
576  var scores = from option in this.promptOptions.Options
577  let score = ScoreMatch(option, message.Text)
578  select new { score, option };
579 
580  var winner = scores.MaxBy(s => s.score);
581  if (winner.score != null)
582  {
583  result = winner.option;
584  return true;
585  }
586  }
587 
588  result = default(T);
589  return false;
590  }
591  }

727  async Task IDialog<T>.StartAsync(IDialogContext context)
728  {
729  await context.PostAsync(this.MakePrompt(context, promptOptions.Prompt, promptOptions.Options, promptOptions.Descriptions));
730  context.Wait(MessageReceivedAsync);
731  }
732 
733  protected virtual async Task MessageReceivedAsync(IDialogContext context, IAwaitable<IMessageActivity> message)
734  {
735  T result;
736  if (this.TryParse(await message, out result))
737  {
738  context.Done(result);
739  }
740  else
741  {
742  --promptOptions.Attempts;
743  if (promptOptions.Attempts > 0)
744  {
745  await context.PostAsync(this.MakePrompt(context, promptOptions.Retry ?? promptOptions.DefaultRetry, promptOptions.Options));
746  context.Wait(MessageReceivedAsync);
747  }
748  else
749  {
750  //too many attempts, throw.
751  await context.PostAsync(this.MakePrompt(context, promptOptions.TooManyAttempts));
752  throw new TooManyAttemptsException(promptOptions.TooManyAttempts);
753  }
754  }
755  }
756 
757  protected abstract bool TryParse(IMessageActivity message, out T result);
758 
759  protected virtual IMessageActivity MakePrompt(IDialogContext context, string prompt, IReadOnlyList<U> options = null, IReadOnlyList<string> descriptions = null)
760  {
761  var msg = context.MakeMessage();
762  if (options != null && options.Count > 0)
763  {
764  promptOptions.PromptStyler.Apply(ref msg, prompt, options, descriptions);
765  }
766  else
767  {
768  promptOptions.PromptStyler.Apply(ref msg, prompt);
769  }
770  return msg;
771  }
772  }
