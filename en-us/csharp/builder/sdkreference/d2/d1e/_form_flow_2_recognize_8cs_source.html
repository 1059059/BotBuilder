113  {

116  if (_noPreference != null)

143  public IEnumerable<TermMatch> Matches(string input, object defaultValue)
144  {
145  // if the user hit enter on an optional prompt, then consider taking the current choice as a low confidence option
146  bool userSkippedPrompt = string.IsNullOrWhiteSpace(input) && (defaultValue != null || _noPreference != null);
147  if (userSkippedPrompt)
148  {
149  yield return new TermMatch(0, input.Length, 1.0, defaultValue);
150  }
151 
152  foreach (var expression in _expressions)
153  {
154  double maxWords = expression.MaxWords;
155  foreach (Match match in expression.Expression.Matches(input))
156  {
157  var group1 = match.Groups[1];
158  var group2 = match.Groups[2];
159  var group3 = match.Groups[3];
160  object newValue;
161  if (group1.Success)
162  {
163  if (ConvertSpecial(expression.Value, defaultValue, out newValue))
164  {
165  yield return new TermMatch(group1.Index, group1.Length, 1.0, newValue);
166  }
167  }
168  if (group2.Success)
169  {
170  var words = group2.Value.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries).Length;
171  var confidence = System.Math.Min(words / maxWords, 1.0);
172  if (ConvertSpecial(expression.Value, defaultValue, out newValue))
173  {
174  yield return new TermMatch(group2.Index, group2.Length, confidence, newValue);
175  }
176  }
177  else if (group3.Success)
178  {
179  if (ConvertSpecial(expression.Value, defaultValue, out newValue))
180  {
181  yield return new TermMatch(group3.Index, group3.Length, 1.0, newValue);
182  }
183  }
184  }
185  }

202  private enum Special { CurrentChoice, NoPreference };
203 
204  // Word character, any word character, any digit, any positive group over word characters
205  private const string WORD = @"(\w|\\w|\\d|(\[(?>(\w|-)+|\[(?<number>)|\](?<-number>))*(?(number)(?!))\]))";
206  private static Regex _wordStart = new Regex(string.Format(@"^{0}|\(", WORD), RegexOptions.Compiled);
207  private static Regex _wordEnd = new Regex(string.Format(@"({0}|\))(\?|\*|\+|\{{\d+\}}|\{{,\d+\}}|\{{\d+,\d+\}})?$", WORD), RegexOptions.Compiled);
208 
209  private void BuildPerValueMatcher(IEnumerable<string> currentChoice)
210  {
211  if (currentChoice != null)
212  {
213  // 0 is reserved for current default if any
214  AddExpression(0, Special.CurrentChoice, currentChoice, _allowNumbers);
215  }
216  var n = 1;
217  foreach (var value in _values)
218  {
219  n = AddExpression(n, value, _termsDelegate(value), _allowNumbers);
220  }
221  if (_noPreference != null)
222  {
223  // Add recognizer for no preference
224  n = AddExpression(n, Special.NoPreference, _noPreference, _allowNumbers);
225  }
226  if (_terms != null && _terms.Count() > 0)
227  {
228  // Add field terms to help disambiguate
229  AddExpression(n, SpecialValues.Field, _terms, false);
230  }
231  _max = n - 1;
232  }

234  private int NumberOfWords(string regex)
235  {
236  return regex.Split(new string[] { @"\s", " " }, StringSplitOptions.RemoveEmptyEntries).Length;
237  }
238 
239  // Generate a regex with 3 parts:
240  // Group 1: startWord <number> endWord
241  // Group 2: startWord (all word terms) endWord
242  // Group 3: all nonWord terms
243  // If a group is empty, match on NOMATCH constant.
244  private const string NOMATCH = "__qqqq__";
245  private int AddExpression(int n, object value, IEnumerable<string> terms, bool allowNumbers)
246  {
247  var orderedTerms = (from term in terms orderby term.Length descending select term).ToArray();
248  var words = new StringBuilder();
249  var nonWords = new StringBuilder();
250  var first = true;
251  var firstNonWord = true;
252  int maxWords = 0;
253  if (orderedTerms.Length > 0)
254  {
255  maxWords = terms.Max(NumberOfWords);
256  foreach (var term in orderedTerms)
257  {
258  var nterm = term.Trim().Replace(" ", @"\s+");
259  if (nterm != "")
260  {
261  if (_wordStart.Match(nterm).Success && _wordEnd.Match(nterm).Success)
262  {
263  if (first)
264  {
265  first = false;
266  words.Append(@"(\b(?:");
267  }
268  else
269  {
270  words.Append('|');
271  }
272  words.Append(@"(?:");
273  words.Append(nterm);
274  words.Append(')');
275  }
276  else
277  {
278  if (firstNonWord)
279  {
280  firstNonWord = false;
281  nonWords.Append('(');
282  }
283  else
284  {
285  nonWords.Append('|');
286  }
287  nonWords.Append(@"(?:");
288  nonWords.Append(nterm);
289  nonWords.Append(')');
290  }
291  }
292  }
293  }
294  if (first)
295  {
296  words.Append(@"(\b(?:");
297  words.Append(NOMATCH);
298  }
299  words.Append(@")\b)");
300  if (firstNonWord)
301  {
302  nonWords.Append('(');
303  nonWords.Append(NOMATCH);
304  }
305  nonWords.Append(')');
306  var numbers = allowNumbers ? $"(\\b{n}\\b)" : NOMATCH;
307  var expr = $"{numbers}|{words.ToString()}|{nonWords.ToString()}";
308  _expressions.Add(new ValueAndExpression(value, new Regex(expr.ToString(), RegexOptions.IgnoreCase), maxWords));
309  ++n;
310  return n;
311  }

313  private bool ConvertSpecial(object value, object defaultValue, out object newValue)
314  {
315  bool ok = true;
316  newValue = value;
317  if (value is Special)
318  {
319  var special = (Special)value;
320  if (special == Special.CurrentChoice && (_noPreference != null || defaultValue != null))
321  {
322  newValue = defaultValue;
323  }
324  else if (special == Special.NoPreference)
325  {
326  newValue = null;
327  }
328  else
329  {
330  ok = false;
331  }
332  }
333  return ok;
334  }
