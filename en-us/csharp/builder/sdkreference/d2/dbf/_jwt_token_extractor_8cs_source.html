63  public async Task<ClaimsIdentity> GetIdentityAsync(string scheme, string parameter)
64  {
65  // No header in correct scheme?
66  if (scheme != "Bearer")
67  return null;
68 
69  // Issuer isn't allowed? No need to check signature
70  if (!HasAllowedIssuer(parameter))
71  return null;
72 
73  try
74  {
75  ClaimsPrincipal claimsPrincipal = await ValidateTokenAsync(parameter).ConfigureAwait(false);
76  return claimsPrincipal.Identities.OfType<ClaimsIdentity>().FirstOrDefault();
77  }
78  catch (Exception e)
79  {
80  Trace.TraceWarning("Invalid token. " + e.ToString());
81  return null;
82  }

85  private bool HasAllowedIssuer(string jwtToken)
86  {
87  JwtSecurityToken token = new JwtSecurityToken(jwtToken);
88  if (_tokenValidationParameters.ValidIssuer != null && _tokenValidationParameters.ValidIssuer == token.Issuer)
89  return true;
90 
91  if ((_tokenValidationParameters.ValidIssuers ?? Enumerable.Empty<string>()).Contains(token.Issuer))
92  return true;
93

97  public string GetAppIdFromClaimsIdentity(ClaimsIdentity identity)
98  {
99  if (identity == null)
100  return null;
101 
102  Claim botClaim = identity.Claims.FirstOrDefault(c => _tokenValidationParameters.ValidIssuers.Contains(c.Issuer) && c.Type == "appid");
103  if (botClaim != null)
104  return botClaim.Value;
105 
106  // Fallback for BF-issued tokens
107  botClaim = identity.Claims.FirstOrDefault(c => c.Issuer == "https://api.botframework.com" && c.Type == "aud");
108  if (botClaim != null)
109  return botClaim.Value;
110 
111  return null;
112  }

114  private async Task<ClaimsPrincipal> ValidateTokenAsync(string jwtToken)
115  {
116  // _openIdMetadata only does a full refresh when the cache expires every 5 days
117  OpenIdConnectConfiguration config = null;
118  try
119  {
120  config = await _openIdMetadata.GetConfigurationAsync().ConfigureAwait(false);
121  }
122  catch (Exception e)
123  {
124  Trace.TraceError($"Error refreshing OpenId configuration: {e}");
125 
126  // No config? We can't continue
127  if (config == null)
128  throw;
129  }
130 
131  // Update the signing tokens from the last refresh
132  _tokenValidationParameters.IssuerSigningTokens = config.SigningTokens;
133 
134  JwtSecurityTokenHandler tokenHandler = new JwtSecurityTokenHandler();
135 
136  try
137  {
138  SecurityToken parsedToken;
139  ClaimsPrincipal principal = tokenHandler.ValidateToken(jwtToken, _tokenValidationParameters, out parsedToken);
140  return principal;
141  }
142  catch (SecurityTokenSignatureKeyNotFoundException)
143  {
144  string keys = string.Join(", ", ((config?.SigningTokens) ?? Enumerable.Empty<SecurityToken>()).Select(t => t.Id));
145  Trace.TraceError("Error finding key for token. Available keys: " + keys);
146  throw;
147  }
148  }
