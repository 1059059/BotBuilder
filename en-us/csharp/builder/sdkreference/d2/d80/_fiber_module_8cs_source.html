61  private static IEnumerable<object> DoNotSerialize(IComponentContext context, IEnumerable<Parameter> parameters)
62  {
63  foreach (var registration in context.ComponentRegistry.Registrations)
64  {
65  foreach (var service in registration.Services)
66  {
67  var keyed = service as KeyedService;
68  if (keyed != null)
69  {
70  if (keyed.ServiceKey == Key_DoNotSerialize)
71  {
72  object instance;
73  if (context.TryResolveService(service, parameters, out instance))
74  {
75  yield return instance;
76  }
77  }
78  }
79  }
80  }
