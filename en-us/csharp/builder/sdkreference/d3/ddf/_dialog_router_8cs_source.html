50  public static IEnumerable<IScorable<IActivity, double>> EnumerateRelevant(
51  IDialogStack stack,
52  IEnumerable<IScorable<IActivity, double>> fromActivity,
53  IEnumerable<IScorable<IResolver, double>> fromResolver,
54  Func<IActivity, IResolver> makeResolver)
55  {
56  // first, let's go through stack frames
57  var targets = stack.Frames.Select(f => f.Target);
58  foreach (var target in targets)
59  {
60  var activityScorable = target as IScorable<IActivity, double>;
61  if (activityScorable != null)
62  {
63  yield return activityScorable;
64  }
65 
66  var resolverScorable = target as IScorable<IResolver, double>;
67  if (resolverScorable != null)
68  {
69  yield return resolverScorable.SelectItem(makeResolver);
70  }
71  }
72 
73  // then global scorables "on the side"
74  foreach (var activityScorable in fromActivity)
75  {
76  yield return activityScorable;
77  }
78 
79  foreach (var resolverScorable in fromResolver)
80  {
81  yield return resolverScorable.SelectItem(makeResolver);
82  }
83  }

85  public static IScorable<IActivity, double> MakeDelegate(
86  IDialogStack stack,
87  IEnumerable<IScorable<IActivity, double>> fromActivity,
88  IEnumerable<IScorable<IResolver, double>> fromResolver,
89  Func<IActivity, IResolver> makeResolver,
90  ITraits<double> traits,
91  IComparer<double> comparer)
92  {
93  // since the stack of scorables changes over time, this should be lazy
94  var relevant = EnumerateRelevant(stack, fromActivity, fromResolver, makeResolver);
95  var significant = relevant.Select(s => s.WhereScore((_, score) => comparer.Compare(score, traits.Minimum) >= 0));
96  var scorable = new TraitsScorable<IActivity, double>(traits, comparer, significant);
97  return scorable;
98  }
