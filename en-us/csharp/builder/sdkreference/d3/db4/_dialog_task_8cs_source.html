198  void IDialogStack.Call<R>(IDialog<R> child, ResumeAfter<R> resume)
199  {
200  var callRest = ToRest(child.StartAsync);
201  if (resume != null)
202  {
203  var doneRest = ToRest(resume);
204  this.wait = this.fiber.Call<DialogTask, object, R>(callRest, null, doneRest);
205  }
206  else
207  {
208  this.wait = this.fiber.Call<DialogTask, object>(callRest, null);
209  }
210  }

236  async Task IDialogStack.PollAsync(CancellationToken token)
237  {
238  try
239  {
240  await this.fiber.PollAsync(this, token);
241 
242  // this line will throw an error if the code does not schedule the next callback
243  // to wait for the next message sent from the user to the bot.
244  this.fiber.Wait.ValidateNeed(Need.Wait);
245  }
246  catch
247  {
248  this.store.Reset();
249  throw;
250  }
251  finally
252  {
253  this.store.Save(this.fiber);
254  this.store.Flush();
255  }
256  }

292  async Task IPostToBot.PostAsync<T>(T item, CancellationToken token)
293  {
294  try
295  {
296  if (dialogTask.Frames.Count == 0)
297  {
298  var root = this.makeRoot();
299  var loop = root.Loop();
300  dialogTask.Call(loop, null);
301  await dialogTask.PollAsync(token);
302  }
303 
304  await dialogTask.PostAsync(item, token);
305  }
306  catch
307  {
308  dialogTask.Reset();
309  throw;
310  }
311  }

340  public struct LocalizedScope : IDisposable
341  {
342  private readonly CultureInfo previousCulture;
343  private readonly CultureInfo previousUICulture;
344 
345  public LocalizedScope(string locale)
346  {
347  this.previousCulture = Thread.CurrentThread.CurrentCulture;
348  this.previousUICulture = Thread.CurrentThread.CurrentUICulture;
349 
350  if (!string.IsNullOrWhiteSpace(locale))
351  {
352  CultureInfo found = null;
353  try
354  {
355  found = CultureInfo.GetCultureInfo(locale);
356  }
357  catch (CultureNotFoundException)
358  {
359  }
360 
361  if (found != null)
362  {
363  Thread.CurrentThread.CurrentCulture = found;
364  Thread.CurrentThread.CurrentUICulture = found;
365  }
366  }
367  }
368 
369  public void Dispose()
370  {
371  Thread.CurrentThread.CurrentCulture = previousCulture;
372  Thread.CurrentThread.CurrentUICulture = previousUICulture;
373  }
