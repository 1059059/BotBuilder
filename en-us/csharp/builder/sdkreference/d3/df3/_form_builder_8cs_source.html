54  public virtual IForm<T> Build(Assembly resourceAssembly = null, string resourceName = null)
55  {
56  if (resourceAssembly == null)
57  {
58  resourceAssembly = typeof(T).Assembly;
59  }
60  if (resourceName == null)
61  {
62  resourceName = typeof(T).FullName;
63  }
64  var lang = resourceAssembly.GetCustomAttribute<NeutralResourcesLanguageAttribute>();
65  if (lang != null && !string.IsNullOrWhiteSpace(lang.CultureName))
66  {
67  try
68  {
69  IEnumerable<string> missing, extra;
70  string name = null;
71  foreach (var resource in resourceAssembly.GetManifestResourceNames())
72  {
73  if (resource.Contains(resourceName))
74  {
75  var pieces = resource.Split('.');
76  name = string.Join(".", pieces.Take(pieces.Count() - 1));
77  break;
78  }
79  }
80  if (name != null)
81  {
82  var rm = new ResourceManager(name, resourceAssembly);
83  var rs = rm.GetResourceSet(Thread.CurrentThread.CurrentUICulture, true, true);
84  _form.Localize(rs.GetEnumerator(), out missing, out extra);
85  if (missing.Any())
86  {
87  throw new MissingManifestResourceException($"Missing resources {missing}");
88  }
89  }
90  }
91  catch (MissingManifestResourceException)
92  {
93  // Resource was not localized
94  }
95  }
96  Validate();
97  return this._form;
98  }

214  private void Validate()
215  {
216  foreach (var step in _form._steps)
217  {
218  // Validate prompt
219  var annotation = step.Annotation;
220  if (annotation != null)
221  {
222  var name = step.Type == StepType.Field ? step.Name : "";
223  foreach (var pattern in annotation.Patterns)
224  {
225  ValidatePattern(pattern, name, 5);
226  }
227  if (step.Type != StepType.Message)
228  {
229  foreach (TemplateUsage usage in Enum.GetValues(typeof(TemplateUsage)))
230  {
231  if (usage != TemplateUsage.None)
232  {
233  foreach (var pattern in step.Field.Template(usage).Patterns)
234  {
235  ValidatePattern(pattern, name, TemplateArgs(usage));
236  }
237  }
238  }
239  }
240  }
241  }
242  ValidatePattern(_form.Configuration.DefaultPrompt.ChoiceFormat, "", 2);
243  }
244 
245  private void ValidatePattern(string pattern, string pathName, int maxArgs)
246  {
247  if (!Prompter<T>.ValidatePattern(_form, pattern, pathName, maxArgs))
248  {
249  throw new ArgumentException(string.Format("Illegal pattern: \"{0}\"", pattern));

253  private IFormBuilder<T> AddField(IField<T> field)
254  {
255  field.Form = _form;
256  _form._fields.Add(field);
257  var step = new FieldStep<T>(field.Name, _form);
258  var stepIndex = this._form._steps.FindIndex(s => s.Name == field.Name);
259  if (stepIndex >= 0)
260  {
261  _form._steps[stepIndex] = step;
262  }
263  else
264  {
265  _form._steps.Add(step);
266  }
267  return this;
268  }

379  public override IForm<T> Build(Assembly resourceAssembly = null, string resourceName = null)
380  {
381  if (!_form._steps.Any((step) => step.Type == StepType.Field))
382  {
383  var paths = new List<string>();
384  FieldPaths(typeof(T), "", paths);
385  foreach (var path in paths)
386  {
387  Field(new FieldReflector<T>(path, _ignoreAnnotations));
388  }
389  Confirm(new PromptAttribute(_form.Configuration.Template(TemplateUsage.Confirmation)));
390  }
391  return base.Build(resourceAssembly, resourceName);
392  }

416  public override IFormBuilder<T> AddRemainingFields(IEnumerable<string> exclude = null)
417  {
418  var exclusions = (exclude == null ? Array.Empty<string>() : exclude.ToArray());
419  var paths = new List<string>();
420  FieldPaths(typeof(T), "", paths);
421  foreach (var path in paths)
422  {
423  if (!exclusions.Contains(path) && !HasField(path))
424  {

426  }
427  }

431  private void FieldPaths(Type type, string path, List<string> paths)
432  {
433  var newPath = (path == "" ? path : path + ".");
434  foreach (var field in type.GetFields(BindingFlags.Public | BindingFlags.Instance))
435  {
436  TypePaths(field.FieldType, newPath + field.Name, paths);
437  }
438 
439  foreach (var property in type.GetProperties(BindingFlags.Public | BindingFlags.Instance))
440  {
441  if (property.CanRead && property.CanWrite)
442  {
443  TypePaths(property.PropertyType, newPath + property.Name, paths);
444  }
445  }
446  }
447 
448  private void TypePaths(Type type, string path, List<string> paths)
449  {
450  if (type.IsClass)
451  {
452  if (type == typeof(string))
453  {
454  paths.Add(path);
455  }
456  else if (type.IsIEnumerable())
457  {
458  var elt = type.GetGenericElementType();
459  if (elt.IsEnum)
460  {
461  paths.Add(path);
462  }
463  else
464  {
465  // TODO: What to do about enumerations of things other than enums?
466  }
467  }
468  else
469  {
470  FieldPaths(type, path, paths);
471  }
472  }
473  else if (type.IsEnum)
474  {
475  paths.Add(path);
476  }
477  else if (type == typeof(bool))
478  {
479  paths.Add(path);
480  }
481  else if (type.IsIntegral())
482  {
483  paths.Add(path);
484  }
485  else if (type.IsDouble())
486  {
487  paths.Add(path);
488  }
489  else if (type.IsNullable() && type.IsValueType)
490  {
491  paths.Add(path);
492  }
493  else if (type == typeof(DateTime))
494  {
495  paths.Add(path);
496  }
497  }
498  }
