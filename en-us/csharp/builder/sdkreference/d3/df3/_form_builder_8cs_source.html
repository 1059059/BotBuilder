57  public virtual IForm<T> Build(Assembly resourceAssembly = null, string resourceName = null)
58  {
59  if (resourceAssembly == null)
60  {
61  resourceAssembly = typeof(T).Assembly;
62  }
63  if (resourceName == null)
64  {
65  resourceName = typeof(T).FullName;
66  }
67  if (this._form._prompter == null)
68  {
69  this._form._prompter = async (context, prompt) =>
70  {
71  var msg = context.MakeMessage();
72  if (prompt.Buttons?.Count > 0)
73  {
74  var style = prompt.Style;
75  if (style == ChoiceStyleOptions.Auto)
76  {
77  foreach (var button in prompt.Buttons)
78  {
79  // Images require carousel
80  if (button.Image != null)
81  {
82  style = ChoiceStyleOptions.Carousel;
83  break;
84  }
85  }
86  }
87  if (style == ChoiceStyleOptions.Carousel)
88  {
89  msg.AttachmentLayout = AttachmentLayoutTypes.Carousel;
90  msg.Attachments = prompt.GenerateHeroCards();
91  }
92  else
93  {
94  msg.AttachmentLayout = AttachmentLayoutTypes.List;
95  msg.Attachments = prompt.GenerateHeroCard();
96  }
97  }
98  else if (prompt.Description?.Image != null)
99  {
100  msg.AttachmentLayout = AttachmentLayoutTypes.List;
101  var card = new HeroCard() { Title = prompt.Prompt, Images = new List<CardImage> { new CardImage(prompt.Description.Image) } };
102  msg.Attachments = new List<Attachment> { card.ToAttachment() };
103  }
104  else
105  {
106  msg.Text = prompt.Prompt;
107  }
108  await context.PostAsync(msg);
109  return prompt;
110  };
111  }
112  var lang = resourceAssembly.GetCustomAttribute<NeutralResourcesLanguageAttribute>();
113  if (lang != null && !string.IsNullOrWhiteSpace(lang.CultureName))
114  {
115  try
116  {
117  IEnumerable<string> missing, extra;
118  string name = null;
119  foreach (var resource in resourceAssembly.GetManifestResourceNames())
120  {
121  if (resource.Contains(resourceName))
122  {
123  var pieces = resource.Split('.');
124  name = string.Join(".", pieces.Take(pieces.Count() - 1));
125  break;
126  }
127  }
128  if (name != null)
129  {
130  var rm = new ResourceManager(name, resourceAssembly);
131  var rs = rm.GetResourceSet(Thread.CurrentThread.CurrentUICulture, true, true);
132  _form.Localize(rs.GetEnumerator(), out missing, out extra);
133  if (missing.Any())
134  {
135  throw new MissingManifestResourceException($"Missing resources {missing}");
136  }
137  }
138  }
139  catch (MissingManifestResourceException)
140  {
141  // Resource was not localized
142  }
143  }
144  Validate();
145  return this._form;
146  }

268  private void Validate()
269  {
270  foreach (var step in _form._steps)
271  {
272  // Validate prompt
273  var annotation = step.Annotation;
274  if (annotation != null)
275  {
276  var name = step.Type == StepType.Field ? step.Name : "";
277  foreach (var pattern in annotation.Patterns)
278  {
279  ValidatePattern(pattern, _form.Fields.Field(name), 5);
280  }
281  if (step.Type != StepType.Message)
282  {
283  foreach (TemplateUsage usage in Enum.GetValues(typeof(TemplateUsage)))
284  {
285  if (usage != TemplateUsage.None)
286  {
287  foreach (var pattern in step.Field.Template(usage).Patterns)
288  {
289  ValidatePattern(pattern, _form.Fields.Field(name), TemplateArgs(usage));
290  }
291  }
292  }
293  }
294  }
295  }
296  ValidatePattern(_form.Configuration.DefaultPrompt.ChoiceFormat, null, 2);
297  }

307  private IFormBuilder<T> AddField(IField<T> field)
308  {
309  field.Form = _form;
310  _form._fields.Add(field);
311  var step = new FieldStep<T>(field.Name, _form);
312  var stepIndex = this._form._steps.FindIndex(s => s.Name == field.Name);
313  if (stepIndex >= 0)
314  {
315  _form._steps[stepIndex] = step;
316  }
317  else
318  {
319  _form._steps.Add(step);
320  }
321  return this;

439  public override IForm<T> Build(Assembly resourceAssembly = null, string resourceName = null)
440  {
441  if (!_form._steps.Any((step) => step.Type == StepType.Field))
442  {
443  var paths = new List<string>();
444  FieldPaths(typeof(T), "", paths);
445  foreach (var path in paths)
446  {
447  Field(new FieldReflector<T>(path, _ignoreAnnotations));
448  }
449  Confirm(new PromptAttribute(_form.Configuration.Template(TemplateUsage.Confirmation)));
450  }
451  return base.Build(resourceAssembly, resourceName);
452  }

476  public override IFormBuilder<T> AddRemainingFields(IEnumerable<string> exclude = null)
477  {
478  var exclusions = (exclude == null ? Array.Empty<string>() : exclude.ToArray());
479  var paths = new List<string>();
480  FieldPaths(typeof(T), "", paths);
481  foreach (var path in paths)
482  {
483  if (!exclusions.Contains(path) && !HasField(path))

486  }
487  }
488  return this;

491  private void FieldPaths(Type type, string path, List<string> paths)
492  {
493  var newPath = (path == "" ? path : path + ".");
494  foreach (var field in type.GetFields(BindingFlags.Public | BindingFlags.Instance))
495  {
496  TypePaths(field.FieldType, newPath + field.Name, paths);
497  }
498 
499  foreach (var property in type.GetProperties(BindingFlags.Public | BindingFlags.Instance))
500  {
501  if (property.CanRead && property.CanWrite)
502  {
503  TypePaths(property.PropertyType, newPath + property.Name, paths);
504  }
505  }
506  }
507 
508  private void TypePaths(Type type, string path, List<string> paths)
509  {
510  if (type.IsClass)
511  {
512  if (type == typeof(string))
513  {
514  paths.Add(path);
515  }
516  else if (type.IsIEnumerable())
517  {
518  var elt = type.GetGenericElementType();
519  if (elt.IsEnum)
520  {
521  paths.Add(path);
522  }
523  else
524  {
525  // TODO: What to do about enumerations of things other than enums?
526  }
527  }
528  else
529  {
530  FieldPaths(type, path, paths);
531  }
532  }
533  else if (type.IsEnum)
534  {
535  paths.Add(path);
536  }
537  else if (type == typeof(bool))
538  {
539  paths.Add(path);
540  }
541  else if (type.IsIntegral())
542  {
543  paths.Add(path);
544  }
545  else if (type.IsDouble())
546  {
547  paths.Add(path);
548  }
549  else if (type.IsNullable() && type.IsValueType)
550  {
551  paths.Add(path);
552  }
553  else if (type == typeof(DateTime))
554  {
555  paths.Add(path);
556  }
557  }
558  }
