212  .Keyed<Regex>(Key_DeleteProfile_Regex)

214 
215  builder
216  .Register(c => new DeleteProfileScorable(c.Resolve<IDialogStack>(), c.Resolve<IBotData>(), c.Resolve<IBotToUser>(), c.ResolveKeyed<Regex>(Key_DeleteProfile_Regex)))
217  .As<IScorable<IActivity, double>>()
218  .InstancePerLifetimeScope();
219 
220  builder
221  .Register(c =>
222  {
223  var stack = c.Resolve<IDialogStack>();
224  var fromStack = stack.Frames.Select(f => f.Target).OfType<IScorable<IActivity, double>>();
225  var fromGlobal = c.Resolve<IScorable<IActivity, double>[]>();
226  // since the stack of scorables changes over time, this should be lazy
227  var lazyScorables = fromStack.Concat(fromGlobal);
228  var scorable = new TraitsScorable<IActivity, double>(c.Resolve<ITraits<double>>(), c.Resolve<IComparer<double>>(), lazyScorables);
229  return scorable;
230  })
231  .InstancePerLifetimeScope()
232  .AsSelf();
233 
234  builder
235  .RegisterType<NullActivityLogger>()
236  .AsImplementedInterfaces()
237  .SingleInstance();
238 
239  builder
240  .Register(c =>
241  {
242  var cc = c.Resolve<IComponentContext>();
243  Func<IPostToBot> makeInner = () =>
244  {
245  IPostToBot post = new ReactiveDialogTask(cc.Resolve<IDialogTask>(), cc.Resolve<Func<IDialog<object>>>());
246  post = new ExceptionTranslationDialogTask(post);
247  post = new LocalizedDialogTask(post);
248  post = new ScoringDialogTask<double>(post, cc.Resolve<TraitsScorable<IActivity, double>>());
249  return post;
250  };
251 
252  IPostToBot outer = new PersistentDialogTask(makeInner, cc.Resolve<IBotData>());
253  outer = new SerializingDialogTask(outer, cc.Resolve<IAddress>(), c.Resolve<IScope<IAddress>>());
254  outer = new PostUnhandledExceptionToUserTask(outer, cc.Resolve<IBotToUser>(), cc.Resolve<ResourceManager>(), cc.Resolve<TraceListener>());
255  outer = new LogPostToBot(outer, cc.Resolve<IActivityLogger>());
256  return outer;
257  })
258  .As<IPostToBot>()
259  .InstancePerLifetimeScope();
260 
261  builder
262  .RegisterType<AlwaysSendDirect_BotToUser>()
263  .AsSelf()
264  .InstancePerLifetimeScope();
265 
266  builder

271  .InstancePerLifetimeScope();
272 
273  }
