125  protected override bool OnFold(IScorable<Item, Score> scorable, Item item, object state, Score score)
126  {

128  {
129  throw new ArgumentOutOfRangeException(nameof(score));
130  }
131 
132  var maximum = this.comparer.Compare(score, this.traits.Maximum);
133  if (maximum > 0)
134  {
135  throw new ArgumentOutOfRangeException(nameof(score));
136  }
137  else if (maximum == 0)
138  {
139  return false;
