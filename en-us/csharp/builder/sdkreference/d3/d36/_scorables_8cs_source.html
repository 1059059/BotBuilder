49  public static async Task<bool> TryPostAsync<Item, Score>(this IScorable<Item, Score> scorable, Item item, CancellationToken token)
50  {
51  var state = await scorable.PrepareAsync(item, token);
52  try
53  {
54  if (scorable.HasScore(item, state))
55  {
56  var score = scorable.GetScore(item, state);
57  await scorable.PostAsync(item, state, token);
58  return true;
59  }
60 
61  return false;
62  }
63  finally
64  {
65  await scorable.DoneAsync(item, state, token);
66  }
67  }

96  public static IScorable<Item, Score> Fold<Item, Score>(this IEnumerable<IScorable<Item, Score>> scorables, IComparer<Score> comparer)
97  {
98  var list = scorables as IReadOnlyList<IScorable<Item, Score>>;
99  if (list != null)
100  {
101  if (list.Count == 0)
102  {
103  return NullScorable<Item, Score>.Instance;
104  }
105  else if (list.Count == 1)
106  {
107  return list[0];
108  }
109  else if (list.All(s => s is NullScorable<Item, Score>))
110  {
111  return NullScorable<Item, Score>.Instance;
112  }
113  }
114 
115  return new FoldScorable<Item, Score>(comparer, scorables);

226  protected override bool OnFold(IScorable<Item, Score> scorable, Item item, object state, Score score)
227  {
228  if (this.comparer.Compare(score, this.traits.Minimum) < 0)
229  {
230  throw new ArgumentOutOfRangeException(nameof(score));
231  }
232 
233  var maximum = this.comparer.Compare(score, this.traits.Maximum);
234  if (maximum > 0)
235  {
236  throw new ArgumentOutOfRangeException(nameof(score));
237  }
238  else if (maximum == 0)
239  {
240  return false;
241  }
242 
243  return true;
244  }
