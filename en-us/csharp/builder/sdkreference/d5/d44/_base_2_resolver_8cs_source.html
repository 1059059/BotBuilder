186  public override bool TryResolve(Type type, object tag, out object value)
187  {
188  if (tag == null)
189  {
190  for (int index = 0; index < this.services.Count; ++index)
191  {
192  var service = this.services[index];
193  if (service != null)
194  {
195  var serviceType = service.GetType();
196  if (type.IsAssignableFrom(serviceType))
197  {
198  value = service;
199  return true;
200  }
201  }
202  }
203  }
204
