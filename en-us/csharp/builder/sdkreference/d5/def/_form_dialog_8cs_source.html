222  async Task IDialog<T>.StartAsync(IDialogContext context)
223  {
224  if (this._entities.Any())
225  {
226  var inputs = new List<Tuple<int, string>>();
227  var entityGroups = (from entity in this._entities group entity by entity.Type);
228  foreach (var entityGroup in entityGroups)
229  {
230  var step = _form.Step(entityGroup.Key);
231  if (step != null)
232  {
233  var builder = new StringBuilder();
234  var first = true;
235  foreach (var entity in entityGroup)
236  {
237  if (first)
238  {
239  first = false;
240  }
241  else
242  {
243  builder.Append(' ');
244  }
245  builder.Append(entity.Entity);
246  }
247  inputs.Add(Tuple.Create(_form.StepIndex(step), builder.ToString()));
248  }
249  }
250  if (inputs.Any())
251  {
252  // Descending because last entry is first processed
253  _formState.FieldInputs = (from input in inputs orderby input.Item1 descending select input).ToList();
254  }
255  }
256  await SkipSteps();
257  _formState.Step = 0;
258  _formState.StepState = null;
259 
260  if (this._options.HasFlag(FormOptions.PromptInStart))
261  {
262  await MessageReceived(context, null);
263  }
264  else
265  {
266  context.Wait(MessageReceived);
267  }
268  }

270  public async Task MessageReceived(IDialogContext context, IAwaitable<Connector.IMessageActivity> toBot)
271  {
272  try
273  {
274  var toBotText = (toBot != null ? (await toBot).Text : null);
275  var stepInput = toBotText == null ? "" : toBotText.Trim();
276  if (stepInput.StartsWith("\""))
277  {
278  stepInput = stepInput.Substring(1);
279  }
280  if (stepInput.EndsWith("\""))
281  {
282  stepInput = stepInput.Substring(0, stepInput.Length - 1);
283  }
284  // Ensure we have initial definition for field steps
285  foreach (var step in _form.Steps)
286  {
287  if (step.Type == StepType.Field && step.Field.Prompt == null)
288  {
289  await step.DefineAsync(_state);
290  }
291  }
292  var next = (_formState.Next == null ? new NextStep() : ActiveSteps(_formState.Next, _state));
293  bool waitForMessage = false;
294  FormPrompt lastPrompt = _formState.LastPrompt;
295  Func<FormPrompt, Task<FormPrompt>> PostAsync = async (prompt) =>
296  {
297  return await _form.Prompt(context, prompt);
298  };
299  Func<IStep<T>, IEnumerable<TermMatch>, Task<bool>> DoStepAsync = async (step, matches) =>
300  {
301  var result = await step.ProcessAsync(context, _state, _formState, stepInput, matches);
302  await SkipSteps();
303  next = result.Next;
304  if (result.Feedback?.Prompt != null)
305  {
306  await PostAsync(result.Feedback);
307  if (_formState.Phase() != StepPhase.Completed)
308  {
309  if (!_formState.ProcessInputs)
310  {
311  await PostAsync(lastPrompt);
312  waitForMessage = true;
313  }
314  else
315  {
316  // After feedback, reset to ready
317  _formState.SetPhase(StepPhase.Ready);
318  }
319  }
320  }
321  if (result.Prompt != null)
322  {
323  lastPrompt = await PostAsync(result.Prompt);
324  waitForMessage = true;
325  }
326  return true;
327  };
328  while (!waitForMessage && MoveToNext(next))
329  {
330  IStep<T> step = null;
331  IEnumerable<TermMatch> matches = null;
332  if (next.Direction == StepDirection.Named && next.Names.Count() > 1)
333  {
334  // We need to choose between multiple next steps
335  bool start = (_formState.Next == null);
336  _formState.Next = next;
337  step = new NavigationStep<T>(_form.Steps[_formState.Step].Name, _form, _state, _formState);
338  if (start)
339  {
340  lastPrompt = await PostAsync(step.Start(context, _state, _formState));
341  waitForMessage = true;
342  }
343  else
344  {
345  // Responding
346  matches = step.Match(context, _state, _formState, stepInput);
347  }
348  }
349  else
350  {
351  // Processing current step
352  step = _form.Steps[_formState.Step];
353  if (await step.DefineAsync(_state))
354  {
355  if (_formState.Phase() == StepPhase.Ready)
356  {
357  if (step.Type == StepType.Message)
358  {
359  await PostAsync(step.Start(context, _state, _formState));
360  next = new NextStep();
361  }
362  else if (_formState.ProcessInputs)
363  {
364  stepInput = _formState.FieldInputs.Last().Item2;
365  lastPrompt = step.Start(context, _state, _formState);
366  }
367  else
368  {
369  lastPrompt = await PostAsync(step.Start(context, _state, _formState));
370  waitForMessage = true;
371  }
372  }
373  else if (_formState.Phase() == StepPhase.Responding)
374  {
375  matches = step.Match(context, _state, _formState, stepInput);
376  }
377  }
378  else
379  {
380  _formState.SetPhase(StepPhase.Completed);
381  lastPrompt = null;
382  next = new NextStep(StepDirection.Next);
383  }
384  }
385  if (matches != null)
386  {
387  matches = MatchAnalyzer.Coalesce(matches, stepInput).ToArray();
388  if (MatchAnalyzer.IsFullMatch(stepInput, matches))
389  {
390  await DoStepAsync(step, matches);
391  }
392  else
393  {
394  // Filter non-active steps out of command matches
395  var commands =
396  (toBotText == null || toBotText.Trim().StartsWith("\""))
397  ? new TermMatch[0]
398  : (from command in MatchAnalyzer.Coalesce(_commands.Prompt.Recognizer.Matches(toBotText), toBotText)
399  where (command.Value is FormCommand
400  || (!_formState.ProcessInputs && _form.Fields.Field((string)command.Value).Active(_state)))
401  select command).ToArray();
402  if (MatchAnalyzer.IsFullMatch(toBotText, commands))
403  {
404  FormPrompt feedback;
405  next = DoCommand(context, _state, _formState, step, commands, out feedback);
406  if (feedback != null)
407  {
408  await PostAsync(feedback);
409  await PostAsync(lastPrompt);
410  waitForMessage = true;
411  }
412  }
413  else
414  {
415  if (matches.Count() == 0 && commands.Count() == 0)
416  {
417  await PostAsync(step.NotUnderstood(context, _state, _formState, stepInput));
418  if (_formState.ProcessInputs && !step.InClarify(_formState))
419  {
420  _formState.SetPhase(StepPhase.Ready);
421  }
422  else
423  {
424  waitForMessage = true;
425  }
426  }
427  else
428  {
429  // Go with response since it looks possible
430  var bestMatch = MatchAnalyzer.BestMatches(matches, commands);
431  if (bestMatch == 0)
432  {
433  await DoStepAsync(step, matches);
434  }
435  else
436  {
437  FormPrompt feedback;
438  next = DoCommand(context, _state, _formState, step, commands, out feedback);
439  if (feedback != null)
440  {
441  await PostAsync(feedback);
442  await PostAsync(lastPrompt);
443  waitForMessage = true;
444  }
445  }
446  }
447  }
448  }
449  }
450  next = ActiveSteps(next, _state);
451  }
452  if (next.Direction == StepDirection.Complete || next.Direction == StepDirection.Quit)
453  {
454  if (next.Direction == StepDirection.Complete)
455  {
456  if (_form.Completion != null)
457  {
458  await _form.Completion(context, _state);
459  }
460  context.Done(_state);
461  }
462  else if (next.Direction == StepDirection.Quit)
463  {
464  throw new FormCanceledException<T>("Form quit.")
465  {
466  LastForm = _state,
467  Last = _form.Steps[_formState.Step].Name,
468  Completed = (from step in _form.Steps
469  where _formState.Phase(_form.StepIndex(step)) == StepPhase.Completed
470  select step.Name).ToArray()
471  };
472  }
473  else
474  {
475  throw new NotImplementedException();
476  }
477  }
478  else
479  {
480  _formState.LastPrompt = (FormPrompt)lastPrompt?.Clone();
481  context.Wait(MessageReceived);
482  }
483  }
484  catch (Exception inner)
485  {
486  if (!(inner is FormCanceledException<T>))
487  {
488  throw new FormCanceledException<T>(inner.Message, inner)
489  {
490  LastForm = _state,
491  Last = _form.Steps[_formState.Step].Name,
492  Completed = (from step in _form.Steps
493  where _formState.Phase(_form.StepIndex(step)) == StepPhase.Completed
494  select step.Name).ToArray()
495  };
496  }
497  else
498  {
499  throw;
500  }
501  }
502  }

508  private async Task SkipSteps()
509  {
510  if (!_options.HasFlag(FormOptions.PromptFieldsWithValues))
511  {
512  // Skip steps that already have a value if they are nullable and valid.
513  foreach (var step in _form.Steps)
514  {
515  int stepi = _form.StepIndex(step);
516  if (step.Type == StepType.Field
517  && _formState.Phase(stepi) == StepPhase.Ready
518  && !step.Field.IsUnknown(_state)
519  && step.Field.IsNullable)
520  {
521  var defined = await step.DefineAsync(_state);
522  if (defined)
523  {
524  var val = step.Field.GetValue(_state);
525  var result = await step.Field.ValidateAsync(_state, val);
526  if (result.IsValid)
527  {
528  bool ok = true;
529  double min, max;
530  if (step.Field.Limits(out min, out max))
531  {
532  var num = (double)Convert.ChangeType(val, typeof(double));
533  ok = (num >= min && num <= max);
534  }
535  if (ok)
536  {
537  _formState.SetPhase(stepi, StepPhase.Completed);
538  }
539  }
540  }
541  }
542  }
543  }
544  }

546  private NextStep ActiveSteps(NextStep next, T state)
547  {
548  var result = next;
549  if (next.Direction == StepDirection.Named)
550  {
551  var names = (from name in next.Names where _form.Fields.Field(name).Active(state) select name);
552  var count = names.Count();
553  if (count == 0)
554  {
555  result = new NextStep();
556  }
557  else if (count != result.Names.Count())
558  {
559  result = new NextStep(names);
560  }
561  }
562  return result;
563  }

570  private bool MoveToNext(NextStep next)
571  {
572  bool found = false;
573  switch (next.Direction)
574  {
575  case StepDirection.Complete:
576  break;
577  case StepDirection.Named:
578  _formState.StepState = null;
579  if (next.Names.Count() == 0)
580  {
581  goto case StepDirection.Next;
582  }
583  else if (next.Names.Count() == 1)
584  {
585  var name = next.Names.First();
586  var nextStep = -1;
587  for (var i = 0; i < _form.Steps.Count(); ++i)
588  {
589  if (_form.Steps[i].Name == name)
590  {
591  nextStep = i;
592  break;
593  }
594  }
595  if (nextStep == -1)
596  {
597  throw new ArgumentOutOfRangeException("NextStep", "Does not correspond to a field in the form.");
598  }
599  if (_form.Steps[nextStep].Active(_state))
600  {
601  var current = _form.Steps[_formState.Step];
602  _formState.SetPhase(_form.Fields.Field(current.Name).IsUnknown(_state) ? StepPhase.Ready : StepPhase.Completed);
603  _formState.History.Push(_formState.Step);
604  _formState.Step = nextStep;
605  _formState.SetPhase(StepPhase.Ready);
606  found = true;
607  }
608  else
609  {
610  // If we went to a state which is not active fall through to the next active if any
611  goto case StepDirection.Next;
612  }
613  }
614  else
615  {
616  // Always mark multiple names as found so we can handle the user navigation
617  found = true;
618  }
619  break;
620  case StepDirection.Next:
621  {
622  var start = _formState.Step;
623  // Reset any non-optional field step that has been reset to no value
624  for (var i = 0; i < _form.Steps.Count; ++i)
625  {
626  var step = _form.Steps[i];
627  if (step.Type == StepType.Field && _formState.Phase(i) == StepPhase.Completed && !step.Field.Optional && step.Field.IsUnknown(_state))
628  {
629  _formState.SetPhase(i, StepPhase.Ready);
630  }
631  }
632  // Next ready step including current one
633  for (var offset = 0; offset < _form.Steps.Count; ++offset)
634  {
635  var istep = (start + offset) % _form.Steps.Count;
636  var step = _form.Steps[istep];
637  _formState.Step = istep;
638  if (offset > 0)
639  {
640  _formState.StepState = null;
641  _formState.Next = null;
642  }
643  if ((_formState.Phase(istep) == StepPhase.Ready || _formState.Phase(istep) == StepPhase.Responding)
644  && step.Active(_state))
645  {
646  // Ensure all dependencies have values
647  foreach (var dependency in step.Dependencies)
648  {
649  var dstep = _form.Step(dependency);
650  var dstepi = _form.StepIndex(dstep);
651  if (dstep.Active(_state) && _formState.Phases[dstepi] != StepPhase.Completed)
652  {
653  _formState.Step = dstepi;
654  break;
655  }
656  }
657  found = true;
658  break;
659  }
660  }
661  if (!found)
662  {
663  next.Direction = StepDirection.Complete;
664  }
665  else
666  {
667  var normalStep = _formState.Step;
668  // Process initial messages first, then FieldInputs
669  if ((_formState.ProcessInputs || _form.Steps[normalStep].Type != StepType.Message) && _formState.FieldInputs != null)
670  {
671  // Override normal choice with FieldInputs
672  Func<bool> NextFieldInput = () =>
673  {
674  var foundInput = false;
675  while (_formState.FieldInputs.Any() && !foundInput)
676  {
677  var possible = _formState.FieldInputs.Last().Item1;
678  if (_form.Steps[possible].Active(_state))
679  {
680  _formState.Step = possible;
681  foundInput = true;
682  }
683  else
684  {
685  _formState.FieldInputs.Pop();
686  }
687  }
688  if (!_formState.FieldInputs.Any())
689  {
690  if (_options.HasFlag(FormOptions.PromptFieldsWithValues))
691  {
692  _formState.Reset();
693  }
694  else
695  {
696  _formState.ProcessInputs = false;
697  _formState.FieldInputs = null;
698  _formState.Step = 0;
699  }
700  }
701  return foundInput;
702  };
703  if (!_formState.ProcessInputs)
704  {
705  // Start of processing inputs
706  _formState.ProcessInputs = NextFieldInput();
707  }
708  else if (_formState.Phase(start) == StepPhase.Completed || _formState.Phase(start) == StepPhase.Ready)
709  {
710  // Reset state of just completed step
711  if (_options.HasFlag(FormOptions.PromptFieldsWithValues))
712  {
713  _formState.SetPhase(StepPhase.Ready);
714  }
715  // Move on to next field input if any
716  _formState.FieldInputs.Pop();
717  NextFieldInput();
718  }
719  }
720  else
721  {
722  if (_formState.Step != start && _form.Steps[start].Type != StepType.Message)
723  {
724  _formState.History.Push(start);
725  }
726  }
727  }
728  }
729  break;
730  case StepDirection.Previous:
731  while (_formState.History.Count() > 0)
732  {
733  var lastStepIndex = _formState.History.Pop();
734  var lastStep = _form.Steps[lastStepIndex];
735  if (lastStep.Active(_state))
736  {
737  var step = _form.Steps[_formState.Step];
738  _formState.SetPhase(step.Field.IsUnknown(_state) ? StepPhase.Ready : StepPhase.Completed);
739  _formState.Step = lastStepIndex;
740  _formState.SetPhase(StepPhase.Ready);
741  _formState.StepState = null;
742  _formState.Next = null;
743  found = true;
744  break;
745  }
746  }
747  if (!found)
748  {
749  next.Direction = StepDirection.Quit;
750  }
751  break;
752  case StepDirection.Quit:
753  break;
754  case StepDirection.Reset:
755  _formState.Reset();
756  // Because we redo phase they can go through everything again but with defaults.
757  found = true;
758  break;
759  }
760  return found;
761  }

763  private NextStep DoCommand(IDialogContext context, T state, FormState form, IStep<T> step, IEnumerable<TermMatch> matches, out FormPrompt feedback)
764  {
765  // TODO: What if there are more than one command?
766  feedback = null;
767  var next = new NextStep();
768  var value = matches.First().Value;
769  if (value is FormCommand)
770  {
771  switch ((FormCommand)value)
772  {
773  case FormCommand.Backup:
774  {
775  next.Direction = step.Back(context, state, form) ? StepDirection.Next : StepDirection.Previous;
776  }
777  break;
778  case FormCommand.Help:
779  {
780  var field = step.Field;
781  var builder = new StringBuilder();
782  foreach (var entry in _form.Configuration.Commands)
783  {
784  builder.Append("* ");
785  builder.AppendLine(entry.Value.Help);
786  }
787  var navigation = new Prompter<T>(field.Template(TemplateUsage.NavigationCommandHelp), _form, null);
788  var active = (from istep in _form.Steps
789  where !form.ProcessInputs && istep.Type == StepType.Field && istep.Active(state)
790  select istep.Field.FieldDescription.Description).ToArray();
791  if (active.Length > 1)
792  {
793  var activeList = Language.BuildList(active, navigation.Annotation.ChoiceSeparator, navigation.Annotation.ChoiceLastSeparator);
794  builder.Append("* ");
795  builder.Append(navigation.Prompt(state, null, activeList));
796  }
797  feedback = step.Help(state, form, builder.ToString());
798  }
799  break;
800  case FormCommand.Quit: next.Direction = StepDirection.Quit; break;
