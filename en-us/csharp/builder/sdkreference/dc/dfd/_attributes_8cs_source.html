674  public abstract class TemplateBaseAttribute : FormFlowAttribute
675  {
676  private static Random _generator = new Random();
677 
681  public BoolDefault AllowDefault { get; set; }
682 
686  public CaseNormalization ChoiceCase { get; set; }
687 
692  public string ChoiceFormat { get; set; }
693 
697  public string ChoiceLastSeparator { get; set; }
698 
702  public BoolDefault ChoiceParens { get; set; }
703 
707  public string ChoiceSeparator { get; set; }
708 
712  public ChoiceStyleOptions ChoiceStyle { get; set; }
713 
717  public FeedbackOptions Feedback { get; set; }
718 
722  public CaseNormalization FieldCase { get; set; }
723 
727  public string LastSeparator { get; set; }
728 
732  public string Separator { get; set; }
733 
737  public CaseNormalization ValueCase { get; set; }
738 
739  internal bool AllowNumbers
740  {
741  get
742  {
743  // You can match on numbers only if they are included in Choices and choices are shown
744  return ChoiceFormat.Contains("{0}") && Patterns.Any((pattern) => pattern.Contains("{||}"));
745  }
746  }
747 
753  public string Pattern()
754  {
755  var choice = 0;
756  if (Patterns.Length > 1)
757  {
758  lock (_generator)
759  {
760  choice = _generator.Next(Patterns.Length);
761  }
762  }
763  return Patterns[choice];
764  }
765 
770  public string[] Patterns { get; set; }
771 
776  public void ApplyDefaults(TemplateBaseAttribute defaultTemplate)
777  {
778  if (AllowDefault == BoolDefault.Default) AllowDefault = defaultTemplate.AllowDefault;
779  if (ChoiceCase == CaseNormalization.Default) ChoiceCase = defaultTemplate.ChoiceCase;
780  if (ChoiceFormat == null) ChoiceFormat = defaultTemplate.ChoiceFormat;
781  if (ChoiceLastSeparator == null) ChoiceLastSeparator = defaultTemplate.ChoiceLastSeparator;
782  if (ChoiceParens == BoolDefault.Default) ChoiceParens = defaultTemplate.ChoiceParens;
783  if (ChoiceSeparator == null) ChoiceSeparator = defaultTemplate.ChoiceSeparator;
784  if (ChoiceStyle == ChoiceStyleOptions.Default) ChoiceStyle = defaultTemplate.ChoiceStyle;
785  if (FieldCase == CaseNormalization.Default) FieldCase = defaultTemplate.FieldCase;
786  if (Feedback == FeedbackOptions.Default) Feedback = defaultTemplate.Feedback;
787  if (LastSeparator == null) LastSeparator = defaultTemplate.LastSeparator;
788  if (Separator == null) Separator = defaultTemplate.Separator;
789  if (ValueCase == CaseNormalization.Default) ValueCase = defaultTemplate.ValueCase;
790  }
791 
796  public TemplateBaseAttribute(params string[] patterns)
797  {
798  Patterns = patterns;
799  Initialize();
800  }
801 
806  public TemplateBaseAttribute(TemplateBaseAttribute other)
807  {
808  Patterns = other.Patterns;
809  AllowDefault = other.AllowDefault;
810  ChoiceCase = other.ChoiceCase;
811  ChoiceFormat = other.ChoiceFormat;
812  ChoiceLastSeparator = other.ChoiceLastSeparator;
813  ChoiceParens = other.ChoiceParens;
814  ChoiceSeparator = other.ChoiceSeparator;
815  ChoiceStyle = other.ChoiceStyle;
816  FieldCase = other.FieldCase;
817  Feedback = other.Feedback;
818  LastSeparator = other.LastSeparator;
819  Separator = other.Separator;
820  ValueCase = other.ValueCase;
821  }
822 
823  private void Initialize()
824  {
825  AllowDefault = BoolDefault.Default;
826  ChoiceCase = CaseNormalization.Default;
827  ChoiceFormat = null;
828  ChoiceLastSeparator = null;
829  ChoiceParens = BoolDefault.Default;
830  ChoiceSeparator = null;
831  ChoiceStyle = ChoiceStyleOptions.Default;
832  FieldCase = CaseNormalization.Default;
833  Feedback = FeedbackOptions.Default;
834  LastSeparator = null;
835  Separator = null;
836  ValueCase = CaseNormalization.Default;
837  }
838  }
