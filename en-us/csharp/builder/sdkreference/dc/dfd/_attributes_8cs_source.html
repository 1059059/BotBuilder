732  internal bool AllowNumbers
733  {
734  get
735  {
736  // You can match on numbers only if they are included in Choices and choices are shown
737  return ChoiceFormat.Contains("{0}") && Patterns.Any((pattern) => pattern.Contains("{||}"));
738  }
739  }

746  public string Pattern()
747  {
748  var choice = 0;
749  if (Patterns.Length > 1)
750  {
751  lock (_generator)
752  {
753  choice = _generator.Next(Patterns.Length);
754  }
755  }
756  return Patterns[choice];
757  }
758 
763  public string[] Patterns { get; set; }
764 
769  public void ApplyDefaults(TemplateBaseAttribute defaultTemplate)
770  {
771  if (AllowDefault == BoolDefault.Default) AllowDefault = defaultTemplate.AllowDefault;
772  if (ChoiceCase == CaseNormalization.Default) ChoiceCase = defaultTemplate.ChoiceCase;
773  if (ChoiceFormat == null) ChoiceFormat = defaultTemplate.ChoiceFormat;
774  if (ChoiceLastSeparator == null) ChoiceLastSeparator = defaultTemplate.ChoiceLastSeparator;
775  if (ChoiceParens == BoolDefault.Default) ChoiceParens = defaultTemplate.ChoiceParens;
776  if (ChoiceSeparator == null) ChoiceSeparator = defaultTemplate.ChoiceSeparator;
777  if (ChoiceStyle == ChoiceStyleOptions.Default) ChoiceStyle = defaultTemplate.ChoiceStyle;
778  if (FieldCase == CaseNormalization.Default) FieldCase = defaultTemplate.FieldCase;
779  if (Feedback == FeedbackOptions.Default) Feedback = defaultTemplate.Feedback;
780  if (LastSeparator == null) LastSeparator = defaultTemplate.LastSeparator;
781  if (Separator == null) Separator = defaultTemplate.Separator;
782  if (ValueCase == CaseNormalization.Default) ValueCase = defaultTemplate.ValueCase;
783  }
