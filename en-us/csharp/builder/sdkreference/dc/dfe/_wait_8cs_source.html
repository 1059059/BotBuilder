241  async Task<IWait<C>> IWait<C>.PollAsync(IFiber<C> fiber, C context, CancellationToken token)
242  {
243  this.ValidateNeed(Need.Poll);
244 
245  this.need = Need.Call;
246  try
247  {
248  return await this.rest(fiber, context, this, token);
249  }
250  finally
251  {
252  this.need = Need.Done;

256  void IWait.Post<D>(D item)
257  {
258  this.ValidateNeed(Need.Wait);
259 
260  var post = this as IPost<D>;
261  if (post == null)
262  {
263  if (typeof(D).IsValueType)
264  {
265  var postBoxed = this as IPost<object>;

280  post.Post(item);
