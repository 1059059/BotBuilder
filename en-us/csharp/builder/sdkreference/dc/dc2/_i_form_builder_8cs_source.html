424  public TemplateAttribute Template(TemplateUsage usage)
425  {
426  TemplateAttribute result = null;
427  foreach (var template in Templates)
428  {
429  if (template.Usage == usage)
430  {
431  result = template;
432  break;
433  }
434  }
435  Debug.Assert(result != null);
436  return result;
437  }
