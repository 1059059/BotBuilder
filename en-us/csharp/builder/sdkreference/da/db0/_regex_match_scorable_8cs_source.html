93  IScorable<IResolver, Match> IScorableFactory<IResolver, Match>.ScorableFor(IEnumerable<MethodInfo> methods)
94  {
95  var specs =
96  from method in methods
97  from pattern in InheritedAttributes.For<RegexPatternAttribute>(method)
98  select new { method, pattern };
99 
100  var scorableByMethod = methods.ToDictionary(m => m, m => new MethodScorable(m));
101 
102  // for a given regular expression pattern, fold the corresponding method scorables together to enable overload resolution
103  var scorables =
104  from spec in specs
105  group spec by spec.pattern into patterns
106  let method = patterns.Select(m => scorableByMethod[m.method]).ToArray().Fold(Binding.ResolutionComparer.Instance)
107  let regex = this.make(patterns.Key.Pattern)
108  select new RegexMatchScorable<Binding, Binding>(regex, method);
109 
110  var all = scorables.ToArray().Fold(MatchComparer.Instance);
111 
112  return all;
113  }
114  }

132  public override bool TryResolve(Type type, object tag, out object value)
133  {
134  var name = tag as string;
135  if (name != null)
136  {
137  var capture = this.Match.Groups[name];
138  if (capture != null && capture.Success)
139  {
140  if (type.IsAssignableFrom(typeof(Capture)))
141  {
142  value = capture;
143  return true;
144  }
145  else if (type.IsAssignableFrom(typeof(string)))
146  {
147  value = capture.Value;
148  return true;
149  }
150  }
151  }
152 
153  if (type.IsAssignableFrom(typeof(Regex)))
154  {

156  return true;
157  }
158 
159  if (type.IsAssignableFrom(typeof(Match)))
160  {
161  value = this.Match;
162  return true;
163  }
164 
165  var captures = this.Match.Captures;
166  if (type.IsAssignableFrom(typeof(CaptureCollection)))
167  {
168  value = captures;
169  return true;
170  }
171 
172  // i.e. for IActivity

183  {
184  IMessageActivity message;
185  if (!resolver.TryResolve(null, out message))
186  {
187  return null;

189

191  if (text == null)
192  {
193  return null;
194  }
195

197  if (!match.Success)
198  {
199  return null;
200  }
201 
202  var scope = new Scope(this.regex, match, resolver);
203  scope.Scorable = this.inner;
204  scope.State = await this.inner.PrepareAsync(scope, token);
205  return scope;
206  }
