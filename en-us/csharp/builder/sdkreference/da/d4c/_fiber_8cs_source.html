201  async Task<IWait<C>> IFiberLoop<C>.PollAsync(C context, CancellationToken token)
202  {
203  while (this.stack.Count > 0)
204  {
205  var leaf = this.stack.Peek();
206  var wait = leaf.Wait;
207  switch (wait.Need)
208  {
209  case Need.None:
210  case Need.Wait:
211  case Need.Done:
212  return wait;
213  case Need.Poll:
214  break;
215  default:
216  throw new InvalidNeedException(wait, Need.Poll);
217  }
218 
219  try
220  {
221  var next = await leaf.PollAsync(this, context, token);
222  var peek = this.stack.Peek();
223  bool fine = object.ReferenceEquals(next, peek.Wait) || next is NullWait<C>;
224  if (!fine)
225  {
226  throw new InvalidNextException(next);
227  }
228  }
229  catch (Exception error)
230  {
231  this.stack.Pop();
232  if (this.stack.Count == 0)
233  {
234  throw;
235  }
236  else
237  {
238  var parent = this.stack.Peek();
239  parent.Wait.Fail(error);
240  }
241  }
242  }
243 
244  return NullWait<C>.Instance;
245  }
246  }
