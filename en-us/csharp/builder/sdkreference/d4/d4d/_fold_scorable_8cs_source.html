62  public override async Task<Token<Item, Score>> PrepareAsync(Item item, CancellationToken token)
63  {
64  Score maximumScore = default(Score);
65  object maximumState = null;
66  IScorable<Item, Score> maximumScorable = null;
67 
68  foreach (var scorable in this.scorables)
69  {
70  var state = await scorable.PrepareAsync(item, token);
71  if (scorable.HasScore(item, state))
72  {
73  var score = scorable.GetScore(item, state);
74  bool better;
75  if (maximumScorable == null)
76  {
77  better = true;
78  }
79  else
80  {
81  var compare = this.comparer.Compare(score, maximumScore);
82  better = compare > 0;
83  }
84 
85  if (better)
86  {
87  maximumScore = score;
88  maximumState = state;
89  maximumScorable = scorable;
90 
91  if (! OnFold(scorable, item, state, score))
92  {
93  break;
94  }
95  }
96  }
97  }
98 
99  if (maximumScorable != null)
100  {
101  return new Token<Item, Score>() { Scorable = maximumScorable, State = maximumState };
102  }
103  else
104  {
105  return null;
106  }
107  }
108 
109  public override Score GetScore(Item item, Token<Item, Score> state)
110  {
111  return state.Scorable.GetScore(item, state.State);
112  }
113  }
