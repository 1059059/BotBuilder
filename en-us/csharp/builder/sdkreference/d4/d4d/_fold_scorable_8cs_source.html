76  protected override async Task<IReadOnlyList<State>> PrepareAsync(Item item, CancellationToken token)
77  {
78  var states = new List<State>();
79 
80  foreach (var scorable in this.scorables)
81  {
82  var state = await scorable.PrepareAsync(item, token);
83  states.Add(new State(scorable, state));
84  if (scorable.HasScore(item, state))
85  {
86  var score = scorable.GetScore(item, state);
87  if (!OnFold(scorable, item, state, score))
88  {
89  break;
90  }
91  }
92  }
93 
94  states.Sort((one, two) =>
95  {
96  var oneHasScore = one.scorable.HasScore(item, one.state);
97  var twoHasScore = two.scorable.HasScore(item, two.state);
98  if (oneHasScore && twoHasScore)
99  {
100  var oneScore = one.scorable.GetScore(item, one.state);
101  var twoScore = two.scorable.GetScore(item, two.state);
102  return this.comparer.Compare(twoScore, oneScore);
103  }
104  else if (oneHasScore)
105  {
106  return -1;
107  }
108  else if (twoHasScore)
109  {
110  return +1;
111  }
112  else
113  {
114  return 0;
115  }
116  });
117 
118  return states;
119  }
120  protected override bool HasScore(Item item, IReadOnlyList<State> states)
121  {
122  if (states.Count > 0)
123  {
124  var state = states[0];
125  return state.scorable.HasScore(item, state.state);
126  }
127 
128  return false;
129  }
130  protected override Score GetScore(Item item, IReadOnlyList<State> states)

134  }
135  protected override Task PostAsync(Item item, IReadOnlyList<State> states, CancellationToken token)

141  }

144  return Task.FromCanceled(error.CancellationToken);
145  }
146  catch (Exception error)

148  return Task.FromException(error);
149  }
150  }

152  {

155  await state.scorable.DoneAsync(item, state.state, token);
156  }
157  }
