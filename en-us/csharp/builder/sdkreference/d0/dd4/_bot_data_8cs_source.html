112  public Task SaveAsync(BotDataKey key, BotStoreType botStoreType, object value)
113  {
114  try
115  {
116  if (!ValidateKey(key))
117  {
118  throw new ArgumentException("Invalid bot data key!");
119  }
120 
121  switch (botStoreType)
122  {
123  case BotStoreType.BotConversationData:
124  message.BotConversationData = value;
125  break;
126  case BotStoreType.BotPerUserInConversationData:
127  message.BotPerUserInConversationData = value;
128  break;
129  case BotStoreType.BotUserData:
130  message.BotUserData = value;
131  break;
132  default:
133  throw new ArgumentException($"key {botStoreType} is not supported by this message store");
134  }
135 
136  return Task.FromResult(Type.Missing);
137  }
138  catch(Exception e)
139  {
140  return Task.FromException(e);
141  }
142  }

144  public Task<T> LoadAsync<T>(BotDataKey key, BotStoreType botStoreType)
145  {
146  try
147  {
148  if (!ValidateKey(key))
149  {
150  throw new ArgumentException("Invalid bot data key!");
151  }
152 
153  var value = default(T);
154 
155  switch (botStoreType.ToString())
156  {
157  case nameof(message.BotConversationData):
158  value = (T)message.BotConversationData;
159  break;
160  case nameof(message.BotPerUserInConversationData):
161  value = (T)message.BotPerUserInConversationData;
162  break;
163  case nameof(message.BotUserData):
164  value = (T)message.BotUserData;
165  break;
166  default:
167  value = default(T);
168  break;
169  }
170  return Task.FromResult(value);
171  }
172  catch (Exception e)
173  {
174  return Task.FromException<T>(e);

193  public async Task<bool> FlushAsync(BotDataKey key)
194  {
195  DataEntry entry = default(DataEntry);
196  if (cache.TryRemove(key, out entry))
197  {
198  this.Save(key, entry);
199  return true;
200  }
201  else
202  {
203  return false;

207  public async Task<T> LoadAsync<T>(BotDataKey key, BotStoreType botStoreType)
208  {
209  DataEntry entry;
210  T obj = default(T);
211  if (!cache.TryGetValue(key, out entry))
212  {
213  entry = new DataEntry();
214  cache[key] = entry;
215 
216  string value;
217 
218  if (store.TryGetValue(GetKey(key, botStoreType), out value))
219  {
220  obj = (T)JsonConvert.DeserializeObject(value);
221  }
222 
223  SetValue(entry, botStoreType, obj);
224  return obj;
225  }
226  else
227  {
228  switch (botStoreType)
229  {
230  case BotStoreType.BotConversationData:
231  if (entry.BotConversationData != null)
232  {
233  obj = (T)entry.BotConversationData;
234  }
235  break;
236  case BotStoreType.BotPerUserInConversationData:
237  if (entry.BotPerUserInConversationData != null)
238  {
239  obj = (T)entry.BotPerUserInConversationData;
240  }
241  break;
242  case BotStoreType.BotUserData:
243  if (entry.BotUserData != null)
244  {
245  obj = (T)entry.BotUserData;
246  }
247  break;
248  }
249 
250  if(obj == null)
251  {
252  string value;
253  if (store.TryGetValue(GetKey(key, botStoreType), out value))
254  {
255  obj = (T)JsonConvert.DeserializeObject(value);
256  SetValue(entry, botStoreType, obj);
257  }
258  }
259 
260  return obj;
261  }
262  }

264  public async Task SaveAsync(BotDataKey key, BotStoreType botStoreType, object value)

292  private string GetKey(BotDataKey key, BotStoreType botStoreType)
293  {
294  switch (botStoreType)
295  {
296  case BotStoreType.BotConversationData:
297  return $"conversation:{key.BotId}:{key.ConversationId}";
298  case BotStoreType.BotUserData:
299  return $"user:{key.BotId}:{key.UserId}";
300  case BotStoreType.BotPerUserInConversationData:
301  return $"perUserInConversation:{key.BotId}:{key.UserId}:{key.ConversationId}";
302  default:
303  throw new ArgumentException("Unsupported bot store type!");
304  }
305  }
306 
307  private void Save(BotDataKey key, DataEntry entry)
308  {
309  if(entry?.BotConversationData != null)
310  {
311  store[GetKey(key, BotStoreType.BotConversationData)] = JsonConvert.SerializeObject(entry.BotConversationData);
312  }
313 
314  if(entry?.BotUserData != null)
315  {
316  store[GetKey(key, BotStoreType.BotUserData)] = JsonConvert.SerializeObject(entry.BotUserData);
317  }
318 
319  if(entry?.BotPerUserInConversationData != null)
320  {
321  store[GetKey(key, BotStoreType.BotPerUserInConversationData)] = JsonConvert.SerializeObject(entry.BotPerUserInConversationData);
322  }
323  }

388  private async Task<IBotDataBag> LoadData(BotStoreType botStoreType)
389  {
390  T data = await this.botDataStore.LoadAsync<T>(botDataKey, botStoreType);
391  if (data == null)
392  {
393  data = this.MakeData();
394  await this.botDataStore.SaveAsync(botDataKey, botStoreType, data);
395  }
396  return this.WrapData(data);
