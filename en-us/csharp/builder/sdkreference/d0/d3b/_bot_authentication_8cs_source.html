69  public override async Task OnActionExecutingAsync(HttpActionContext actionContext, CancellationToken cancellationToken)
70  {
71  var provider = this.GetCredentialProvider();
72  var botAuthenticator = new BotAuthenticator(provider, OpenIdConfigurationUrl, DisableEmulatorTokens);
73  var identityToken = await botAuthenticator.TryAuthenticateAsync(actionContext.Request, cancellationToken);
74 
75  // the request is not authenticated, fail with 401.
76  if (!identityToken.Authenticated)
77  {
78  actionContext.Response = GenerateUnauthorizedResponse(actionContext.Request);
79  return;
80  }
81 
82  // authenticated but no identity, auth is disabled;
83  if (identityToken.Authenticated && identityToken.Identity == null)
84  {
85  await base.OnActionExecutingAsync(actionContext, cancellationToken);
86  return;
87  }
88 
89  // trust the service url in activities
90  var activities = GetActivities(actionContext);
91  if (activities.Any())
92  {
93  foreach (var activity in activities)
94  {
95  MicrosoftAppCredentials.TrustServiceUrl(activity.ServiceUrl);
96  }
97  }
98  else
99  {
100  Trace.TraceWarning("No ServiceUrls added to trusted list");
101  }
102 
103  await base.OnActionExecutingAsync(actionContext, cancellationToken);
104  }

106  private IList<Activity> GetActivities(HttpActionContext actionContext)
107  {
108  var activties = actionContext.ActionArguments.Select(t => t.Value).OfType<Activity>().ToList();
109  if (activties.Any())
110  {
111  return activties;
112  }
113  else
114  {
115  var objects =
116  actionContext.ActionArguments.Where(t => t.Value is JObject || t.Value is JArray)
117  .Select(t => t.Value).ToArray();

119  {
120  activties = new List<Activity>();

213  internal async Task<IdentityToken> TryAuthenticateAsync(HttpRequestMessage request,
214  CancellationToken token)
215  {
216  // debugger is attached and no app Id is set, this implies that the auth is disabled
217  // and no token validation is necessary.
218  if (Debugger.IsAttached && this.credentialProvider is SimpleCredentialProvider)
219  {
220  if (String.IsNullOrEmpty(((SimpleCredentialProvider)this.credentialProvider).AppId))
221  {
222  // then auth is disabled
223  return new IdentityToken(true, null);
224  }
225  }
226 
227  ClaimsIdentity identity = null;
228  var tokenExtractor = GetTokenExtractor();
229  identity = await tokenExtractor.GetIdentityAsync(request);
230 
231  // No identity? If we're allowed to, fall back to MSA
232  // This code path is used by the emulator
233  if (identity == null && !this.disableEmulatorTokens)
234  {
235  tokenExtractor = new JwtTokenExtractor(JwtConfig.ToBotFromMSATokenValidationParameters, JwtConfig.ToBotFromMSAOpenIdMetadataUrl);
236  identity = await tokenExtractor.GetIdentityAsync(request);
237  }
238 
239  if (identity != null)
240  {
241  var appId = tokenExtractor.GetAppIdFromClaimsIdentity(identity);
242  if (await credentialProvider.IsValidAppIdAsync(appId) == false) // keep context
243  {
244  // not valid appid, drop the identity
245  identity = null;
246  }
247  else
248  {
249  var password = await credentialProvider.GetAppPasswordAsync(appId); // Keep context
250  if (password != null)
251  {
252  // add password as claim so that it is part of ClaimsIdentity and accessible by ConnectorClient()
253  identity.AddClaim(new Claim(ClaimsIdentityEx.AppPasswordClaim, password));
254  }
255  }
256  }
257 
258  if (identity != null)
259  {
260  Thread.CurrentPrincipal = new ClaimsPrincipal(identity);
261 
262  // Inside of ASP.NET this is required
263  if (HttpContext.Current != null)
264  HttpContext.Current.User = Thread.CurrentPrincipal;
265 
266  return new IdentityToken(true, identity);
267  }
268 
269  return new IdentityToken(false, null);
270  }
