69  public Task InvokeAsync(CancellationToken token)
70  {
71  try
72  {
73  // late-bound provide the CancellationToken
74  for (int index = 0; index < this.parameters.Count; ++index)
75  {
76  var type = this.parameters[index].ParameterType;
77  bool cancel = type.IsAssignableFrom(typeof(CancellationToken));
78  if (cancel)
79  {
80  this.arguments[index] = token;
81  }
82  }
83 
84  var result = this.method.Invoke(this.instance, this.arguments);
85  // if the result is a task, wait for its completion and propagate any exceptions
86  var task = result as Task;
87  if (task != null)
88  {
89  return task;
90  }
91 
92  return Task.CompletedTask;
93  }
94  catch (OperationCanceledException error)
95  {
96  return Task.FromCanceled(error.CancellationToken);
97  }
98  catch (Exception error)
99  {
100  return Task.FromException(error);
101  }
102  }

128  public sealed class ResolutionComparer : IComparer<Binding>
129  {
130  public static readonly IComparer<Binding> Instance = new ResolutionComparer();
131  private ResolutionComparer()
132  {
133  }
134  private bool TryCompareParameterTypeAssignability(Binding one, Binding two, int index, out int compare)
135  {
136  var l = one.parameters[index].ParameterType;
137  var r = two.parameters[index].ParameterType;
138  if (l.Equals(r))
139  {
140  compare = 0;
141  return true;
142  }
143  if (l.IsAssignableFrom(r))
144  {
145  compare = -1;
146  return true;
147  }
148  else if (r.IsAssignableFrom(l))
149  {
150  compare = +1;
151  return true;
152  }
153 
154  compare = 0;
155  return false;
156  }

158  public static int UpdateComparisonConsistently(Binding one, Binding two, int compareOld, int compareNew)
159  {
160  if (compareOld == 0)
161  {
162  return compareNew;
163  }
164  else if (compareNew == 0)
165  {
166  return compareOld;
167  }
168  else if (compareNew == compareOld)
169  {
170  return compareOld;
171  }
172  else
173  {
174  throw new MethodResolutionException("inconsistent parameter overrides", one, two);
175  }
176  }
177 
178  int IComparer<Binding>.Compare(Binding one, Binding two)
179  {
180  if (one.method.ReturnType != two.method.ReturnType)
181  {
182  throw new MethodResolutionException("inconsistent return types", one, two);
183  }
184 
185  int compare = 0;
186 
187  var count = Math.Min(one.parameters.Count, two.parameters.Count);
188  for (int index = 0; index < count; ++index)
189  {
190  int parameter;
191  if (TryCompareParameterTypeAssignability(one, two, index, out parameter))
192  {
193  compare = UpdateComparisonConsistently(one, two, compare, parameter);
194  }
195  else
196  {
197  throw new MethodResolutionException("inconsistent parameter types", one, two);
198  }
199  }
200 
201  int length = one.parameters.Count.CompareTo(two.parameters.Count);
202  compare = UpdateComparisonConsistently(one, two, compare, length);
203  return compare;
204  }
205  }
