46  public static void TrustServiceUrl(string serviceUrl, DateTime expirationTime = default(DateTime))
47  {
48  try
49  {
50  if (expirationTime == default(DateTime))
51  {
52  // by default the service url is valid for one day
53  TrustedHostNames.AddOrUpdate(new Uri(serviceUrl).Host, DateTime.UtcNow.AddDays(1), (key, oldValue) => DateTime.UtcNow.AddDays(1));

56  {

94  public async Task<string> GetTokenAsync(bool forceRefresh = false)
95  {
96  string token;
97  var oAuthToken = (OAuthResponse)System.Web.HttpRuntime.Cache.Get(TokenCacheKey);
98  if (oAuthToken != null && !forceRefresh && TokenNotExpired(oAuthToken))
99  {
100  token = oAuthToken.access_token;
101  }
102  else
103  {
104  oAuthToken = await RefreshTokenAsync().ConfigureAwait(false);
105  System.Web.HttpRuntime.Cache.Insert(TokenCacheKey,
106  oAuthToken,
107  null,
108  DateTime.UtcNow.AddSeconds(oAuthToken.expires_in),
109  System.Web.Caching.Cache.NoSlidingExpiration);
110  token = oAuthToken.access_token;
111  }
112  return token;
113  }

126  private static bool TrustedUri(Uri uri)

140  private async Task<OAuthResponse> RefreshTokenAsync()
141  {
142  OAuthResponse oauthResponse;
143 
144  using (HttpClient httpClient = new HttpClient())
145  {
146  HttpResponseMessage response = await httpClient.PostAsync(OAuthEndpoint, new FormUrlEncodedContent(new Dictionary<string, string>()
147  {
148  { "grant_type", "client_credentials" },
149  { "client_id", MicrosoftAppId },
150  { "client_secret", MicrosoftAppPassword },
151  { "scope", OAuthScope }
152  })).ConfigureAwait(false);
153 
154  string body = await response.Content.ReadAsStringAsync().ConfigureAwait(false);
155 
156  oauthResponse = JsonConvert.DeserializeObject<OAuthResponse>(body);
157  oauthResponse.expiration_time = DateTime.UtcNow.AddSeconds(oauthResponse.expires_in).Subtract(TimeSpan.FromSeconds(60));
158  return oauthResponse;
