90  Uri ILuisService.BuildUri(string text)
91  {
92  var id = HttpUtility.UrlEncode(this.model.ModelID);
93  var sk = HttpUtility.UrlEncode(this.model.SubscriptionKey);
94  var q = HttpUtility.UrlEncode(text);
95 
96  var builder = new UriBuilder(UriBase);
97  builder.Query = $"id={id}&subscription-key={sk}&q={q}";
98  return builder.Uri;
99  }
