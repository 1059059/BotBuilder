76  async Task<BotData> IBotDataStore<BotData>.LoadAsync(IAddress key, BotStoreType botStoreType, CancellationToken cancellationToken)
77  {
78  string serializedData;
79  serializedData = store.GetOrAdd(GetKey(key, botStoreType),
80  dictionaryKey => Serialize(new BotData { ETag = DateTime.UtcNow.Ticks.ToString() }));
81  return Deserialize(serializedData);

84  async Task IBotDataStore<BotData>.SaveAsync(IAddress key, BotStoreType botStoreType, BotData botData, CancellationToken cancellationToken)
85  {
86  lock (locks[botStoreType])
87  {
88  store.AddOrUpdate(GetKey(key, botStoreType), Serialize(botData), (dictionaryKey, value) =>
89  {
90  if (botData.ETag != "*" && Deserialize(value).ETag != botData.ETag)
91  {
92  throw new HttpException((int)HttpStatusCode.PreconditionFailed, "Inconsistent SaveAsync based on ETag!");
93  }
94  botData.ETag = DateTime.UtcNow.Ticks.ToString();
95  return Serialize(botData);
96  });
97  }
98  }

106  private static string GetKey(IAddress key, BotStoreType botStoreType)
107  {
108  switch (botStoreType)
109  {
110  case BotStoreType.BotConversationData:
111  return $"conversation:{key.BotId}:{key.ChannelId}:{key.ConversationId}";
112  case BotStoreType.BotUserData:
113  return $"user:{key.BotId}:{key.ChannelId}:{key.UserId}";
114  case BotStoreType.BotPrivateConversationData:
115  return $"privateConversation:{key.BotId}:{key.ChannelId}:{key.UserId}:{key.ConversationId}";
116  default:
117  throw new ArgumentException("Unsupported bot store type!");
118  }
119  }

121  private static string Serialize(BotData data)
122  {
123  using (var cmpStream = new MemoryStream())
124  using (var stream = new GZipStream(cmpStream, CompressionMode.Compress))
125  using (var streamWriter = new StreamWriter(stream))
126  {
127  var serializedJSon = JsonConvert.SerializeObject(data);
128  streamWriter.Write(serializedJSon);
129  streamWriter.Close();
130  stream.Close();
131  return Convert.ToBase64String(cmpStream.ToArray());
132  }
133  }
134 
135  private static BotData Deserialize(string str)
136  {
137  byte[] bytes = Convert.FromBase64String(str);
138  using (var stream = new MemoryStream(bytes))
139  using (var gz = new GZipStream(stream, CompressionMode.Decompress))
140  using (var streamReader = new StreamReader(gz))
141  {
142  return JsonConvert.DeserializeObject<BotData>(streamReader.ReadToEnd());
143  }
144  }
145  }

240  async Task<bool> IBotDataStore<BotData>.FlushAsync(IAddress key, CancellationToken cancellationToken)
241  {
242  CacheEntry entry;
243  if (cache.TryGetValue(key, out entry))
244  {
245  // Removing the cached entry to make sure that we are not leaking
246  // flushed entries when CachingBotDataStore is registered as a singleton object.
247  // Also since this store is not updating ETags on LoadAsync(...), there
248  // will be a conflict if we reuse the cached entries after flush.
249  cache.Remove(key);
250  await this.Save(key, entry, cancellationToken);
251  return true;
252  }
253  else
254  {
255  return false;
256  }
257  }

259  async Task<BotData> IBotDataStore<BotData>.LoadAsync(IAddress key, BotStoreType botStoreType, CancellationToken cancellationToken)
260  {
261  CacheEntry cacheEntry;
262  BotData value = null;
263  if (!cache.TryGetValue(key, out cacheEntry))
264  {
265  cacheEntry = new CacheEntry();
266  cache.Add(key, cacheEntry);
267  value = await LoadFromInnerAndCache(cacheEntry, botStoreType, key, cancellationToken);
268  }
269  else
270  {
271  switch (botStoreType)
272  {
273  case BotStoreType.BotConversationData:
274  if (cacheEntry.BotConversationData != null)
275  {
276  value = cacheEntry.BotConversationData;
277  }
278  break;
279  case BotStoreType.BotPrivateConversationData:
280  if (cacheEntry.BotPrivateConversationData != null)
281  {
282  value = cacheEntry.BotPrivateConversationData;
283  }
284  break;
285  case BotStoreType.BotUserData:
286  if (cacheEntry.BotUserData != null)
287  {
288  value = cacheEntry.BotUserData;
289  }
290  break;
291  default:
292  throw new NotImplementedException();
293  }
294 
295  if (value == null)
296  {
297  value = await LoadFromInnerAndCache(cacheEntry, botStoreType, key, cancellationToken);
298  }
299  }
300 
301  return value;
302  }

304  async Task IBotDataStore<BotData>.SaveAsync(IAddress key, BotStoreType botStoreType, BotData value, CancellationToken cancellationToken)
305  {
306  CacheEntry entry;
307  if (!cache.TryGetValue(key, out entry))
308  {
309  entry = new CacheEntry();
310  cache.Add(key, entry);
311  }
312 
313  SetCachedValue(entry, botStoreType, value);
314  }

316  private async Task<BotData> LoadFromInnerAndCache(CacheEntry cacheEntry, BotStoreType botStoreType, IAddress key, CancellationToken token)
317  {
318  var value = await inner.LoadAsync(key, botStoreType, token);
319 
320  if (value != null)
321  {
322  SetCachedValue(cacheEntry, botStoreType, value);
323  }
324  else
325  {
326  // inner store returned null, we create a new instance of BotData with ETag = "*"
327  value = new BotData() { ETag = "*" };
328  SetCachedValue(cacheEntry, botStoreType, value);
329  }
330  return value;
331  }

333  private void SetCachedValue(CacheEntry entry, BotStoreType botStoreType, BotData value)
334  {
335  switch (botStoreType)
336  {
337  case BotStoreType.BotConversationData:
338  entry.BotConversationData = value;
339  break;
340  case BotStoreType.BotPrivateConversationData:
341  entry.BotPrivateConversationData = value;
342  break;
343  case BotStoreType.BotUserData:
344  entry.BotUserData = value;
345  break;
346  default:
347  throw new NotImplementedException();
348  }
349  }

351  private async Task Save(IAddress key, CacheEntry entry, CancellationToken cancellationToken)
352  {
353  switch (this.dataConsistencyPolicy)
354  {
355  case CachingBotDataStoreConsistencyPolicy.LastWriteWins:
356  if (entry?.BotConversationData != null)
357  {
358  entry.BotConversationData.ETag = "*";
359  }
360 
361  if (entry?.BotUserData != null)
362  {
363  entry.BotUserData.ETag = "*";
364  }
365 
366  if (entry?.BotPrivateConversationData != null)
367  {
368  entry.BotPrivateConversationData.ETag = "*";
369  }
370  break;
371  case CachingBotDataStoreConsistencyPolicy.ETagBasedConsistency:
372  // no action needed, store relies on the ETags returned by inner store
373  break;
374  default:
375  throw new ArgumentException($"{this.dataConsistencyPolicy} is not a valid consistency policy!");
376  }
377 
378  var tasks = new List<Task>(capacity: 3);
379 
380  if (entry?.BotConversationData != null)
381  {
382  tasks.Add(inner.SaveAsync(key, BotStoreType.BotConversationData, entry.BotConversationData, cancellationToken));
383  }
384 
385  if (entry?.BotUserData != null)
386  {
387  tasks.Add(inner.SaveAsync(key, BotStoreType.BotUserData, entry.BotUserData, cancellationToken));
388  }
389 
390  if (entry?.BotPrivateConversationData != null)
391  {
392  tasks.Add(inner.SaveAsync(key, BotStoreType.BotPrivateConversationData, entry.BotPrivateConversationData, cancellationToken));
393  }
394  await Task.WhenAll(tasks);
395  }
396  }

486  {
487  var botData = await this.botDataStore.LoadAsync(botDataKey, botStoreType, cancellationToken);
488  if (botData?.Data == null)

490  botData.Data = this.MakeData();
491  await this.botDataStore.SaveAsync(botDataKey, botStoreType, botData, cancellationToken);
492  }
