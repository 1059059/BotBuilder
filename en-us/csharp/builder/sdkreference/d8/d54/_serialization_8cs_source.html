269  ISerializationSurrogate ISurrogateSelector.GetSurrogate(Type type, StreamingContext context, out ISurrogateSelector selector)
270  {
271  int maximumPriority = -int.MaxValue;
272  ISurrogateProvider maximumProvider = null;
273  for (int index = 0; index < this.providers.Count; ++index)
274  {
275  var provider = this.providers[index];
276  int priority;
277  if (provider.Handles(type, context, out priority) && priority > maximumPriority)
278  {
279  maximumPriority = priority;
280  maximumProvider = provider;
281  }
282  }
283 
284  if (maximumProvider != null)
285  {
286  selector = this;
287  return maximumProvider;
288  }
289  else
290  {
291  selector = null;
292  return null;
293  }
294  }
295  }
