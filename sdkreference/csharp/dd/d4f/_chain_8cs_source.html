531  private async Task AfterAntecedent(IDialogContext context, IAwaitable<T> result)
532  {
533  var itemT = await result;
534  R itemR = default(R);
535  foreach (var condition in this.Cases)
536  {
537  if (condition.Condition(itemT))
538  {
539  itemR = condition.Selector(context, itemT);

541  }
542  }
543  context.Done(itemR);
