60  private async Task MessageReceived(IDialogContext context, IAwaitable<Connector.Message> message)
61  {
62  var text = (await message).Text;
63  Command matched = null;
64  for (int idx = 0; idx < commands.Count; idx++)
65  {
66  var handler = commands[idx];
67  if (handler.Expression.Match(text).Success)
68  {
69  matched = handler;
70  break;
71  }
72  }
73 
74  if (matched == null && this.defaultCommand != null)
75  {
76  matched = this.defaultCommand;
77  }
78 
79  if (matched != null)
80  {
81  await matched.CommandHandler(context, message);
82  }
83  else
84  {
85  string error = $"CommandDialog doesn't have a registered command handler for this message: {text}";
86  throw new Exception(error);
87  }
88 
89  }
