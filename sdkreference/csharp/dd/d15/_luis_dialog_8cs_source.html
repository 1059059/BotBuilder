141  protected virtual async Task MessageReceived(IDialogContext context, IAwaitable<Message> item)
142  {
143  if (this.handlerByIntent == null)
144  {
145  this.handlerByIntent = new Dictionary<string, IntentHandler>(GetHandlersByIntent());
146  }
147 
148  var message = await item;
149  var messageText = await GetLuisQueryTextAsync(context, message);
150  var luisRes = await this.service.QueryAsync(messageText);
151 
152  var intent = BestIntentFrom(luisRes);
153 
154  IntentHandler handler = null;
155  if (intent == null || !this.handlerByIntent.TryGetValue(intent.Intent, out handler))
156  {
157  handler = this.handlerByIntent[string.Empty];
158  }
159 
160  if (handler != null)
161  {
162  await handler(context, luisRes);
163  }
164  else
165  {
166  var text = $"No default intent handler found.";
167  throw new Exception(text);
168  }
169  }

189  public static IEnumerable<KeyValuePair<string, IntentHandler>> EnumerateHandlers(object dialog)
190  {
191  var type = dialog.GetType();
192  var methods = type.GetMethods(BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Public);
193  foreach (var method in methods)
194  {
195  var intents = method.GetCustomAttributes<LuisIntentAttribute>(inherit: true).ToArray();
196  var intentHandler = (IntentHandler)Delegate.CreateDelegate(typeof(IntentHandler), dialog, method, throwOnBindFailure: false);
197  if (intentHandler != null)
198  {
199  var intentNames = intents.Select(i => i.IntentName).DefaultIfEmpty(method.Name);
200 
201  foreach (var intentName in intentNames)
202  {
203  var key = string.IsNullOrWhiteSpace(intentName) ? string.Empty : intentName;
204  yield return new KeyValuePair<string, IntentHandler>(intentName, intentHandler);
205  }
206  }
207  else
208  {
209  if (intents.Length > 0)
210  {
211  throw new InvalidIntentHandlerException(string.Join(";", intents.Select(i => i.IntentName)), method);
212  }
213  }
214  }
215  }
