112  protected async Task MessageReceived(IDialogContext context, IAwaitable<Message> item)
113  {
114  if (this.handlerByIntent == null)

116  this.handlerByIntent = EnumerateHandlers(this).ToDictionary(kv => kv.Key, kv => kv.Value);
117  }
118 
119  var message = await item;
120  var luisRes = await this.service.QueryAsync(message.Text);
121 
122  var maximum = luisRes.Intents.Max(t => t.Score);
123  var intent = luisRes.Intents.FirstOrDefault(i => i.Score == maximum);
124 
125  IntentHandler handler = null;
126  if (intent == null || !this.handlerByIntent.TryGetValue(intent.Intent, out handler))
127  {

133  await handler(context, luisRes);
