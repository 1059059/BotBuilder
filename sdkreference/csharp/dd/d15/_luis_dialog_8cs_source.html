140  protected async Task MessageReceived(IDialogContext context, IAwaitable<Message> item)
141  {
142  var message = await item;
143  var luisRes = await GetLuisResult(this.luisUrl, message.Text);
144  var intent = luisRes.Intents.FirstOrDefault(i => i.Score == luisRes.Intents.Select(t => t.Score).Max());
145 
146  if (this.handlerByIntent == null)
147  {
148  this.AddAttributeBasedHandlers();
149  }
150 
151  IntentHandler handler;
152  if (intent == null || !this.handlerByIntent.TryGetValue(intent.Intent, out handler))
153  {
154  handler = this.handlerByIntent[string.Empty];
155  }
156 
157  if (handler != null)
158  {
159  await handler(context, luisRes);
160  }
161  else
162  {
163  var text = $"LuisModel[{this.modelID}] no default intent handler.";
164  throw new Exception(text);
165  }
166  }

168  private void AddAttributeBasedHandlers()
169  {
170  if (this.handlerByIntent != null)
171  {
172  throw new InvalidOperationException();
173  }
174 
175  this.handlerByIntent = new Dictionary<string, IntentHandler>();
176 
177  var methods = from m in this.GetType().GetMethods(BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Public)
178  let attr = m.GetCustomAttributes(typeof(LuisIntent), true)
179  where attr.Length > 0
180  select new { method = m, attributes = attr.Select(s => (LuisIntent)s).ToList() };
181 
182  var intentHandlers = from m in methods
183  select new { method = m.method, intents = m.attributes.Select(i => i.intentName) };
184 
185  foreach (var handler in intentHandlers)
186  {
187  // TODO: use handler.method.CreateDelegate?
188  //var intentHandler = (IntentHandler) handler.method.CreateDelegate(typeof(IntentHandler));
189 
190  // assign to local variable to capture less in closure below
191  var method = handler.method;
192  var intentHandler = new IntentHandler(async (context, result) =>
193  {
194  var task = (Task)method.Invoke(this, new object[] { context, result });
195  await task;
196  });
197 
198  foreach (var intent in handler.intents)
199  {
200  var key = string.IsNullOrWhiteSpace(intent) ? string.Empty : intent;
201  this.handlerByIntent.Add(key, intentHandler);
202  }
203  }
204  }
205  }
