163  async Task IDialog.StartAsync(IDialogContext context)
164  {
165  var entityGroups = (from entity in this._entities group entity by entity.Type);
166  foreach (var entityGroup in entityGroups)
167  {
168  var step = _form.Step(entityGroup.Key);
169  if (step != null)
170  {
171  _formState.Step = _form.StepIndex(step);
172  _formState.StepState = null;
173  var builder = new StringBuilder();
174  foreach (var entity in entityGroup)
175  {
176  builder.Append(entity.Entity);
177  builder.Append(' ');
178  }
179  var input = builder.ToString();
180  string feedback;
181  string prompt = step.Start(context, _state, _formState);
182  var matches = MatchAnalyzer.Coalesce(step.Match(context, _state, _formState, input, out prompt), input);
183  if (MatchAnalyzer.IsFullMatch(input, matches, 0.5))
184  {
185  // TODO: In the case of clarification I could
186  // 1) Go through them while supporting only quit or back and reset
187  // 2) Drop them
188  // 3) Just pick one (found in form.StepState, but that is opaque here)
189  var result = await step.ProcessAsync(context, _state, _formState, input, matches);
190  feedback = result.Feedback;
191  prompt = result.Prompt;
192  }
193  else
194  {
195  _formState.SetPhase(StepPhase.Ready);
196  }
197  }
198  }
199  _formState.Step = 0;
200  _formState.StepState = null;
201 
202  if (this._options.HasFlag(FormOptions.PromptInStart))
203  {
204  await MessageReceived(context, null);
205  }
206  else
207  {
208  context.Wait(MessageReceived);
209  }
210  }

212  public async Task MessageReceived(IDialogContext context, IAwaitable<Connector.Message> toBot)
213  {
214  var toBotText = toBot != null ? (await toBot).Text : null;
215  string message = null;
216  string prompt = null;
217  bool useLastPrompt = false;
218  bool requirePrompt = false;
219  var next = (_formState.Next == null ? new NextStep() : ActiveSteps(_formState.Next, _state));
220  while (prompt == null && (message == null || requirePrompt) && MoveToNext(_state, _formState, next))
221  {
222  IStep<T> step;
223  IEnumerable<TermMatch> matches = null;
224  string lastInput = null;
225  string feedback = null;
226  if (next.Direction == StepDirection.Named && next.Names.Count() > 1)
227  {
228  // We need to choose between multiple next steps
229  bool start = (_formState.Next == null);
230  _formState.Next = next;
231  step = new NavigationStep<T>(_form.Steps[_formState.Step].Name, _form, _state, _formState);
232  if (start)
233  {
234  prompt = step.Start(context, _state, _formState);
235  }
236  else
237  {
238  matches = step.Match(context, _state, _formState, toBotText, out lastInput);
239  }
240  }
241  else
242  {
243  // Processing current step
244  step = _form.Steps[_formState.Step];
245  if (_formState.Phase() == StepPhase.Ready)
246  {
247  if (step.Type == StepType.Message)
248  {
249  feedback = step.Start(context, _state, _formState);
250  requirePrompt = true;
251  useLastPrompt = false;
252  next = new NextStep();
253  }
254  else
255  {
256  prompt = step.Start(context, _state, _formState);
257  }
258  }
259  else if (_formState.Phase() == StepPhase.Responding)
260  {
261  matches = step.Match(context, _state, _formState, toBotText, out lastInput);
262  }
263  }
264  if (matches != null)
265  {
266  matches = MatchAnalyzer.Coalesce(matches, lastInput).ToArray();
267  if (MatchAnalyzer.IsFullMatch(lastInput, matches))
268  {
269  var result = await step.ProcessAsync(context, _state, _formState, lastInput, matches);
270  next = result.Next;
271  feedback = result.Feedback;
272  prompt = result.Prompt;
273 
274  // 1) Not completed, not valid -> Not require, last
275  // 2) Completed, feedback -> require, not last
276  requirePrompt = (_formState.Phase() == StepPhase.Completed);
277  useLastPrompt = !requirePrompt;
278  }
279  else
280  {
281  // Filter non-active steps out of command matches
282  var commands =
283  (from command in MatchAnalyzer.Coalesce(_commands.Matches(lastInput), lastInput)
284  where (command.Value is FormCommand
285  || _form.Fields.Field(command.Value as string).Active(_state))
286  select command).ToArray();
287  if (MatchAnalyzer.IsFullMatch(lastInput, commands))
288  {
289  next = DoCommand(context, _state, _formState, step, commands, out feedback);
290  requirePrompt = false;
291  useLastPrompt = true;
292  }
293  else
294  {
295  if (matches.Count() == 0 && commands.Count() == 0)
296  {
297  // TODO: If we implement fallback, opportunity to call parent dialogs
298  feedback = step.NotUnderstood(context, _state, _formState, lastInput);
299  requirePrompt = false;
300  useLastPrompt = false;
301  }
302  else
303  {
304  // Go with response since it looks possible
305  var bestMatch = MatchAnalyzer.BestMatches(matches, commands);
306  if (bestMatch == 0)
307  {
308  var result = await step.ProcessAsync(context, _state, _formState, lastInput, matches);
309  next = result.Next;
310  feedback = result.Feedback;
311  prompt = result.Prompt;
312 
313  requirePrompt = (_formState.Phase() == StepPhase.Completed);
314  useLastPrompt = !requirePrompt;
315  }
316  else
317  {
318  next = DoCommand(context, _state, _formState, step, commands, out feedback);
319  requirePrompt = false;
320  useLastPrompt = true;
321  }
322  }
323  }
324  }
325  }
326  next = ActiveSteps(next, _state);
327  if (feedback != null)
328  {
329  message = (message == null ? feedback : message + "\n\n" + feedback);
330  }
331  }
332  if (next.Direction == StepDirection.Complete || next.Direction == StepDirection.Quit)
333  {
334  if (next.Direction == StepDirection.Complete)
335  {
336  if (_form.Completion != null)
337  {
338  await _form.Completion(context, _state);
339  }
340  context.Done(_state);
341  }
342  else if (next.Direction == StepDirection.Quit)
343  {
344  throw new OperationCanceledException();
345  }
346  else
347  {
348  throw new NotImplementedException();
349  }
350  }
351  else
352  {
353  if (message != null)
354  {
355  if (requirePrompt)
356  {
357  _formState.LastPrompt = prompt;
358  prompt = message + "\n\n" + prompt;
359  }
360  else if (useLastPrompt)
361  {
362  prompt = message + "\n\n" + _formState.LastPrompt;
363  }
364  else
365  {
366  prompt = message;
367  }
368  }
369  else
370  {
371  _formState.LastPrompt = prompt;
372  }
373 
374  await context.PostAsync(prompt);
375  context.Wait(MessageReceived);
376  }
377  }

383  private NextStep ActiveSteps(NextStep next, T state)
384  {
385  var result = next;
386  if (next.Direction == StepDirection.Named)
387  {
388  var names = (from name in next.Names where _form.Fields.Field(name).Active(state) select name);
389  var count = names.Count();
390  if (count == 0)
391  {
392  result = new NextStep();
393  }
394  else if (count != result.Names.Count())
395  {
396  result = new NextStep(names);
397  }
398  }
399  return result;
400  }

409  private bool MoveToNext(T state, FormState form, NextStep next)
410  {
411  bool found = false;
412  switch (next.Direction)
413  {
414  case StepDirection.Complete:
415  break;
416  case StepDirection.Named:
417  form.StepState = null;
418  if (next.Names.Count() == 0)
419  {
420  goto case StepDirection.Next;
421  }
422  else if (next.Names.Count() == 1)
423  {
424  var name = next.Names.First();
425  var nextStep = -1;
426  for (var i = 0; i < _form.Steps.Count(); ++i)
427  {
428  if (_form.Steps[i].Name == name)
429  {
430  nextStep = i;
431  break;
432  }
433  }
434  if (nextStep == -1)
435  {
436  throw new ArgumentOutOfRangeException("NextStep", "Does not correspond to a field in the form.");
437  }
438  if (_form.Steps[nextStep].Active(state))
439  {
440  var current = _form.Steps[form.Step];
441  form.SetPhase(_form.Fields.Field(current.Name).IsUnknown(state) ? StepPhase.Ready : StepPhase.Completed);
442  form.History.Push(form.Step);
443  form.Step = nextStep;
444  form.SetPhase(StepPhase.Ready);
445  found = true;
446  }
447  else
448  {
449  // If we went to a state which is not active fall through to the next active if any
450  goto case StepDirection.Next;
451  }
452  }
453  else
454  {
455  // Always mark multiple names as found so we can handle the user navigation
456  found = true;
457  }
458  break;
459  case StepDirection.Next:
460  var start = form.Step;
461  // Next ready step including current one
462  for (var offset = 0; offset < _form.Steps.Count; ++offset)
463  {
464  form.Step = (start + offset) % _form.Steps.Count;
465  if (offset > 0)
466  {
467  form.StepState = null;
468  form.Next = null;
469  }
470  var step = _form.Steps[form.Step];
471  if ((form.Phase() == StepPhase.Ready || form.Phase() == StepPhase.Responding)
472  && step.Active(state))
473  {
474  if (step.Type == StepType.Confirm)
475  {
476  // Ensure all dependencies have values
477  foreach (var dependency in step.Dependencies())
478  {
479  var dstep = _form.Step(dependency);
480  var dstepi = _form.StepIndex(dstep);
481  if (dstep.Active(state) && form.Phases[dstepi] != StepPhase.Completed)
482  {
483  form.Step = dstepi;
484  break;
485  }
486  }
487  found = true;
488  }
489  else
490  {
491  found = true;
492  }
493  if (form.Step != start && _form.Steps[start].Type != StepType.Message)
494  {
495  form.History.Push(start);
496  }
497  break;
498  }
499  }
500  if (!found)
501  {
502  next.Direction = StepDirection.Complete;
503  }
504  break;
505  case StepDirection.Previous:
506  while (form.History.Count() > 0)
507  {
508  var lastStepIndex = form.History.Pop();
509  var lastStep = _form.Steps[lastStepIndex];
510  if (lastStep.Active(state))
511  {
512  var step = _form.Steps[form.Step];
513  form.SetPhase(step.Field.IsUnknown(state) ? StepPhase.Ready : StepPhase.Completed);
514  form.Step = lastStepIndex;
515  form.SetPhase(StepPhase.Ready);
516  form.StepState = null;
517  form.Next = null;
518  found = true;
519  break;
520  }
521  }
522  if (!found)
523  {
524  next.Direction = StepDirection.Quit;
525  }
526  break;
527  case StepDirection.Quit:
528  break;
529  case StepDirection.Reset:
530  form.Reset();
531  // TODO: Should I reset the state as well?
532  // Because we redo phase they can go through everything again but with defaults.
533  found = true;
534  break;
535  }
536  return found;
537  }

539  private NextStep DoCommand(IDialogContext context, T state, FormState form, IStep<T> step, IEnumerable<TermMatch> matches, out string feedback)
540  {
541  // TODO: What if there are more than one command?
542  feedback = null;
543  var next = new NextStep();
544  var value = matches.First().Value;
545  if (value is FormCommand)
546  {
547  switch ((FormCommand)value)
548  {
549  case FormCommand.Backup:
550  {
551  next.Direction = step.Back(context, state, form) ? StepDirection.Next : StepDirection.Previous;
552  }
553  break;
554  case FormCommand.Help:
555  {
556  var field = step.Field;
557  var builder = new StringBuilder();
558  foreach (var entry in _form.Configuration.Commands)
559  {
560  builder.Append("* ");
561  builder.AppendLine(entry.Value.Help);
562  }
563  var navigation = new Prompter<T>(field.Template(TemplateUsage.NavigationCommandHelp), _form, null);
564  var active = (from istep in _form.Steps
565  where istep.Type == StepType.Field && istep.Active(state)
566  select istep.Field.Description());
567  var activeList = Language.BuildList(active, navigation.Annotation().Separator, navigation.Annotation().LastSeparator);
568  builder.Append("* ");
569  builder.Append(navigation.Prompt(state, "", activeList));
570  feedback = step.Help(state, form, builder.ToString());
571  }
572  break;
573  case FormCommand.Quit: next.Direction = StepDirection.Quit; break;
574  case FormCommand.Reset: next.Direction = StepDirection.Reset; break;
575  case FormCommand.Status:
576  {
577  var prompt = new Prompt("{*}");
578  feedback = new Prompter<T>(prompt, _form, null).Prompt(state, "");
579  }
580  break;
581  }
582  }
583  else
584  {
585  var name = value as string;
586  var istep = _form.Step(name);
587  if (istep != null && istep.Active(state))
588  {
589  next = new NextStep(new string[] { name });
590  }
591  }
592  return next;
593  }
