61  public IForm<T> Build()
62  {
63  if (!_form._steps.Any((step) => step.Type == StepType.Field))
64  {
65  var paths = new List<string>();
66  FormBuilder<T>.FieldPaths(typeof(T), "", paths);
67  IFormBuilder<T> builder = this;
68  foreach (var path in paths)
69  {
70  builder.Field(new FieldReflector<T>(path));
71  }
72  builder.Confirm("Is this your selection?\n{*}");
73  }
74  Validate();
75  return this._form;
76  }

188  private IFormBuilder<T> AddField(IField<T> field)
189  {
190  field.Form = _form;
191  _form._fields.Add(field);
192  var step = new FieldStep<T>(field.Name, _form);
193  var stepIndex = this._form._steps.FindIndex(s => s.Name == field.Name);
194  if (stepIndex >= 0)
195  {
196  _form._steps[stepIndex] = step;
197  }
198  else
199  {
200  _form._steps.Add(step);
201  }
202  return this;
203  }

250  private void Validate()
251  {
252  foreach (var step in _form._steps)
253  {
254  // Validate prompt
255  var annotation = step.Annotation;
256  if (annotation != null)
257  {
258  var name = step.Type == StepType.Field ? step.Name : "";
259  foreach (var pattern in annotation.Patterns)
260  {
261  ValidatePattern(pattern, name, 5);
262  }
263  if (step.Type != StepType.Message)
264  {
265  foreach (TemplateUsage usage in Enum.GetValues(typeof(TemplateUsage)))
266  {
267  if (usage != TemplateUsage.None)
268  {
269  foreach (var pattern in step.Field.Template(usage).Patterns)
270  {
271  ValidatePattern(pattern, name, TemplateArgs(usage));
272  }
273  }
274  }
275  }
276  }
277  }
278  ValidatePattern(_form.Configuration.DefaultPrompt.ChoiceFormat, "", 2);
279  }

289  internal static void FieldPaths(Type type, string path, List<string> paths)
290  {
291  var newPath = (path == "" ? path : path + ".");
292  foreach (var field in type.GetFields(BindingFlags.Public | BindingFlags.Instance))
293  {
294  TypePaths(field.FieldType, newPath + field.Name, paths);
295  }
296 
297  foreach (var property in type.GetProperties(BindingFlags.Public | BindingFlags.Instance))
298  {
299  if (property.CanRead && property.CanWrite)
300  {
301  TypePaths(property.PropertyType, newPath + property.Name, paths);
302  }
303  }
304  }
305 
306  internal static void TypePaths(Type type, string path, List<string> paths)
307  {
308  if (type.IsClass)
309  {
310  if (type == typeof(string))
311  {
312  paths.Add(path);
313  }
314  else if (type.IsIEnumerable())
315  {
316  var elt = type.GetGenericElementType();
317  if (elt.IsEnum)
318  {
319  paths.Add(path);
320  }
321  else
322  {
323  // TODO: What to do about enumerations of things other than enums?
324  }
325  }
326  else
327  {
328  FieldPaths(type, path, paths);
329  }
330  }
331  else if (type.IsEnum)
332  {
333  paths.Add(path);
334  }
335  else if (type == typeof(bool))
336  {
337  paths.Add(path);
338  }
339  else if (type.IsIntegral())
340  {
341  paths.Add(path);
342  }
343  else if (type.IsDouble())
344  {
345  paths.Add(path);
346  }
347  else if (type.IsNullable() && type.IsValueType)
348  {
349  paths.Add(path);
350  }
351  else if (type == typeof(DateTime))
352  {
353  paths.Add(path);
354  }
355  }
356  }
