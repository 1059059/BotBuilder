150  public IFormBuilder<T> Confirm(Prompt prompt, ConditionalDelegate<T> condition = null, IEnumerable<string> dependencies = null)
151  {
152  if (condition == null) condition = state => true;
153  if (dependencies == null)
154  {
155  // Default next steps go from previous field ignoring confirmations back to next confirmation
156  // Last field before confirmation
157  var end = _form._steps.Count();
158  while (end > 0)
159  {
160  if (_form._steps[end - 1].Type == StepType.Field)
161  {
162  break;
163  }
164  --end;
165  }
166 
167  var start = end;
168  while (start > 0)
169  {
170  if (_form._steps[start - 1].Type == StepType.Confirm)
171  {
172  break;
173  }
174  --start;
175  }
176  var fields = new List<string>();
177  for (var i = start; i < end; ++i)
178  {
179  if (_form._steps[i].Type == StepType.Field)
180  {
181  fields.Add(_form._steps[i].Name);
182  }
183  }
184  dependencies = fields;
185  }
186  var confirmation = new Confirmation<T>(prompt, condition, dependencies, _form);
187  _form._fields.Add(confirmation);
188  _form._steps.Add(new ConfirmStep<T>(confirmation));
189  return this;
190  }

204  private IFormBuilder<T> AddField(IField<T> field)
205  {
206  _form._fields.Add(field);
207  var step = new FieldStep<T>(field.Name, _form);
208  var stepIndex = this._form._steps.FindIndex(s => s.Name == field.Name);
209  if (stepIndex >= 0)
210  {
211  _form._steps[stepIndex] = step;
212  }
213  else
214  {
215  _form._steps.Add(step);
216  }
217  return this;
218  }

221  internal static void FieldPaths(Type type, string path, List<string> paths)
222  {
223  var newPath = (path == "" ? path : path + ".");
224  foreach (var field in type.GetFields(BindingFlags.Public | BindingFlags.Instance))
225  {
226  TypePaths(field.FieldType, newPath + field.Name, paths);
227  }
228 
229  foreach (var property in type.GetProperties(BindingFlags.Public | BindingFlags.Instance))
230  {
231  if (property.CanRead && property.CanWrite)
232  {
233  TypePaths(property.PropertyType, newPath + property.Name, paths);
234  }
235  }
236  }

238  internal static void TypePaths(Type type, string path, List<string> paths)
239  {
240  if (type.IsClass)
241  {
242  if (type == typeof(string))
243  {
244  paths.Add(path);
245  }
246  else if (type.IsIEnumerable())
247  {
248  var elt = type.GetGenericElementType();
249  if (elt.IsEnum)
250  {
251  paths.Add(path);
252  }
253  else
254  {
255  // TODO: What to do about enumerations of things other than enums?
256  }
257  }
258  else
259  {
260  FieldPaths(type, path, paths);
261  }
262  }
263  else if (type.IsEnum)
264  {
265  paths.Add(path);
266  }
267  else if (type == typeof(bool))
268  {
269  paths.Add(path);
270  }
271  else if (type.IsIntegral())
272  {
273  paths.Add(path);
274  }
275  else if (type.IsDouble())
276  {
277  paths.Add(path);
278  }
279  else if (type.IsNullable() && type.IsValueType)
280  {
281  paths.Add(path);
282  }
283  else if (type == typeof(DateTime))
284  {
285  paths.Add(path);
286  }
287  }
288  }
