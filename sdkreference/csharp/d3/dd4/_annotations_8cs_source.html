292  public void ApplyDefaults(TemplateBase defaultTemplate)
293  {
294  if (AllowDefault == BoolDefault.Default) AllowDefault = defaultTemplate.AllowDefault;
295  if (ChoiceStyle == ChoiceStyleOptions.Default) ChoiceStyle = defaultTemplate.ChoiceStyle;
296  if (FieldCase == CaseNormalization.Default) FieldCase = defaultTemplate.FieldCase;
297  if (Feedback == FeedbackOptions.Default) Feedback = defaultTemplate.Feedback;
298  if (ChoiceFormat == null) ChoiceFormat = defaultTemplate.ChoiceFormat;
