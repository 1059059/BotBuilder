295  public void ApplyDefaults(TemplateBase defaultTemplate)
296  {

299  if (FieldCase == CaseNormalization.Default) FieldCase = defaultTemplate.FieldCase;

302  if (LastSeparator == null) LastSeparator = defaultTemplate.LastSeparator;
