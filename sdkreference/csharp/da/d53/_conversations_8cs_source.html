90  {
91  IWaitFactory waits = new WaitFactory();

93  IBotData toBotData = new JObjectBotData(toBot);
94  IConnectorClient client = new ConnectorClient();
95  var provider = new Serialization.SimpleServiceLocator(singletons)
96  {
97  waits, frames, toBotData, client
98  };
99  var formatter = Conversation.MakeBinaryFormatter(provider);
100 
101  Internals.DialogContext context;
102 
103  byte[] blobOld;
104  bool found = toBotData.PerUserInConversationData.TryGetValue(BlobKey, out blobOld);
105  if (found)
106  {
107  using (var streamOld = new MemoryStream(blobOld))
108  using (var gzipOld = new GZipStream(streamOld, CompressionMode.Decompress))
109  {
110  context = (Internals.DialogContext)formatter.Deserialize(gzipOld);
111  }
112  }
113  else
114  {
115  IFiberLoop fiber = new Fiber(frames);
116  context = new Internals.DialogContext(client, toBotData, fiber);
117  var root = MakeRoot();
118  var loop = Methods.Void(Methods.Loop(context.ToRest(root.StartAsync), int.MaxValue));
119  fiber.Call(loop, null);
120  await fiber.PollAsync();
121  }
122 
123  IUserToBot userToBot = context;
124  var toUser = await userToBot.SendAsync(toBot, token);
125 
126  // even with no bot response, try to save state
127  if (toUser == null)
128  {
129  toUser = Internals.DialogContext.ToUser(toBot, toUserText: null);
130  }
131 
132  byte[] blobNew;
133  using (var streamNew = new MemoryStream())
134  using (var gzipNew = new GZipStream(streamNew, CompressionMode.Compress))
135  {
136  formatter.Serialize(gzipNew, context);
137  gzipNew.Close();
138  blobNew = streamNew.ToArray();
139  }
140 
141  IBotData toUserData = new Internals.JObjectBotData(toUser);
142 
143  toUserData.PerUserInConversationData.SetValue(BlobKey, blobNew);
144 
145  return toUser;
146  }
