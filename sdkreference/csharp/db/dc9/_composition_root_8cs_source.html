67  public static async Task<Message> SendAsync(Message toBot, Func<IDialog> MakeRoot, CancellationToken token = default(CancellationToken), params object[] singletons)
68  {
69  IWaitFactory waits = new WaitFactory();
70  IFrameFactory frames = new FrameFactory(waits);
71  IBotData toBotData = new Internals.JObjectBotData(toBot);
72  IConnectorClient client = new ConnectorClient();
73  var provider = new Serialization.SimpleServiceLocator(singletons)
74  {
75  waits, frames, toBotData, client
76  };
77  var formatter = Conversation.MakeBinaryFormatter(provider);
78 
79  Internals.DialogContext context;
80 
81  byte[] blobOld;
82  bool found = toBotData.PerUserInConversationData.TryGetValue(BlobKey, out blobOld);
83  if (found)
84  {
85  using (var streamOld = new MemoryStream(blobOld))
86  using (var gzipOld = new GZipStream(streamOld, CompressionMode.Decompress))
87  {
88  context = (Internals.DialogContext)formatter.Deserialize(gzipOld);
89  }
90  }
91  else
92  {
93  IFiberLoop fiber = new Fiber(frames);
94  context = new Internals.DialogContext(client, toBotData, fiber);
95  var root = MakeRoot();
96  var loop = Methods.Void(Methods.Loop(context.ToRest(root.StartAsync), int.MaxValue));
97  fiber.Call(loop, null);
98  await fiber.PollAsync();
99  }
100 
101  IUserToBot userToBot = context;
102  var toUser = await userToBot.SendAsync(toBot, token);
103 
104  // even with no bot response, try to save state
105  if (toUser == null)
106  {
107  toUser = Internals.DialogContext.ToUser(toBot, toUserText: null);
108  }
109 
110  byte[] blobNew;
111  using (var streamNew = new MemoryStream())
112  using (var gzipNew = new GZipStream(streamNew, CompressionMode.Compress))
113  {
114  formatter.Serialize(gzipNew, context);
115  gzipNew.Close();
116  blobNew = streamNew.ToArray();
117  }
118 
119  IBotData toUserData = new Internals.JObjectBotData(toUser);
120 
121  toUserData.PerUserInConversationData.SetValue(BlobKey, blobNew);
122 
123  return toUser;
124  }
