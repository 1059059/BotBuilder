824  // If we went to a state which is not active fall through to the next active if any
825  goto case StepDirection.Next;
826  }
827  }
828  else
829  {
830  // Always mark multiple names as found so we can handle the user navigation
831  found = true;
832  }
833  break;
834  case StepDirection.Next:
835  var start = form.Step;
836  // Next ready step including current one
837  for (var offset = 0; offset < _steps.Count(); ++offset)
838  {
839  form.Step = (start + offset) % _steps.Count();
840  if (offset > 0)
841  {
842  form.StepState = null;
843  form.Next = null;
844  }
845  var step = _steps[form.Step];
846  if ((form.Phase() == StepPhase.Ready || form.Phase() == StepPhase.Responding)
847  && step.Active(state))
848  {
849  if (step.Type() == StepType.Confirm)
850  {
851  // Ensure all dependencies have values
852  foreach (var dependency in step.Dependencies())
853  {
854  var dstep = Step(dependency);
855  var dstepi = StepIndex(dstep);
856  if (dstep.Active(state) && form.Phases[dstepi] != StepPhase.Completed)
857  {
858  form.Step = dstepi;
859  break;
860  }
861  }
862  found = true;
863  }
864  else
865  {
866  found = true;
867  }
868  if (form.Step != start && _steps[start].Type() != StepType.Message)
869  {
870  form.History.Push(start);
871  }
872  break;
873  }
874  }
875  if (!found)
876  {
877  next.Direction = StepDirection.Complete;
878  }
879  break;
880  case StepDirection.Previous:
881  while (form.History.Count() > 0)
882  {
883  var lastStepIndex = form.History.Pop();
884  var lastStep = _steps[lastStepIndex];
885  if (lastStep.Active(state))
886  {
887  var step = _steps[form.Step];
888  form.SetPhase(step.Field().IsUnknown(state) ? StepPhase.Ready : StepPhase.Completed);
889  form.Step = lastStepIndex;
890  form.SetPhase(StepPhase.Ready);
891  form.StepState = null;
892  form.Next = null;
893  found = true;
894  break;
895  }
896  }
897  if (!found)
898  {
899  next.Direction = StepDirection.Quit;
900  }
901  break;
902  case StepDirection.Quit:
903  break;
904  case StepDirection.Reset:
905  form.Reset();
906  // TODO: Should I reset the state as well?
907  // Because we redo phase they can go through everything again but with defaults.
908  found = true;
909  break;
910  }
911  return found;
912  }

1052  out string feedback, out string prompt)
1053  {

1058  object response = null;

1061  // Response to prompt
1062  var firstMatch = matches.FirstOrDefault();

1064  {
1065  response = SetValue(state, firstMatch.Value, form, out feedback);
1066  }
1067  else if (matches.Count() > 1)
1068  {
1069  // Check multiple matches for ambiguity
1070  var groups = MatchAnalyzer.GroupedMatches(matches);
1071  // 1) Could be multiple match groups like for ingredients.
1072  // 2) Could be overlapping matches like "onion".
1073  // 3) Could be multiple matches where only one is expected.
1074 
1075  if (!_field.AllowsMultiple())
1076  {
1077  // Create a single group of all possibilities if only want one value
1078  var mergedGroup = groups.SelectMany((group) => group).ToList();
1079  groups = new List<List<TermMatch>>() { mergedGroup };
1080  }
1081  var ambiguous = new List<Ambiguous>();
1082  var settled = new List<object>();
1083  foreach (var choices in groups)
1084  {
1085  if (choices.Count() > 1)
1086  {
1087  var unclearResponses = string.Join(" ", (from choice in choices select input.Substring(choice.Start, choice.Length)).Distinct());
1088  var values = from match in choices select match.Value;
1089  ambiguous.Add(new Ambiguous(unclearResponses, values));
1090  }
1091  else
1092  {

1094  if (matchValue.GetType().IsIEnumerable())
1095  {
1096  foreach (var value in matchValue as System.Collections.IEnumerable)
1097  {
1098  settled.Add(value);
1099  }
1100  }
1101  else
1102  {
1103  settled.Add(choices.First().Value);
1104  }
1105  }
1106  }

1119  else

1121  if (_field.AllowsMultiple())
1122  {
1123  response = SetValue(state, settled, form, out feedback);
1124  }
1125  else
1126  {
1127  Debug.Assert(settled.Count() == 1);
1128  response = SetValue(state, settled.First(), form, out feedback);
1129  }
1130  }
1131  }
1132  var unmatched = MatchAnalyzer.Unmatched(input, matches);
1133  var unmatchedWords = string.Join(" ", unmatched);

1136  if (_field.Prompt().Annotation().Feedback == FeedbackOptions.Always)
1137  {
1138  fieldState.Unmatched = string.Join(" ", nonNoise);
1139  }
1140  else if (_field.Prompt().Annotation().Feedback == FeedbackOptions.Auto
1141  && nonNoise.Length > 0
1142  && unmatched.Count() > 0)
1143  {
1144  fieldState.Unmatched = string.Join(" ", nonNoise);
1145  }
1146  }
1147  else if (fieldState.State == FieldStepStates.SentClarify)
1148  {
1149  Ambiguous clarify;
1150  var iChoicePrompt = NextClarifyPrompt(state, fieldState, iprompt.Recognizer(), out clarify);
1151  if (matches.Count() == 1)
1152  {
1153  // Clarified ambiguity
1154  fieldState.Settled.Add(matches.First().Value);
1155  fieldState.Clarifications.Remove(clarify);
1156  Ambiguous newClarify;
1157  var newiChoicePrompt = NextClarifyPrompt(state, fieldState, iprompt.Recognizer(), out newClarify);
1158  if (newiChoicePrompt != null)
1159  {
1160  prompt = newiChoicePrompt.Prompt(state, _name, newClarify.Response);
1161  }
1162  else
1163  {
1164  // No clarification left, so set the field
1165  if (_field.AllowsMultiple())
1166  {
1167  response = SetValue(state, fieldState.Settled, form, out feedback);
1168  }
1169  else
1170  {
1171  Debug.Assert(fieldState.Settled.Count() == 1);
1172  response = SetValue(state, fieldState.Settled.First(), form, out feedback);
1173  }
1174  form.SetPhase(StepPhase.Completed);
1175  }
1176  }
1177  }
1178  if (form.Phase() == StepPhase.Completed)
1179  {
1180  form.StepState = null;
1181  if (fieldState.Unmatched != null)
1182  {
1183  if (fieldState.Unmatched != "")
1184  {
1185  feedback = new Prompter<T>(_field.Template(TemplateUsage.Feedback), _form, null).Prompt(state, _name, fieldState.Unmatched);
1186  }
1187  else
1188  {
1189  feedback = new Prompter<T>(_field.Template(TemplateUsage.Feedback), _form, null).Prompt(state, _name);
1190  }
1191  }
1192  }
1193  return _field.Next(response, state);
