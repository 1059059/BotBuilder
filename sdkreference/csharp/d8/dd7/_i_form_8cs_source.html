465  public object Process(Session session, T state, FormState form, string input, IEnumerable<TermMatch> matches, out string prompt)
466  {
467  prompt = null;
468  var iprompt = _field.Prompt();
469  var fieldState = form.StepState as FieldStepState;
470  if (fieldState.State == FieldStepStates.SentPrompt)
471  {
472  // Response to prompt
473  var firstMatch = matches.FirstOrDefault();
474  if (matches.Count() == 1)
475  {
476  SetValue(state, firstMatch.Value);
477  form.StepState = null;
478  form.SetPhase(StepPhase.Completed);
479  }
480  else if (matches.Count() > 1)
481  {
482  // Check multiple matches for ambiguity
483  var groups = MatchAnalyzer.GroupedMatches(matches);
484  // 1) Could be multiple match groups like for ingredients.
485  // 2) Could be overlapping matches like "onion".
486  // 3) Could be multiple matches where only one is expected.
487 
488  // Did you mean {x} or {y}?
489  var ambiguous = new List<Ambiguous>();
490  var settled = new List<object>();
491  foreach (var choices in groups)
492  {
493  if (choices.Count() > 1)
494  {
495  var unclearMatch = choices.First();
496  var unclearResponse = input.Substring(unclearMatch.Start, unclearMatch.Length);
497  var values = from match in choices select match.Value;
498  ambiguous.Add(new Ambiguous(unclearResponse, values));
499  }
500  else
501  {
502  var matchValue = choices.First().Value;
503  if (matchValue.GetType().IsIEnumerable())
504  {
505  foreach (var value in matchValue as System.Collections.IEnumerable)
506  {
507  settled.Add(value);
508  }
509  }
510  else
511  {
512  settled.Add(choices.First().Value);
513  }
514  }
515  }
516 
517  if (ambiguous.Count() > 0)
518  {
519  // Need 1 or more clarifications
520  Ambiguous clarify;
521  form.StepState = new FieldStepState(FieldStepStates.SentClarify) { Settled = settled, Clarifications = ambiguous };
522  _field.SetValue(state, null);
523  var iChoicePrompt = NextClarifyPrompt(state, form.StepState as FieldStepState, iprompt.Recognizer(), out clarify);
524  prompt = iChoicePrompt.Prompt(state, _name, clarify.Response);
525  }
526  else
527  {
528  if (_field.AllowsMultiple())
529  {
530  SetValue(state, settled);
531  }
532  else
533  {
534  // Set the resulting values.
535  // TODO: what about when multiple values are not allowed...
536  foreach (var value in settled)
537  {
538  SetValue(state, value);
539  }
540  }
541  form.SetPhase(StepPhase.Completed);
542  }
543  }
544  }
545  else if (fieldState.State == FieldStepStates.SentClarify)
546  {
547  Ambiguous clarify;
548  var iChoicePrompt = NextClarifyPrompt(state, fieldState, iprompt.Recognizer(), out clarify);
549  var choices = iChoicePrompt.Recognizer().Matches(input);
550  if (choices.Count() == 1)
551  {
552  // Clarified ambiguity
553  fieldState.Settled.Add(choices.First().Value);
554  fieldState.Clarifications.Remove(clarify);
555  Ambiguous newClarify;
556  var newiChoicePrompt = NextClarifyPrompt(state, fieldState, iprompt.Recognizer(), out newClarify);
557  if (newiChoicePrompt != null)
558  {
559  prompt = newiChoicePrompt.Prompt(state, _name, newClarify.Response);
560  }
561  else
562  {
563  // No clarification left, so set the field
564  if (_field.AllowsMultiple())
565  {
566  SetValue(state, fieldState.Settled);
567  }
568  else
569  {
570  // Set the resulting values.
571  // TODO: what about when multiple values are not allowed...
572  foreach (var value in fieldState.Settled)
573  {
574  SetValue(state, value);
575  }
576  }
577  fieldState = null;
578  form.SetPhase(StepPhase.Completed);
579  }
580  }
581  }
582  return +1;
583  }
