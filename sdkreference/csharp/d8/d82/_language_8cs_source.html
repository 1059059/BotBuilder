64  public static bool NonWord(string word)
65  {
66  bool nonWord = true;

69  if (!(char.IsControl(ch) || char.IsPunctuation(ch) || char.IsWhiteSpace(ch)))
70  {
71  nonWord = false;
72  break;

147  public static string CamelCase(string original)
148  {
149  var builder = new StringBuilder();
150  var name = original.Trim();
151  var previousUpper = Char.IsUpper(name[0]);
152  var previousLetter = Char.IsLetter(name[0]);
153  bool first = true;
154  for (int i = 0; i < name.Length; ++i)
155  {
156  var ch = name[i];
157  if (!first && (ch == '_' || ch == ' '))
158  {
159  // Non begin _ as space
160  builder.Append(' ');
161  }
162  else
163  {
164  var isUpper = Char.IsUpper(ch);
165  var isLetter = Char.IsLetter(ch);
166  if ((!previousUpper && isUpper)
167  || (isLetter != previousLetter)
168  || (!first && isUpper && (i + 1) < name.Length && Char.IsLower(name[i + 1])))
169  {
170  // Break on lower to upper, number boundaries and Upper to lower
171  builder.Append(' ');
172  }
173  previousUpper = isUpper;
174  previousLetter = isLetter;
175  builder.Append(ch);
176  if (first)
177  {
178  first = false;
179  }
180  }
181  }
182  return builder.ToString();
183  }

215  public static string[] GenerateTerms(string phrase, int maxLength)
216  {
217  var words = (from word in phrase.Split(' ') select word.ToLower()).ToArray();
218  var terms = new List<string>();
219  for (var length = 1; length <= Math.Min(words.Length, maxLength); ++length)
220  {
221  for (var start = 0; start <= words.Length - length; ++start)
222  {
223  var ngram = new ArraySegment<string>(words, start, length);
224  if (!ArticleOrNone(ngram.First()) && !ArticleOrNone(ngram.Last()))
225  {
226  terms.Add(string.Join(" ", OptionalPlurals(ngram)));
227  }
228  }
229  }
230  if (words.Length > maxLength)
231  {
232  terms.Add(string.Join(" ", words));
233  }
234  return terms.ToArray();
235  }

247  public static string ANormalization(string input)
248  {
249  var builder = new StringBuilder();
250  var last = 0;
251  foreach (Match match in _aOrAn.Matches(input))
252  {
253  var currentWord = match.Groups[1];
254  builder.Append(input.Substring(last, currentWord.Index - last));
255  if (match.Groups[2].Success)
256  {
257  builder.Append("an");
258  }
259  else
260  {
261  builder.Append("a");
262  }
263  last = currentWord.Index + currentWord.Length;
264  }
265  builder.Append(input.Substring(last));
266  return builder.ToString();
267  }
