103  public static string CamelCase(string name)
104  {
105  var builder = new StringBuilder();
106  var previousUpper = Char.IsUpper(name[0]);
107  var previousLetter = Char.IsLetter(name[0]);
108  bool first = true;
109  foreach (var ch in name)
110  {
111  if (!first && ch == '_')
112  {
113  // Non begin _ as space
114  builder.Append(' ');
115  }
116  else
117  {
118  if (first)
119  {
120  first = false;
121  }
122  var isUpper = Char.IsUpper(ch);
123  var isLetter = Char.IsLetter(ch);
124  if ((!previousUpper && isUpper) || (isLetter != previousLetter))
125  {
126  // Break on lower to upper or number boundaries
127  builder.Append(' ');
128  }
129  previousUpper = isUpper;
130  previousLetter = isLetter;
131  builder.Append(ch);
132  }
133  }
134  return builder.ToString();
