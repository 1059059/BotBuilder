445  public override object GetValue(T state)
446  {
447  object current = state;
448  Type ftype = null;
449  foreach (var step in _path)
450  {
451  ftype = StepType(step);
452  var field = step as FieldInfo;
453  if (field != null)
454  {
455  current = field.GetValue(current);
456  }
457  else
458  {
459  var prop = step as PropertyInfo;
460  current = prop.GetValue(current);
461  }
462  if (current == null)
463  {
464  break;
465  }
466  }
467  // Convert value types to null if appropriate
468  return (ftype.IsEnum
469  ? ((int)current == 0 ? null : current)
470  : (ftype == typeof(DateTime) && ((DateTime)current) == DateTime.MinValue)
471  ? null
472  : current);
473  }

475  public override void SetValue(T state, object value)
476  {
477  object current = state;
478  object lastClass = state;
479  var last = _path.Last();
480  foreach (var step in _path)
481  {
482  var field = step as FieldInfo;
483  var prop = step as PropertyInfo;
484  Type ftype = StepType(step);
485  if (step == last)
486  {
487  object newValue = value;
488  if (ftype.IsIEnumerable())
489  {
490  if (value != null && ftype != typeof(string))
491  {
492  // Build list and coerce elements
493  var list = Activator.CreateInstance(ftype);
494  var addMethod = list.GetType().GetMethod("Add");
495  foreach (var elt in value as System.Collections.IEnumerable)
496  {
497  addMethod.Invoke(list, new object[] { elt });
498  }
499  newValue = list;
500  }
501  }
502  else
503  {
504  if (value == null && (ftype.IsEnum || ftype.IsIntegral() || ftype.IsDouble()))
505  {
506  // Default value for numbers and enums
507  newValue = 0;
508  }
509  else if (ftype.IsIntegral())
510  {
511  newValue = Convert.ChangeType(value, ftype);
512  }
513  else if (ftype.IsDouble())
514  {
515  newValue = Convert.ChangeType(value, ftype);
516  }
517  else if (ftype == typeof(bool))
518  {
519  newValue = Convert.ChangeType(value, typeof(bool));
520  }
521  }
522  if (field != null)
523  {
524  field.SetValue(lastClass, newValue);
525  }
526  else
527  {
528  prop.SetValue(lastClass, newValue);
529  }
530  }
531  else
532  {
533  current = (field == null ? prop.GetValue(current) : field.GetValue(current));
534  if (current == null)
535  {
536  var obj = Activator.CreateInstance(ftype);
537  current = obj;
538  if (field != null)
539  {
540  field.SetValue(lastClass, current);
541  }
542  else
543  {
544  prop.SetValue(lastClass, current);
545  }
546  }
547  lastClass = current;
548  }
549  }
550  }

552  public override bool IsUnknown(T state)
553  {
554  var unknown = false;
555  var value = GetValue(state);
556  if (value == null)
557  {
558  unknown = true;
559  }
560  else
561  {
562  var step = _path.Last();
563  var ftype = StepType(step);
564  if (ftype.IsValueType && ftype.IsEnum)
565  {
566  unknown = ((int)value == 0);
567  }
568  else if (ftype == typeof(DateTime))
569  {
570  unknown = ((DateTime)value) == default(DateTime);
571  }
572  else if (ftype.IsIEnumerable())
573  {
574  unknown = !(value as System.Collections.IEnumerable).GetEnumerator().MoveNext();
575  }
576  }
577  return unknown;
578  }

680  protected void AddField(Type type, string[] path, int ipath)
681  {
682  if (ipath < path.Length)
683  {
684  ProcessTemplates(type);
685  var step = path[ipath];
686  object field = type.GetField(step, BindingFlags.Public | BindingFlags.Instance);
687  Type ftype;
688  if (field == null)
689  {
690  var prop = type.GetProperty(step, BindingFlags.Public | BindingFlags.Instance);
691  if (prop == null)
692  {
693  throw new ArgumentException(step + " is not a field or property in your type");
694  }
695  field = prop;
696  ftype = prop.PropertyType;
697  _path.Add(prop);
698  }
699  else
700  {
701  ftype = (field as FieldInfo).FieldType;
702  _path.Add(field);
703  }
704  if (ftype.IsNullable())
705  {
706  _isNullable = true;
707  _keepZero = true;
708  ftype = Nullable.GetUnderlyingType(ftype);
709  }
710  else if (ftype.IsEnum || ftype.IsClass)
711  {
712  _isNullable = true;
713  }
714  if (ftype.IsClass)
715  {
716  if (ftype == typeof(string))
717  {
718  _type = ftype;
719  ProcessFieldAttributes(field);
720  }
721  else if (ftype.IsIEnumerable())
722  {
723  var elt = ftype.GetGenericElementType();
724  if (elt.IsEnum)
725  {
726  _type = elt;
727  _allowsMultiple = true;
728  ProcessFieldAttributes(field);
729  ProcessEnumAttributes(elt);
730  }
731  else
732  {
733  // TODO: What to do about enumerations of things other than enums?
734  throw new NotImplementedException();
735  }
736  }
737  else
738  {
739  AddField(ftype, path, ipath + 1);
740  }
741  }
742  else
743  {
744  if (ftype.IsEnum)
745  {
746  ProcessFieldAttributes(field);
747  ProcessEnumAttributes(ftype);
748  }
749  else if (ftype == typeof(bool))
750  {
751  ProcessFieldAttributes(field);
752  }
753  else if (ftype.IsIntegral())
754  {
755  long min = long.MinValue;
756  long max = long.MaxValue;
757  if (ftype == typeof(sbyte)) { min = sbyte.MinValue; max = sbyte.MaxValue; }
758  else if (ftype == typeof(byte)) { min = byte.MinValue; max = byte.MaxValue; }
759  else if (ftype == typeof(short)) { min = short.MinValue; max = short.MaxValue; }
760  else if (ftype == typeof(ushort)) { min = ushort.MinValue; max = ushort.MaxValue; }
761  else if (ftype == typeof(int)) { min = int.MinValue; max = int.MaxValue; }
762  else if (ftype == typeof(uint)) { min = uint.MinValue; max = uint.MaxValue; }
763  else if (ftype == typeof(long)) { min = long.MinValue; max = long.MaxValue; }
764  else if (ftype == typeof(ulong)) { min = long.MinValue; max = long.MaxValue; }
765  SetLimits(min, max, false);
766  ProcessFieldAttributes(field);
767  }
768  else if (ftype.IsDouble())
769  {
770  double min = long.MinValue;
771  double max = long.MaxValue;
772  if (ftype == typeof(float)) { min = float.MinValue; max = float.MaxValue; }
773  else if (ftype == typeof(double)) { min = double.MinValue; max = double.MaxValue; }
774  SetLimits(min, max, false);
775  ProcessFieldAttributes(field);
776  }
777  else if (ftype == typeof(DateTime))
778  {
779  ProcessFieldAttributes(field);
780  }
781  _type = ftype;
782  }
783  }

797  protected void ProcessFieldAttributes(object step)
798  {
799  _optional = false;
800  if (!_ignoreAnnotations)
801  {
802  var field = step as FieldInfo;
803  var prop = step as PropertyInfo;
804  var name = (field == null ? prop.Name : field.Name);
805  var describe = (field == null ? prop.GetCustomAttribute<Describe>() : field.GetCustomAttribute<Describe>());
806  var terms = (field == null ? prop.GetCustomAttribute<Terms>() : field.GetCustomAttribute<Terms>());
807  var prompt = (field == null ? prop.GetCustomAttribute<Prompt>() : field.GetCustomAttribute<Prompt>());
808  var optional = (field == null ? prop.GetCustomAttribute<Optional>() : field.GetCustomAttribute<Optional>());
809  var numeric = (field == null ? prop.GetCustomAttribute<Numeric>() : field.GetCustomAttribute<Numeric>());
810  if (describe != null)
811  {
812  _description = describe.Description;
813  }
814  else
815  {
816  _description = Language.CamelCase(name);
817  }
818  if (terms != null)
819  {
820  _terms = terms.Alternatives;
821  }
822  else
823  {
824  _terms = Language.GenerateTerms(Language.CamelCase(name), 3);
825  }
826  if (prompt != null)
827  {
828  _promptDefinition = prompt;
829  }
830  if (numeric != null)
831  {
832  double oldMin, oldMax;
833  Limits(out oldMin, out oldMax);
834  SetLimits(numeric.Min, numeric.Max, numeric.Min != oldMin || numeric.Max != oldMax);
835  }
836  _optional = (optional != null);
837  foreach (var attribute in (field == null ? prop.GetCustomAttributes<Template>() : field.GetCustomAttributes<Template>()))
838  {
839  AddTemplate(attribute as Template);
840  }
841  }
842  }

844  protected void ProcessEnumAttributes(Type type)
845  {
846  foreach (var enumField in type.GetFields(BindingFlags.Static | BindingFlags.Public))
847  {
848  var enumValue = enumField.GetValue(null);
849  if (_keepZero || (int)enumValue > 0)
850  {
851  var describe = enumField.GetCustomAttribute<Describe>();
852  var terms = enumField.GetCustomAttribute<Terms>();
853  if (describe != null && !_ignoreAnnotations)
854  {
855  _valueDescriptions.Add(enumValue, describe.Description);
856  }
857  else
858  {
859  _valueDescriptions.Add(enumValue, Language.CamelCase(enumValue.ToString()));
860  }
861  if (terms != null && !_ignoreAnnotations)
862  {
863  _valueTerms.Add(enumValue, terms.Alternatives);
864  }
865  else
866  {
867  _valueTerms.Add(enumValue, Language.GenerateTerms(Language.CamelCase(enumValue.ToString()), 4));
868  }
869  }
