69  object current = state;
70  object lastClass = state;
71  FieldInfo last = _path.Last();
72  foreach (var field in _path)
73  {
74  if (field == last)
75  {

96  }
97  }
98  else
99  {

101  {
102  field.SetValue(lastClass, 0);
103  }
104  else
105  {
106  field.SetValue(lastClass, value);
107  }
108  }
109  }

112  current = field.GetValue(current);
113  if (current == null)
114  {

338  private void AddField(Type type, string[] path, int ipath)
339  {
340  if (ipath < path.Length)
341  {
342  ProcessTemplates(type);
343  var step = path[ipath];
344  var field = type.GetField(step, BindingFlags.Public | BindingFlags.Instance);
345  var ftype = field.FieldType;
346  _path.Add(field);
347  if (ftype.IsClass)
348  {
349  if (ftype == typeof(string))
350  {
351  ProcessFieldAttributes(field);
352  }
353  else if (ftype.IsIEnumerable())
354  {
355  var elt = ftype.GetGenericElementType();
356  if (elt.IsEnum)
357  {
358  _allowsMultiple = true;
359  ProcessFieldAttributes(field);
360  ProcessEnumAttributes(elt);
361  }
362  else

364  // TODO: What to do about enumerations of things other than enums?
365  throw new NotImplementedException();

370  AddField(ftype, path, ipath + 1);

372  }

392  private void ProcessFieldAttributes(FieldInfo field)
393  {
394  var describe = field.GetCustomAttribute<Describe>();
395  var terms = field.GetCustomAttribute<Terms>();
396  var prompt = field.GetCustomAttribute<Prompt>();
397  var optional = field.GetCustomAttribute<Optional>();
398  if (describe != null && !_ignoreAnnotations)
399  {
400  _description = describe.Description;
401  }
402  else
403  {
404  _description = Language.CamelCase(field.Name);
405  }
406  if (terms != null && !_ignoreAnnotations)

412  _terms = Language.GenerateTerms(field.Name, 3);
