323  public static string[] GenerateTerms(string name, int maxLength)
324  {
325  var phrase = CamelCase(name);
326  var words = (from word in phrase.Split(' ') select word.ToLower()).ToArray();
327  var terms = new List<string>();
328  for (var length = 1; length <= Math.Min(words.Length, maxLength); ++length)
329  {
330  for (var start = 0; start <= words.Length - length; ++start)
331  {
332  var ngram = new ArraySegment<string>(words, start, length);
333  if (!Ignorable(ngram.First()) && !Ignorable(ngram.Last()))
334  {
335  terms.Add(string.Join(" ", OptionalPlurals(ngram)));
336  }
337  }
338  }
339  if (words.Length > maxLength)
340  {
341  terms.Add(string.Join(" ", words));
342  }
343  return terms.ToArray();
344  }

346  public static string CamelCase(string name)
347  {
348  var builder = new StringBuilder();
349  var previousUpper = Char.IsUpper(name[0]);
350  var previousLetter = Char.IsLetter(name[0]);
351  bool first = true;
352  foreach (var ch in name)
353  {
354  if (!first && ch == '_')
355  {
356  // Non begin _ as space
357  builder.Append(' ');
358  }
359  else
360  {
361  if (first)
362  {
363  first = false;
364  }
365  var isUpper = Char.IsUpper(ch);
366  var isLetter = Char.IsLetter(ch);
367  if ((!previousUpper && isUpper) || (isLetter != previousLetter))
368  {
369  // Break on lower to upper or number boundaries
370  builder.Append(' ');
371  }
372  previousUpper = isUpper;
373  previousLetter = isLetter;
374  builder.Append(ch);
375  }
376  }
377  return builder.ToString();
378  }
