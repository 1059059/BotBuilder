98  public string Prompt(T state, string pathName, params object[] args)
99  {
100  string currentChoice = null;
101  string noValue = null;
102  if (pathName != "")
103  {
104  var field = _form.Fields.Field(pathName);
105  currentChoice = field.Template(TemplateUsage.CurrentChoice).Pattern();
106  if (field.Optional)
107  {
108  noValue = field.Template(TemplateUsage.NoPreference).Pattern();
109  }
110  else
111  {
112  noValue = field.Template(TemplateUsage.Unspecified).Pattern();
113  }
114  }
115  var response = ExpandTemplate(_annotation.Pattern(), currentChoice, noValue, state, pathName, args);
116  return (response == null ? "" : _spacesPunc.Replace(_spaces.Replace(Language.ANormalization(response), "$1 "), "$1"));
117  }

119  public static bool ValidatePattern(IForm<T> form, string pattern, string pathName, int argLimit = 0)
120  {
121  bool ok = true;
122  var fields = form.Fields;
123  foreach (Match match in _args.Matches(pattern))
124  {
125  var expr = match.Groups[1].Value.Trim();
126  int numeric;
127  if (expr == "||")
128  {
129  ok = true;
130  }
131  else if (expr.StartsWith("&"))
132  {
133  var name = expr.Substring(1);
134  if (name == "") name = pathName;
135  ok = (name == "" || fields.Field(name) != null);
136  }
137  else if (expr.StartsWith("?"))
138  {
139  ok = ValidatePattern(form, expr.Substring(1), pathName, argLimit);
140  }
141  else if (expr.StartsWith("["))
142  {
143  if (expr.EndsWith("]"))
144  {
145  ok = ValidatePattern(form, expr.Substring(1, expr.Length - 2), pathName, argLimit);
146  }
147  else
148  {
149  ok = false;
150  }
151  }
152  else if (expr.StartsWith("*"))
153  {
154  ok = (expr == "*" || expr == "*filled");
155  }
156  else if (TryParseFormat(expr, out numeric))
157  {
158  ok = numeric <= argLimit - 1;
159  }
160  else
161  {
162  var formatArgs = expr.Split(':');
163  var name = formatArgs[0];
164  if (name == "") name = pathName;
165  ok = (name == "" || fields.Field(name) != null);
166  }
167  if (!ok)
168  {
169  break;
170  }
171  }
172  return ok;
173  }

175  private string ExpandTemplate(string template, string currentChoice, string noValue, T state, string pathName, object[] args)
176  {
177  bool foundUnspecified = false;
178  int last = 0;
179  int numeric;
180  var response = new StringBuilder();
181  var field = _form.Fields.Field(pathName);
182  foreach (Match match in _args.Matches(template))
183  {
184  var expr = match.Groups[1].Value.Trim();
185  var substitute = "";
186  if (expr.StartsWith("&"))
187  {
188  var name = expr.Substring(1);
189  if (name == "") name = pathName;
190  var pathField = _form.Fields.Field(name);
191  substitute = Normalize(pathField == null ? pathName : pathField.FieldDescription, _annotation.FieldCase);
192  }
193  else if (expr == "||")
194  {
195  var builder = new StringBuilder();
196  var defaultValue = field.GetValue(state);
197  var values = _recognizer.ValueDescriptions();
198  if (_annotation.AllowDefault != BoolDefault.False)
199  {
200  if (!field.Optional)
201  {
202  if (!field.IsUnknown(state))
203  {
204  builder.Append(ExpandTemplate(currentChoice, null, noValue, state, pathName, args));
205  builder.Append(' ');
206  }
207  }
208  else
209  {
210  if (field.IsUnknown(state))
211  {
212  builder.Append(ExpandTemplate(currentChoice, null, noValue, state, pathName, args));
213  builder.Append(' ');
214  }
215  else
216  {
217  builder.Append(ExpandTemplate(currentChoice, null, noValue, state, pathName, args));
218  builder.Append(' ');
219  values = values.Concat(new string[] { noValue });
220  }
221  }
222  }
223  if (values.Count() > 0)
224  {
225  if ((_annotation.ChoiceStyle == ChoiceStyleOptions.Auto && values.Count() < 4)
226  || (_annotation.ChoiceStyle == ChoiceStyleOptions.Inline))
227  {
228  // Inline choices
229  bool first = true;
230  builder.Append('(');
231  var i = 1;
232  foreach (var value in values)
233  {
234  if (first)
235  {
236  first = false;
237  }
238  else
239  {
240  builder.Append(", ");
241  }
242  builder.AppendFormat(_annotation.ChoiceFormat, i, value);
243  ++i;
244  }
245  builder.Append(')');
246  }
247  else
248  {
249  // Separate line choices
250  var i = 1;
251  foreach (var value in values)
252  {
253  builder.Append("\n ");
254  if (!_annotation.AllowNumbers)
255  {
256  builder.Append("* ");
257  }
258  builder.AppendFormat(_annotation.ChoiceFormat, i, value);
259  ++i;
260  }
261  }
262  }
263  substitute = builder.ToString();
264  }
265  else if (expr.StartsWith("*"))
266  {
267  // Status display of active results
268  var filled = expr.ToLower().Trim().EndsWith("filled");
269  var builder = new StringBuilder();
270  var format = new Prompter<T>(Template(field, TemplateUsage.StatusFormat), _form, null);
271  if (match.Index > 0)
272  {
273  builder.Append("\n");
274  }
275  foreach (var entry in (from step in _form.Fields where (!filled || !step.IsUnknown(state)) && step.Role== FieldRole.Value && step.Active(state) select step))
276  {
277  builder.Append("* ").AppendLine(format.Prompt(state, entry.Name));
278  }
279  substitute = builder.ToString();
280  }
281  else if (expr.StartsWith("[") && expr.EndsWith("]"))
282  {
283  // Generate a list from multiple fields
284  var paths = expr.Substring(1, expr.Length - 2).Split(' ');
285  var values = new List<Tuple<IField<T>, object, string>>();
286  foreach (var spec in paths)
287  {
288  if (!spec.StartsWith("{") || !spec.EndsWith("}"))
289  {
290  throw new ArgumentException("Only {<field>} references are allowed in lists.");
291  }
292  var formatArgs = spec.Substring(1, spec.Length - 2).Trim().Split(':');
293  var name = formatArgs[0];
294  if (name == "") name = pathName;
295  var format = (formatArgs.Length > 1 ? "0:" + formatArgs[1] : "0");
296  var eltDesc = _form.Fields.Field(name);
297  if (!eltDesc.IsUnknown(state))
298  {
299  var value = eltDesc.GetValue(state);
300  if (value.GetType() != typeof(string) && value.GetType().IsIEnumerable())
301  {
302  var eltValues = (value as System.Collections.IEnumerable);
303  foreach (var elt in eltValues)
304  {
305  values.Add(Tuple.Create(eltDesc, elt, format));
306  }
307  }
308  else
309  {
310  values.Add(Tuple.Create(eltDesc, eltDesc.GetValue(state), format));
311  }
312  }
313  }
314  if (values.Count() > 0)
315  {
316  var elements = (from elt in values
317  select Normalize(ValueDescription(elt.Item1, elt.Item2, elt.Item3), _annotation.ValueCase)).ToArray();
318  substitute = Language.BuildList(elements, _annotation.Separator, _annotation.LastSeparator);
319  }
320  }
321  else if (expr.StartsWith("?"))
322  {
323  // Conditional template
324  var subValue = ExpandTemplate(expr.Substring(1), currentChoice, null, state, pathName, args);
325  if (subValue == null)
326  {
327  substitute = "";
328  }
329  else
330  {
331  substitute = subValue;
332  }
333  }
334  else if (TryParseFormat(expr, out numeric))
335  {
336  // Process ad hoc arg
337  if (numeric < args.Length && args[numeric] != null)
338  {
339  substitute = string.Format("{" + expr + "}", args);
340  }
341  else
342  {
343  foundUnspecified = true;
344  break;
345  }
346  }
347  else
348  {
349  var formatArgs = expr.Split(':');
350  var name = formatArgs[0];
351  if (name == "") name = pathName;
352  var pathDesc = _form.Fields.Field(name);
353  if (pathDesc.IsUnknown(state))
354  {
355  if (noValue == null)
356  {
357  foundUnspecified = true;
358  break;
359  }
360  substitute = noValue;
361  }
362  else
363  {
364  var value = pathDesc.GetValue(state);
365  if (value.GetType() != typeof(string) && value.GetType().IsIEnumerable())
366  {
367  var values = (value as System.Collections.IEnumerable);
368  substitute = Language.BuildList(from elt in values.Cast<object>()
369  select Normalize(ValueDescription(pathDesc, elt, "0"), _annotation.ValueCase),
370  _annotation.Separator, _annotation.LastSeparator);
371  }
372  else
373  {
374  var format = (formatArgs.Length > 1 ? "0:" + formatArgs[1] : "0");
375  substitute = ValueDescription(pathDesc, value, format);
376  }
377  }
378  }
379  response.Append(template.Substring(last, match.Index - last)).Append(substitute);
380  last = match.Index + match.Length;
381  }
382  return (foundUnspecified ? null : response.Append(template.Substring(last, template.Length - last)).ToString());
383  }

410  private string Normalize(string value, CaseNormalization normalization)
411  {
412  switch (normalization)
413  {
414  case CaseNormalization.InitialUpper:
415  string.Join(" ", (from word in Language.WordBreak(value)
416  select char.ToUpper(word[0]) + word.Substring(1).ToLower()));
417  break;
418  case CaseNormalization.Lower: value = value.ToLower(); break;
419  case CaseNormalization.Upper: value = value.ToUpper(); break;
420  }
421  return value;
422  }
