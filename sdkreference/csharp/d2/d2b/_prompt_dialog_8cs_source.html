248  public PromptChoice(IEnumerable<T> options, string prompt, string retry, int attempts)

254  public virtual bool IsMatch(T option, string text)
255  {
256  return option.ToString().IndexOf(text, StringComparison.CurrentCultureIgnoreCase) >= 0;
257  }
258 
259  protected override bool TryParse(Message message, out T result)
260  {
261  if (!string.IsNullOrWhiteSpace(message.Text))
262  {
263  var selected = this.options
264  .Where(option => IsMatch(option, message.Text))
265  .ToArray();
266  if (selected.Length == 1)
267  {
268  result = selected[0];
269  return true;
270  }
271  }
272 
273  result = default(T);
274  return false;
275  }
276  }

284  {

289  public Prompt(string prompt, string retry, int attempts)
290  {
291  SetField.NotNull(out this.prompt, nameof(prompt), prompt);
292  SetField.NotNull(out this.retry, nameof(retry), retry ?? prompt);
293  this.attempts = attempts;
294  }
295 
296  async Task IDialog<T>.StartAsync(IDialogContext context)
297  {
298  await context.PostAsync(this.prompt);
299  context.Wait(MessageReceived);
300  }
301 
302  private async Task MessageReceived(IDialogContext context, IAwaitable<Message> message)
303  {
304  T result;
305  if (this.TryParse(await message, out result))
306  {
307  context.Done(result);
308  }
309  else
310  {
311  --this.attempts;
312  if (this.attempts > 0)
313  {
314  var retry = this.retry ?? this.DefaultRetry;
315  await context.PostAsync(retry);
316  context.Wait(MessageReceived);
317  }
318  else
319  {
320  await context.PostAsync("too many attempts");
321  throw new Exception();
322  }
323  }
324  }
325 
326  protected abstract bool TryParse(Message message, out T result);
327

329  {
330  get
331  {
332  return this.prompt;
333  }
