128  private async Task MessageReceived(IDialogContext context, IAwaitable<Message> message)
129  {
130  T result;
131  if (this.TryParse(await message, out result))
132  {
133  context.Done(result);
134  }
135  else
136  {
137  --this.attempts;
138  if (this.attempts > 0)
139  {
140  var retry = this.retry ?? this.DefaultRetry;
141  await context.PostAsync(retry);
142  context.Wait(MessageReceived);
143  }
144  else
145  {
146  await context.PostAsync("too many attempts");
147  throw new Exception();
148  }
149  }
150  }

274  protected override bool TryParse(Message message, out T result)
275  {
276  if (!string.IsNullOrWhiteSpace(message.Text))
277  {
