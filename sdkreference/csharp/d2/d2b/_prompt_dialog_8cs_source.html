95  private async Task MessageReceived(IDialogContext context, IAwaitable<Message> message)
96  {
97  T result;
98  if (this.TryParse(await message, out result))
99  {
100  context.Done(result);
101  }
102  else
103  {
104  --this.attempts;
105  if (this.attempts > 0)
106  {
107  var retry = this.retry ?? this.DefaultRetry;
108  await context.PostAsync(retry);
109  context.Wait(MessageReceived);
110  }
111  else
112  {
113  await context.PostAsync("too many attempts");
114  throw new Exception();
115  }
116  }
117  }

241  protected override bool TryParse(Message message, out T result)
242  {

244  {
245  var selected = this.options
