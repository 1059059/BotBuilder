203  public virtual void Apply<T>(ref Message message, string prompt, IList<T> options)
204  {
205  SetField.CheckNull(nameof(prompt), prompt);
206  SetField.CheckNull(nameof(options), options);
207  switch (PromptStyle)
208  {
209  case PromptStyle.Auto:
210  message.Text = prompt;
211  message.AddButtons(options);
212  break;
213  case PromptStyle.AutoText:
214  Apply(ref message, prompt, options, options?.Count() > 4 ? PromptStyle.PerLine : PromptStyle.Inline);
215  break;
216  case PromptStyle.Inline:
217  //TODO: Refactor buildlist function to a more generic namespace when changing prompt to use recognizers.
218  message.Text = $"{prompt} {FormFlow.Advanced.Language.BuildList(options.Select(option => option.ToString()), Resources.DefaultChoiceSeparator, Resources.DefaultChoiceLastSeparator)}";
219  break;
220  case PromptStyle.PerLine:
221  message.Text = $"{prompt}\n{FormFlow.Advanced.Language.BuildList(options.Select(option => $"* {option.ToString()}"), "\n", "\n")}";
222  break;
223  case PromptStyle.None:
224  default:
225  message.Text = prompt;
226  break;
227  }
228  }

487  public virtual Tuple<bool, int> ScoreMatch(T option, string input)
488  {
489  var trimmed = input.Trim();
490  var text = option.ToString();
491  bool occurs = text.IndexOf(trimmed, StringComparison.CurrentCultureIgnoreCase) >= 0;
492  bool equals = text == trimmed;
493  return occurs
494  ? Tuple.Create(equals, trimmed.Length)
495  : null;
496  }
497 
498  protected override bool TryParse(Message message, out T result)
499  {
500  if (!string.IsNullOrWhiteSpace(message.Text))
501  {
502  var scores = from option in this.promptOptions.Options
503  let score = ScoreMatch(option, message.Text)
504  select new { score, option };
505 
506  var winner = scores.MaxBy(s => s.score);
507  if (winner.score != null)
508  {
509  result = winner.option;
510  return true;
511  }
512  }
513 
514  result = default(T);
515  return false;
516  }
517  }

581  private async Task MessageReceived(IDialogContext context, IAwaitable<Message> message)
582  {
583  T result;
584  if (this.TryParse(await message, out result))
585  {
586  context.Done(result);
587  }
588  else
589  {
590  --promptOptions.Attempts;
591  if (promptOptions.Attempts > 0)
592  {
593  await context.PostAsync(this.MakePrompt(context, promptOptions.Retry ?? promptOptions.DefaultRetry, promptOptions.Options));
594  context.Wait(MessageReceived);
595  }
596  else
597  {
598  //too many attempts, throw.
599  await context.PostAsync(this.MakePrompt(context, promptOptions.TooManyAttempts));
600  throw new TooManyAttemptsException(promptOptions.TooManyAttempts);
601  }
602  }
603  }
604 
605  protected abstract bool TryParse(Message message, out T result);
606 
607  protected virtual Message MakePrompt(IDialogContext context, string prompt, IList<U> options = null)
608  {
609  var msg = context.MakeMessage();
610  if (options != null && options.Count > 0)
611  {
612  promptOptions.PromptStyler.Apply(ref msg, prompt, options);
613  }
614  else
615  {
616  promptOptions.PromptStyler.Apply(ref msg, prompt);
617  }
618  return msg;
619  }
620  }
