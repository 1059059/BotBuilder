253  public override void SetValue(T state, object value)
254  {
255  object current = state;
256  object lastClass = state;
257  FieldInfo last = _path.Last();
258  foreach (var field in _path)
259  {
260  if (field == last)
261  {
262  if (field.FieldType.IsIEnumerable())
263  {
264  if (value == null)
265  {
266  field.SetValue(lastClass, null);
267  }
268  else if (field.FieldType == typeof(string))
269  {
270  field.SetValue(lastClass, value);
271  }
272  else
273  {
274  // Build list and coerce elements
275  var list = Activator.CreateInstance(field.FieldType);
276  var addMethod = list.GetType().GetMethod("Add");
277  foreach (var elt in value as System.Collections.IEnumerable)
278  {
279  addMethod.Invoke(list, new object[] { elt });
280  }
281  field.SetValue(lastClass, list);
282  }
283  }
284  else
285  {
286  if (value == null && field.FieldType.IsEnum)
287  {
288  field.SetValue(lastClass, 0);
289  }
290  else
291  {
292  field.SetValue(lastClass, value);
293  }
294  }
295  }
296  else
297  {
298  current = field.GetValue(current);
299  if (current == null)
300  {
301  var obj = Activator.CreateInstance(field.FieldType);
302  field.SetValue(lastClass, obj);
303  current = obj;
