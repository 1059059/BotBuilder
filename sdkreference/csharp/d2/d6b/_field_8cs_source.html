234  public virtual void SaveResources()
235  {
236  var localizer = _form.Resources;
237  if (_description.IsLocalizable)
238  {
239  localizer.Add(_name + nameof(_description), _description.Description);
240  }
241  if (_terms.IsLocalizable)
242  {
243  localizer.Add(_name + nameof(_terms), _terms.Alternatives);
244  }
245  localizer.Add(_name + nameof(_valueDescriptions), _valueDescriptions);
246  localizer.Add(_name + nameof(_valueTerms), _valueTerms);
247  if (_promptDefinition != null && _promptDefinition.IsLocalizable)
248  {
249  localizer.Add(_name + nameof(_promptDefinition), _promptDefinition.Patterns);
250  }
251  localizer.Add(_name, _templates);
252  }
253 
254  public virtual void Localize()
255  {
256  var localizer = _form.Resources;
257  string description;
258  string[] terms;
259  if (localizer.Lookup(_name + nameof(_description), out description))
260  {
261  _description = new DescribeAttribute(description);
262  }
263  if (localizer.LookupValues(_name + nameof(_terms), out terms))
264  {
265  _terms = new TermsAttribute(terms);
266  }
267  localizer.LookupDictionary(_name + nameof(_valueDescriptions), _valueDescriptions);
268  localizer.LookupDictionary(_name + nameof(_valueTerms), _valueTerms);
269  string[] patterns;
270  if (localizer.LookupValues(_name + nameof(_promptDefinition), out patterns))
271  {
272  _promptDefinition.Patterns = patterns;
273  }
274  localizer.LookupTemplates(_name, _templates);
275  if (!_promptSet)
276  {
277  _promptDefinition = null;
278  }
279  _prompt = null;
280  _recognizer = null;
281  if (_define == null)
282  {
283  DefinePrompt();
284  DefineRecognizer();
285  }
286  }

538  #region Internals
539  protected void DefinePrompt()
540  {
541  if (_promptDefinition == null)
542  {
543  TemplateUsage usage = TemplateUsage.None;
544  if (_type == null || _type.IsEnum)
545  {
546  usage = _allowsMultiple ? TemplateUsage.EnumSelectMany : TemplateUsage.EnumSelectOne;
547  }
548  else if (_type == typeof(string))
549  {
550  usage = TemplateUsage.String;
551  }
552  else if (_type.IsIntegral())
553  {
554  usage = TemplateUsage.Integer;
555  }
556  else if (_type == typeof(bool))
557  {
558  usage = TemplateUsage.Bool;
559  }
560  else if (_type.IsDouble())
561  {
562  usage = TemplateUsage.Double;
563  }
564  else if (_type == typeof(DateTime))
565  {
566  usage = TemplateUsage.DateTime;
567  }
568  else
569  {
570  throw new ArgumentException($"{_name} is not a type FormFlow understands.");
571  }
572  if (usage != TemplateUsage.None)
573  {
574  _promptDefinition = new PromptAttribute(Template(usage));
575  }
576  _promptSet = false;
577  }
578  _promptDefinition.ApplyDefaults(_form.Configuration.DefaultPrompt);
579  }
580 
581  protected void DefineRecognizer()
582  {
583  if (_recognizer == null)
584  {
585  if (_type == null || _type.IsEnum)
586  {
587  _recognizer = new RecognizeEnumeration<T>(this);
588  }
589  else if (_type == typeof(bool))
590  {
591  _recognizer = new RecognizeBool<T>(this);
592  }
593  else if (_type == typeof(string))
594  {
595  _recognizer = new RecognizeString<T>(this);
596  }
597  else if (_type.IsIntegral())
598  {
599  _recognizer = new RecognizeNumber<T>(this, Thread.CurrentThread.CurrentUICulture);
600  }
601  else if (_type.IsDouble())
602  {
603  _recognizer = new RecognizeDouble<T>(this, Thread.CurrentThread.CurrentUICulture);
604  }
605  else if (_type == typeof(DateTime))
606  {
607  _recognizer = new RecognizeDateTime<T>(this, Thread.CurrentThread.CurrentUICulture);
608  }
609  else if (_type.IsIEnumerable())
610  {
611  var elt = _type.GetGenericElementType();
612  if (elt.IsEnum)
613  {
614  _recognizer = new RecognizeEnumeration<T>(this);
615  }
616  }
617  var template = Template(TemplateUsage.Help);
618  _help = new Prompter<T>(template, _form, _recognizer);
619  var prompt = _promptDefinition;
620  _prompt = new Prompter<T>(_promptDefinition, _form, _recognizer);
621  }
622  }
