472  public override object GetValue(T state)
473  {
474  object current = state;
475  bool isEnum = false;
476  foreach (var step in _path)
477  {
478  var field = step as FieldInfo;
479  var ftype = StepType(step);
480  if (field != null)
481  {
482  current = field.GetValue(current);
483  }
484  else
485  {
486  var prop = step as PropertyInfo;
487  current = prop.GetValue(current);
488  }
489  if (current == null)
490  {
491  break;
492  }
493  isEnum = ftype.IsEnum;
494  }
495  return isEnum ? ((int)current == 0 ? null : current) : current;

498  public override void SetValue(T state, object value)
499  {
500  object current = state;
501  object lastClass = state;
502  var last = _path.Last();
503  foreach (var step in _path)
504  {
505  var field = step as FieldInfo;
506  var prop = step as PropertyInfo;
507  Type ftype = StepType(step);
508  if (step == last)
509  {
510  object newValue = value;
511  if (ftype.IsIEnumerable())
512  {
513  if (value != null && ftype != typeof(string))
514  {
515  // Build list and coerce elements
516  var list = Activator.CreateInstance(ftype);
517  var addMethod = list.GetType().GetMethod("Add");
518  foreach (var elt in value as System.Collections.IEnumerable)
519  {
520  addMethod.Invoke(list, new object[] { elt });
521  }
522  newValue = list;
523  }
524  }
525  else
526  {
527  if (value == null && (ftype.IsEnum || ftype.IsIntegral() || ftype.IsDouble()))
528  {
529  // Default value for numbers and enums
530  newValue = 0;
531  }
532  else if (ftype.IsIntegral())
533  {
534  newValue = Convert.ChangeType(value, ftype);
535  }
536  else if (ftype.IsDouble())
537  {
538  newValue = Convert.ChangeType(value, ftype);
539  }
540  else if (ftype == typeof(bool))
541  {
542  newValue = Convert.ChangeType(value, typeof(bool));
543  }
544  }
545  if (field != null)
546  {
547  field.SetValue(lastClass, newValue);
548  }
549  else
550  {
551  prop.SetValue(lastClass, newValue);
552  }
553  }
554  else
555  {
556  current = (field == null ? prop.GetValue(current) : field.GetValue(current));
557  if (current == null)
558  {
559  var obj = Activator.CreateInstance(ftype);
560  current = obj;
561  if (field != null)
562  {
563  field.SetValue(lastClass, current);
564  }
565  else
566  {
567  prop.SetValue(lastClass, current);
568  }
569  }
570  lastClass = current;
571  }
572  }

635  protected void AddField(Type type, string[] path, int ipath)
636  {
637  if (ipath < path.Length)
638  {
639  ProcessTemplates(type);
640  var step = path[ipath];
641  object field = type.GetField(step, BindingFlags.Public | BindingFlags.Instance);
642  Type ftype;
643  if (field == null)
644  {
645  var prop = type.GetProperty(step, BindingFlags.Public | BindingFlags.Instance);
646  if (prop == null)
647  {
648  throw new ArgumentException(step + " is not a field or property in your type");
649  }
650  field = prop;
651  ftype = prop.PropertyType;
652  _path.Add(prop);
653  }
654  else
655  {
656  ftype = (field as FieldInfo).FieldType;
657  _path.Add(field);
658  }
659  if (ftype.IsNullable())
660  {
661  _isNullable = true;
662  _keepZero = true;
663  ftype = Nullable.GetUnderlyingType(ftype);
664  }
665  else if (ftype.IsEnum || ftype.IsClass)
666  {
667  _isNullable = true;
668  }
669  if (ftype.IsClass)
670  {
671  if (ftype == typeof(string))
672  {
673  _type = ftype;
674  ProcessFieldAttributes(field);
675  }
676  else if (ftype.IsIEnumerable())
677  {
678  var elt = ftype.GetGenericElementType();
679  if (elt.IsEnum)
680  {
681  _type = elt;
682  _allowsMultiple = true;
683  ProcessFieldAttributes(field);
684  ProcessEnumAttributes(elt);
685  }
686  else
687  {
688  // TODO: What to do about enumerations of things other than enums?
689  throw new NotImplementedException();
690  }
691  }
692  else
693  {
694  AddField(ftype, path, ipath + 1);
695  }
696  }
697  else
698  {
699  if (ftype.IsEnum)
700  {
701  ProcessFieldAttributes(field);
702  ProcessEnumAttributes(ftype);
703  }
704  else if (ftype == typeof(bool))
705  {
706  ProcessFieldAttributes(field);
707  }
708  else if (ftype.IsIntegral())
709  {
710  long min = long.MinValue;
711  long max = long.MaxValue;
712  if (ftype == typeof(sbyte)) { min = sbyte.MinValue; max = sbyte.MaxValue; }
713  else if (ftype == typeof(byte)) { min = byte.MinValue; max = byte.MaxValue; }
714  else if (ftype == typeof(short)) { min = short.MinValue; max = short.MaxValue; }
715  else if (ftype == typeof(ushort)) { min = ushort.MinValue; max = ushort.MaxValue; }
716  else if (ftype == typeof(int)) { min = int.MinValue; max = int.MaxValue; }
717  else if (ftype == typeof(uint)) { min = uint.MinValue; max = uint.MaxValue; }
718  else if (ftype == typeof(long)) { min = long.MinValue; max = long.MaxValue; }
719  else if (ftype == typeof(ulong)) { min = long.MinValue; max = long.MaxValue; }
720  SetLimits(min, max, false);
721  ProcessFieldAttributes(field);
722  }
723  else if (ftype.IsDouble())
724  {
725  double min = long.MinValue;
726  double max = long.MaxValue;
727  if (ftype == typeof(float)) { min = float.MinValue; max = float.MaxValue; }
728  else if (ftype == typeof(double)) { min = double.MinValue; max = double.MaxValue; }
729  SetLimits(min, max, false);
730  ProcessFieldAttributes(field);
731  }
732  else if (ftype == typeof(DateTime))
733  {
734  ProcessFieldAttributes(field);
735  }
736  _type = ftype;
737  }
738  }

752  protected void ProcessFieldAttributes(object step)
753  {
754  _optional = false;
755  if (!_ignoreAnnotations)
756  {
757  var field = step as FieldInfo;
758  var prop = step as PropertyInfo;
759  var name = (field == null ? prop.Name : field.Name);
760  var describe = (field == null ? prop.GetCustomAttribute<Describe>() : field.GetCustomAttribute<Describe>());
761  var terms = (field == null ? prop.GetCustomAttribute<Terms>() : field.GetCustomAttribute<Terms>());
762  var prompt = (field == null ? prop.GetCustomAttribute<Prompt>() : field.GetCustomAttribute<Prompt>());
763  var optional = (field == null ? prop.GetCustomAttribute<Optional>() : field.GetCustomAttribute<Optional>());
764  var numeric = (field == null ? prop.GetCustomAttribute<Numeric>() : field.GetCustomAttribute<Numeric>());
765  if (describe != null)
766  {
767  _description = describe.Description;
768  }
769  else
770  {
771  _description = Language.CamelCase(name);
772  }
773  if (terms != null)
774  {
775  _terms = terms.Alternatives;
776  }
777  else
778  {
779  _terms = Language.GenerateTerms(Language.CamelCase(name), 3);
780  }
781  if (prompt != null)
782  {
783  _promptDefinition = prompt;
784  }
785  if (numeric != null)
786  {
787  double oldMin, oldMax;
788  Limits(out oldMin, out oldMax);
789  SetLimits(numeric.Min, numeric.Max, numeric.Min != oldMin || numeric.Max != oldMax);
790  }
791  _optional = (optional != null);
792  foreach (var attribute in (field == null ? prop.GetCustomAttributes<Template>() : field.GetCustomAttributes<Template>()))
793  {
794  AddTemplate(attribute as Template);
795  }
796  }
