56  private static IEnumerable<object> DoNotSerialize(IComponentContext context, IEnumerable<Parameter> parameters)
57  {
58  foreach (var registration in context.ComponentRegistry.Registrations)
59  {
60  foreach (var service in registration.Services)
61  {
62  var keyed = service as KeyedService;
63  if (keyed != null)
64  {
65  if (keyed.ServiceKey == Key_DoNotSerialize)
66  {
67  object instance;
68  if (context.TryResolveService(service, parameters, out instance))
69  {
70  yield return instance;
71  }
72  }
73  }
74  }
75  }
76  }
