662  internal bool AllowNumbers
663  {
664  get
665  {
666  // You can match on numbers only if they are included in Choices and choices are shown
667  return ChoiceFormat.Contains("{0}") && Patterns.Any((pattern) => pattern.Contains("{||}"));
668  }
669  }
670 
676  public string Pattern()
677  {
678  var choice = 0;
679  if (Patterns.Length > 1)
680  {
681  lock (_generator)
682  {
683  choice = _generator.Next(Patterns.Length);
684  }
685  }
686  return Patterns[choice];
687  }
688 
693  public string[] Patterns { get; set; }
694 
699  public void ApplyDefaults(TemplateBaseAttribute defaultTemplate)
700  {
701  if (AllowDefault == BoolDefault.Default) AllowDefault = defaultTemplate.AllowDefault;
702  if (ChoiceCase == CaseNormalization.Default) ChoiceCase = defaultTemplate.ChoiceCase;
703  if (ChoiceFormat == null) ChoiceFormat = defaultTemplate.ChoiceFormat;
704  if (ChoiceLastSeparator == null) ChoiceLastSeparator = defaultTemplate.ChoiceLastSeparator;
705  if (ChoiceParens == BoolDefault.Default) ChoiceParens = defaultTemplate.ChoiceParens;
706  if (ChoiceSeparator == null) ChoiceSeparator = defaultTemplate.ChoiceSeparator;
707  if (ChoiceStyle == ChoiceStyleOptions.Default) ChoiceStyle = defaultTemplate.ChoiceStyle;
708  if (FieldCase == CaseNormalization.Default) FieldCase = defaultTemplate.FieldCase;
709  if (Feedback == FeedbackOptions.Default) Feedback = defaultTemplate.Feedback;
710  if (LastSeparator == null) LastSeparator = defaultTemplate.LastSeparator;
711  if (Separator == null) Separator = defaultTemplate.Separator;
712  if (ValueCase == CaseNormalization.Default) ValueCase = defaultTemplate.ValueCase;
713  }
714 
719  public TemplateBaseAttribute(params string[] patterns)
720  {

723  }
