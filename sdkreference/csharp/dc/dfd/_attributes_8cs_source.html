555  public abstract class TemplateBaseAttribute : Attribute
556  {

559 
563  public BoolDefault AllowDefault { get; set; }

569  public string ChoiceFormat { get; set; }

579  public FeedbackOptions Feedback { get; set; }

584  public CaseNormalization FieldCase { get; set; }

594  public string Separator { get; set; }
595 
599  public CaseNormalization ValueCase { get; set; }
600 
601  internal bool AllowNumbers
602  {
603  get
604  {
605  // You can match on numbers only if they are included in Choices and choices are shown
606  return ChoiceFormat.Contains("{0}") && _patterns.Any((pattern) => pattern.Contains("{||}"));
607  }
608  }
609 
615  public string Pattern()
616  {
617  var choice = 0;
618  if (_patterns.Length > 1)
619  {
620  lock (_generator)
621  {
622  choice = _generator.Next(_patterns.Length);
623  }
624  }
625  return _patterns[choice];
626  }
627 
632  public string[] Patterns
633  {
634  get
635  {
636  return _patterns;
637  }
638  }
639 
644  public void ApplyDefaults(TemplateBaseAttribute defaultTemplate)
645  {
646  if (AllowDefault == BoolDefault.Default) AllowDefault = defaultTemplate.AllowDefault;
647  if (ChoiceStyle == ChoiceStyleOptions.Default) ChoiceStyle = defaultTemplate.ChoiceStyle;
648  if (FieldCase == CaseNormalization.Default) FieldCase = defaultTemplate.FieldCase;
649  if (Feedback == FeedbackOptions.Default) Feedback = defaultTemplate.Feedback;
650  if (ChoiceFormat == null) ChoiceFormat = defaultTemplate.ChoiceFormat;
651  if (LastSeparator == null) LastSeparator = defaultTemplate.LastSeparator;
652  if (Separator == null) Separator = defaultTemplate.Separator;
653  if (ValueCase == CaseNormalization.Default) ValueCase = defaultTemplate.ValueCase;
654  }
655 
660  public TemplateBaseAttribute(params string[] patterns)
661  {
662  _patterns = patterns;
663  Initialize();
664  }
665 
670  public TemplateBaseAttribute(TemplateBaseAttribute other)
671  {
672  _patterns = other._patterns;
673  AllowDefault = other.AllowDefault;
674  ChoiceStyle = other.ChoiceStyle;
675  FieldCase = other.FieldCase;
676  Feedback = other.Feedback;
677  ChoiceFormat = other.ChoiceFormat;
678  LastSeparator = other.LastSeparator;
679  Separator = other.Separator;
680  ValueCase = other.ValueCase;
681  }
682 
683  private void Initialize()
684  {
685  AllowDefault = BoolDefault.Default;
686  ChoiceStyle = ChoiceStyleOptions.Default;
687  FieldCase = CaseNormalization.Default;
688  Feedback = FeedbackOptions.Default;
689  ChoiceFormat = null;
690  LastSeparator = null;
691  Separator = null;

693  }
694  }
