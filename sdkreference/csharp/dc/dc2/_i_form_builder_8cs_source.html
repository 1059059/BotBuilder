350  public Template Template(TemplateUsage usage)
351  {
352  Template result = null;
353  foreach (var template in Templates)
354  {
355  if (template.Usage == usage)
356  {
357  result = template;
358  break;
359  }
360  }
361  Debug.Assert(result != null);
362  return result;
363  }
