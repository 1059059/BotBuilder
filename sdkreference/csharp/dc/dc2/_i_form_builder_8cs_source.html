392  public TemplateAttribute Template(TemplateUsage usage)
393  {
394  TemplateAttribute result = null;
395  foreach (var template in Templates)
396  {
397  if (template.Usage == usage)
398  {
399  result = template;
400  break;
401  }
402  }
403  Debug.Assert(result != null);
404  return result;
405  }
