372  public TemplateAttribute Template(TemplateUsage usage)
373  {
374  TemplateAttribute result = null;
375  foreach (var template in Templates)
376  {
377  if (template.Usage == usage)
378  {
379  result = template;
380  break;
381  }
382  }
383  Debug.Assert(result != null);
384  return result;
385  }
