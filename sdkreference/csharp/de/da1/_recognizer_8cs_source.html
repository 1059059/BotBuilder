101  public IEnumerable<TermMatch> Matches(string input, object defaultValue)
102  {
103  // if the user hit enter on an optional prompt, then consider taking the current choice as a low confidence option
104  bool userSkippedPrompt = string.IsNullOrWhiteSpace(input) && (defaultValue != null || _noPreference != null);
105  if (userSkippedPrompt)
106  {
107  yield return new TermMatch(0, input.Length, 1.0, defaultValue);
108  }
109 
110  foreach (var expression in _expressions)
111  {
112  double longest = expression.Longest.Length;
113  foreach (Match match in expression.Expression.Matches(input))
114  {
115  var group1 = match.Groups[1];
116  var group2 = match.Groups[2];
117  if (group1.Success)
118  {
119  var confidence = System.Math.Min(group1.Length / longest, 1.0);
120  if (expression.Value is Special)
121  {
122  var special = (Special)expression.Value;
123  if (special == Special.CurrentChoice && (_noPreference != null || defaultValue != null))
124  {
125  yield return new TermMatch(group1.Index, group1.Length, confidence, defaultValue);
126  }
127  else if (special == Special.NoPreference)
128  {
129  yield return new TermMatch(group1.Index, group1.Length, confidence, null);
130  }
131  }
132  else
133  {
134  yield return new TermMatch(group1.Index, group1.Length, confidence, expression.Value);
135  }
136  }
137  else if (group2.Success)
138  {
139  yield return new TermMatch(group2.Index, group2.Length, 1.0, expression.Value);
140  }
141  }
