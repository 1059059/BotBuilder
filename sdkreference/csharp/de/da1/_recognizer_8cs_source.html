138  public IEnumerable<TermMatch> Matches(string input, object defaultValue)
139  {
140  // if the user hit enter on an optional prompt, then consider taking the current choice as a low confidence option
141  bool userSkippedPrompt = string.IsNullOrWhiteSpace(input) && (defaultValue != null || _noPreference != null);
142  if (userSkippedPrompt)
143  {
144  yield return new TermMatch(0, input.Length, 1.0, defaultValue);
145  }
146 
147  foreach (var expression in _expressions)
148  {
149  double maxWords = expression.MaxWords;
150  foreach (Match match in expression.Expression.Matches(input))
151  {
152  var group1 = match.Groups[1];
153  var group2 = match.Groups[2];
154  if (group1.Success)
155  {
156  int n;
157  var words = group1.Value.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries).Length;
158  var confidence = int.TryParse(group1.Value, out n) ? 1.0 : System.Math.Min(words / maxWords, 1.0);
159  if (expression.Value is Special)
160  {
161  var special = (Special)expression.Value;
162  if (special == Special.CurrentChoice && (_noPreference != null || defaultValue != null))
163  {
164  yield return new TermMatch(group1.Index, group1.Length, confidence, defaultValue);
165  }
166  else if (special == Special.NoPreference)
167  {
168  yield return new TermMatch(group1.Index, group1.Length, confidence, null);
169  }
170  }
171  else
172  {
173  yield return new TermMatch(group1.Index, group1.Length, confidence, expression.Value);
174  }
175  }
176  else if (group2.Success)
177  {
178  yield return new TermMatch(group2.Index, group2.Length, 1.0, expression.Value);
179  }
180  }
181  }
182  }

198  private enum Special { CurrentChoice, NoPreference };

200  // Word character, any word character, any digit, any positive group over word characters
201  private const string WORD = @"(\w|\\w|\\d|(\[(?>(\w|-)+|\[(?<number>)|\](?<-number>))*(?(number)(?!))\]))";
202  private static Regex _wordStart = new Regex(string.Format(@"^{0}|\(", WORD), RegexOptions.Compiled);
203  private static Regex _wordEnd = new Regex(string.Format(@"({0}|\))(\?|\*|\+|\{{\d+\}}|\{{,\d+\}}|\{{\d+,\d+\}})?$", WORD), RegexOptions.Compiled);

205  private void BuildPerValueMatcher(IEnumerable<string> currentChoice)
206  {
207  if (currentChoice != null)
208  {
209  // 0 is reserved for current default if any
210  AddExpression(0, Special.CurrentChoice, currentChoice, _allowNumbers);
211  }
212  var n = 1;
213  foreach (var value in _values)
214  {
215  n = AddExpression(n, value, _termsDelegate(value), _allowNumbers);
216  }
217  if (_noPreference != null)
218  {
219  // Add recognizer for no preference
220  n = AddExpression(n, Special.NoPreference, _noPreference, _allowNumbers);
221  }
222  if (_terms != null && _terms.Count() > 0)
223  {
224  // Add field terms to help disambiguate
225  AddExpression(n, SpecialValues.Field, _terms, false);
226  }
227  _max = n - 1;
228  }

230  private int NumberOfWords(string regex)
231  {

235  private int AddExpression(int n, object value, IEnumerable<string> terms, bool allowNumbers)
236  {
237  var orderedTerms = (from term in terms orderby term.Length descending select term).ToArray();
238  if (orderedTerms.Length > 0)
239  {
240  var maxWords = terms.Max((term) => NumberOfWords(term));
241  var word = new StringBuilder();
242  var nonWord = new StringBuilder();
243  var first = true;
244  var firstNonWord = true;
245  foreach (var term in orderedTerms)
246  {
247  var nterm = term.Trim().Replace(" ", @"\s+");
248  if (nterm == "") nterm = "qqqq";
249  if (_wordStart.Match(nterm).Success && _wordEnd.Match(nterm).Success)
250  {
251  if (first)
252  {
253  first = false;
254  word.Append(@"(\b(?:");
255  }
256  else
257  {
258  word.Append('|');
259  }
260  word.Append(@"(?:");
261  word.Append(nterm);
262  word.Append(')');
263  }
264  else
265  {
266  if (firstNonWord)
267  {
268  firstNonWord = false;
269  nonWord.Append('(');
270  }
271  else
272  {
273  nonWord.Append('|');
274  }
275  nonWord.Append(@"(?:");
276  nonWord.Append(nterm);
277  nonWord.Append(')');
278  }
279  }
280  if (first)
281  {
282  word.Append("(qqqq)");
283  }
284  else
285  {
286  if (n == 0)
287  {
288  word.Append("|c");
289  }
290  else if (allowNumbers)
291  {
292  word.AppendFormat(@"|{0}", n);
293  }
294  word.Append(@")\b)");
295  }
296  if (firstNonWord)
297  {
298  nonWord.Append("(qqqq)");
299  }
300  else
301  {
302  nonWord.Append(')');
303  }
304  ++n;
305  var expr = string.Format("{0}|{1}",
306  word.ToString(),
307  nonWord.ToString());
308  _expressions.Add(new ValueAndExpression(value, new Regex(expr, RegexOptions.IgnoreCase), maxWords));
309  }
310  return n;
311  }
