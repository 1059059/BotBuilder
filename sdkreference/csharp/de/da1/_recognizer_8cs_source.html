90  public IEnumerable<TermMatch> Matches(string input, object defaultValue)
91  {
92  foreach (var expression in _expressions)
93  {
94  double longest = expression.Longest.Length;
95  foreach (Match match in expression.Expression.Matches(input))
96  {
97  var group1 = match.Groups[1];
98  var group2 = match.Groups[2];
99  if (group1.Success)
100  {
101  var confidence = System.Math.Min(group1.Length / longest, 1.0);
102  if (expression.Value is Special)
103  {
104  var special = (Special)expression.Value;
105  if (special == Special.CurrentChoice && (_noPreference != null || defaultValue != null))
106  {
107  yield return new TermMatch(group1.Index, group1.Length, confidence, defaultValue);
108  }
109  else if (special == Special.NoPreference && defaultValue != null)
110  {
111  yield return new TermMatch(group1.Index, group1.Length, confidence, null);
112  }
113  }
114  else
115  {
116  yield return new TermMatch(group1.Index, group1.Length, confidence, expression.Value);
117  }
118  }
119  else if (group2.Success)
120  {
121  yield return new TermMatch(group2.Index, group2.Length, 1.0, expression.Value);
122  }
123  }

173  private int AddExpression(int n, object value, IEnumerable<string> terms, bool allowNumbers)
174  {
175  var orderedTerms = (from term in terms orderby term.Length descending select term).ToArray();
176  var word = new StringBuilder();
177  var nonWord = new StringBuilder();
178  var first = true;
179  var firstNonWord = true;
180  foreach (var term in orderedTerms)
181  {
182  var nterm = term.Trim().Replace(" ", @"\s+");
183  if (nterm == "") nterm = "qqqq";
184  if (_wordStart.Match(nterm).Success && _wordEnd.Match(nterm).Success)
185  {
186  if (first)
187  {
188  first = false;
189  word.Append(@"(\b(?:");
190  }
191  else
192  {
193  word.Append('|');
194  }
195  word.Append(@"(?:");
196  word.Append(nterm);
197  word.Append(')');
198  }
199  else
200  {
201  if (firstNonWord)
202  {
203  firstNonWord = false;
204  nonWord.Append('(');
205  }
206  else
207  {
208  nonWord.Append('|');
209  }
210  nonWord.Append(@"(?:");
211  nonWord.Append(nterm);
212  nonWord.Append(')');
213  }
214  }
215  if (first)
216  {
217  word.Append("(qqqq)");
218  }
219  else
220  {
221  if (allowNumbers)
222  {
223  if (n == 0)
224  {
225  word.Append("|c");
226  }
227  else
228  {
229  word.AppendFormat(@"|{0}", n);
230  }
231  }
232  word.Append(@")\b)");
233  }
234  if (firstNonWord)
235  {
236  nonWord.Append("(qqqq)");
237  }
238  else
239  {
240  nonWord.Append(')');
241  }
242  ++n;
243  var expr = string.Format("{0}|{1}",
244  word.ToString(),
245  nonWord.ToString());
246  _expressions.Add(new ValueAndExpression(value, new Regex(expr, RegexOptions.IgnoreCase), orderedTerms.First()));
247  return n;
