142  public IEnumerable<TermMatch> Matches(string input, object defaultValue)
143  {
144  // if the user hit enter on an optional prompt, then consider taking the current choice as a low confidence option
145  bool userSkippedPrompt = string.IsNullOrWhiteSpace(input) && (defaultValue != null || _noPreference != null);
146  if (userSkippedPrompt)
147  {
148  yield return new TermMatch(0, input.Length, 1.0, defaultValue);
149  }
150 
151  foreach (var expression in _expressions)
152  {
153  double maxWords = expression.MaxWords;
154  foreach (Match match in expression.Expression.Matches(input))
155  {
156  var group1 = match.Groups[1];
157  var group2 = match.Groups[2];
158  if (group1.Success)
159  {
160  int n;
161  var words = group1.Value.Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries).Length;
162  var confidence = int.TryParse(group1.Value, out n) ? 1.0 : System.Math.Min(words / maxWords, 1.0);
163  if (expression.Value is Special)
164  {
165  var special = (Special)expression.Value;
166  if (special == Special.CurrentChoice && (_noPreference != null || defaultValue != null))
167  {
168  yield return new TermMatch(group1.Index, group1.Length, confidence, defaultValue);
169  }
170  else if (special == Special.NoPreference)
171  {
172  yield return new TermMatch(group1.Index, group1.Length, confidence, null);
173  }
174  }
175  else
176  {
177  yield return new TermMatch(group1.Index, group1.Length, confidence, expression.Value);
178  }
179  }
180  else if (group2.Success)
181  {
182  yield return new TermMatch(group2.Index, group2.Length, 1.0, expression.Value);
183  }
184  }
185  }
186  }

202  private enum Special { CurrentChoice, NoPreference };

204  // Word character, any word character, any digit, any positive group over word characters
205  private const string WORD = @"(\w|\\w|\\d|(\[(?>(\w|-)+|\[(?<number>)|\](?<-number>))*(?(number)(?!))\]))";
206  private static Regex _wordStart = new Regex(string.Format(@"^{0}|\(", WORD), RegexOptions.Compiled);
207  private static Regex _wordEnd = new Regex(string.Format(@"({0}|\))(\?|\*|\+|\{{\d+\}}|\{{,\d+\}}|\{{\d+,\d+\}})?$", WORD), RegexOptions.Compiled);

209  private void BuildPerValueMatcher(IEnumerable<string> currentChoice)
210  {
211  if (currentChoice != null)
212  {
213  // 0 is reserved for current default if any
214  AddExpression(0, Special.CurrentChoice, currentChoice, _allowNumbers);
215  }
216  var n = 1;
217  foreach (var value in _values)
218  {
219  n = AddExpression(n, value, _termsDelegate(value), _allowNumbers);
220  }
221  if (_noPreference != null)
222  {
223  // Add recognizer for no preference
224  n = AddExpression(n, Special.NoPreference, _noPreference, _allowNumbers);
225  }
226  if (_terms != null && _terms.Count() > 0)
227  {
228  // Add field terms to help disambiguate
229  AddExpression(n, SpecialValues.Field, _terms, false);
230  }
231  _max = n - 1;
232  }

234  private int NumberOfWords(string regex)
235  {
236  return regex.Split(new string[] { @"\s", " " }, StringSplitOptions.RemoveEmptyEntries).Length;

239  private int AddExpression(int n, object value, IEnumerable<string> terms, bool allowNumbers)
240  {
241  var orderedTerms = (from term in terms orderby term.Length descending select term).ToArray();
242  if (orderedTerms.Length > 0)
243  {
244  var maxWords = terms.Max((term) => NumberOfWords(term));
245  var word = new StringBuilder();
246  var nonWord = new StringBuilder();
247  var first = true;
248  var firstNonWord = true;
249  foreach (var term in orderedTerms)
250  {
251  var nterm = term.Trim().Replace(" ", @"\s+");
252  if (nterm == "") nterm = "qqqq";
253  if (_wordStart.Match(nterm).Success && _wordEnd.Match(nterm).Success)
254  {
255  if (first)
256  {
257  first = false;
258  word.Append(@"(\b(?:");
259  }
260  else
261  {
262  word.Append('|');
263  }
264  word.Append(@"(?:");
265  word.Append(nterm);
266  word.Append(')');
267  }
268  else
269  {
270  if (firstNonWord)
271  {
272  firstNonWord = false;
273  nonWord.Append('(');
274  }
275  else
276  {
277  nonWord.Append('|');
278  }
279  nonWord.Append(@"(?:");
280  nonWord.Append(nterm);
281  nonWord.Append(')');
282  }
283  }
284  if (first)
285  {
286  word.Append("(qqqq)");
287  }
288  else
289  {
290  if (n == 0)
291  {
292  word.Append("|c");
293  }
294  else if (allowNumbers)
295  {
296  word.AppendFormat(@"|{0}", n);
297  }
298  word.Append(@")\b)");
299  }
300  if (firstNonWord)
301  {
302  nonWord.Append("(qqqq)");
303  }
304  else
305  {
306  nonWord.Append(')');
307  }
308  ++n;
309  var expr = string.Format("{0}|{1}",
310  word.ToString(),
311  nonWord.ToString());
312  _expressions.Add(new ValueAndExpression(value, new Regex(expr, RegexOptions.IgnoreCase), maxWords));
313  }
314  return n;
315  }
